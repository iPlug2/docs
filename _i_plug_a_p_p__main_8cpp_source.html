31 int WINAPI WinMain(HINSTANCE hInstance, HINSTANCE hPrevInstance, LPSTR lpszCmdParam, int nShowCmd)
32 {
33  try
34  {
35  HANDLE hMutex = OpenMutex(MUTEX_ALL_ACCESS, 0, BUNDLE_NAME); // BUNDLE_NAME used because it won't have spaces in it
36 
37  if (!hMutex)
38  hMutex = CreateMutex(0, 0, BUNDLE_NAME);
39  else
40  {
41  HWND hWnd = FindWindow(0, BUNDLE_NAME);
42  SetForegroundWindow(hWnd);
43  return 0; // should return 1?
44  }
45 
46  gHINSTANCE = hInstance;
47 
48  InitCommonControls();
49  gScrollMessage = RegisterWindowMessage("MSWHEEL_ROLLMSG");
50 
51  IPlugAPPHost* pAppHost = IPlugAPPHost::Create();
52  pAppHost->Init();
53  pAppHost->TryToChangeAudio();
54 
55  HACCEL hAccel = LoadAccelerators(gHINSTANCE, MAKEINTRESOURCE(IDR_ACCELERATOR1));
56 
57  CreateDialog(gHINSTANCE, MAKEINTRESOURCE(IDD_DIALOG_MAIN), GetDesktopWindow(), IPlugAPPHost::MainDlgProc);
58 
59 #ifndef _DEBUG
60  HMENU menu = GetMenu(gHWND);
61  RemoveMenu(menu, 1, MF_BYPOSITION);
62  DrawMenuBar(gHWND);
63 #endif
64 
65 
66  for(;;)
67  {
68  MSG msg= {0,};
69  int vvv = GetMessage(&msg, NULL, 0, 0);
70 
71  if (!vvv)
72  break;
73 
74  if (vvv<0)
75  {
76  Sleep(10);
77  continue;
78  }
79 
80  if (!msg.hwnd)
81  {
82  DispatchMessage(&msg);
83  continue;
84  }
85 
86  if (gHWND && IsDialogMessage(gHWND, &msg)) continue;
87 
88  // default processing for other dialogs
89  HWND hWndParent = NULL;
90  HWND temphwnd = msg.hwnd;
91 
92  do
93  {
94  if (GetClassLong(temphwnd, GCW_ATOM) == (INT)32770)
95  {
96  hWndParent=temphwnd;
97  if (!(GetWindowLong(temphwnd, GWL_STYLE) &WS_CHILD))
98  break; // not a child, exit
99  }
100  }
101  while (temphwnd = GetParent(temphwnd));
102 
103  if (hWndParent && IsDialogMessage(hWndParent,&msg)) continue;
104 
105  if (!TranslateAccelerator(gHWND, hAccel, &msg))
106  {
107  TranslateMessage(&msg);
108  DispatchMessage(&msg);
109  }
110  }
111 
112  // in case gHWND didnt get destroyed -- this corresponds to SWELLAPP_DESTROY roughly
113  if (gHWND)
114  DestroyWindow(gHWND);
115 
116  ReleaseMutex(hMutex);
117  }
118  catch(...)
119  {
120  DBGMSG("another instance running\n");
121  }
122  return 0;
123 }

152 INT_PTR SWELLAppMain(int msg, INT_PTR parm1, INT_PTR parm2)
153 {
154  IPlugAPPHost* pAppHost = nullptr;
155 
156  switch (msg)
157  {
158  case SWELLAPP_ONLOAD:
159  pAppHost = IPlugAPPHost::Create();
160  pAppHost->Init();
161  pAppHost->TryToChangeAudio();
162  break;
163  case SWELLAPP_LOADED:
164  {
165  pAppHost = IPlugAPPHost::sInstance;
166 
167  HMENU menu = SWELL_GetCurrentMenu();
168 
169  if (menu)
170  {
171  // work on a new menu
172  menu = SWELL_DuplicateMenu(menu);
173  HMENU src = LoadMenu(NULL, MAKEINTRESOURCE(IDR_MENU1));
174 
175  for (int x = 0; x < GetMenuItemCount(src)-1; x++)
176  {
177  HMENU sm = GetSubMenu(src,x);
178 
179  if (sm)
180  {
181  char str[1024];
182  MENUITEMINFO mii = {sizeof(mii), MIIM_TYPE};
183  mii.dwTypeData = str;
184  mii.cch = sizeof(str);
185  str[0] = 0;
186  GetMenuItemInfo(src, x, TRUE, &mii);
187  MENUITEMINFO mi= {sizeof(mi), MIIM_STATE|MIIM_SUBMENU|MIIM_TYPE,MFT_STRING, 0, 0, SWELL_DuplicateMenu(sm), NULL, NULL, 0, str};
188  InsertMenuItem(menu, x+1, TRUE, &mi);
189  }
190  }
191  }
192 
193  if (menu)
194  {
195  HMENU sm = GetSubMenu(menu, 1);
196  DeleteMenu(sm, ID_QUIT, MF_BYCOMMAND); // remove QUIT from our file menu, since it is in the system menu on OSX
197  DeleteMenu(sm, ID_PREFERENCES, MF_BYCOMMAND); // remove PREFERENCES from the file menu, since it is in the system menu on OSX
198 
199  // remove any trailing separators
200  int a = GetMenuItemCount(sm);
201 
202  while (a > 0 && GetMenuItemID(sm, a-1) == 0)
203  DeleteMenu(sm, --a, MF_BYPOSITION);
204 
205  DeleteMenu(menu, 1, MF_BYPOSITION); // delete file menu
206  }
207 #ifndef _DEBUG
208  if (menu)
209  {
210  HMENU sm = GetSubMenu(menu, 1);
211  DeleteMenu(sm, ID_LIVE_EDIT, MF_BYCOMMAND);
212  DeleteMenu(sm, ID_SHOW_DRAWN, MF_BYCOMMAND);
213  DeleteMenu(sm, ID_SHOW_FPS, MF_BYCOMMAND);
214 
215  // remove any trailing separators
216  int a = GetMenuItemCount(sm);
217 
218  while (a > 0 && GetMenuItemID(sm, a-1) == 0)
219  DeleteMenu(sm, --a, MF_BYPOSITION);
220 
221  DeleteMenu(menu, 1, MF_BYPOSITION); // delete debug menu
222  }
223 #else
224  SetMenuItemModifier(menu, ID_LIVE_EDIT, MF_BYCOMMAND, 'E', FCONTROL);
225  SetMenuItemModifier(menu, ID_SHOW_DRAWN, MF_BYCOMMAND, 'D', FCONTROL);
226  SetMenuItemModifier(menu, ID_SHOW_BOUNDS, MF_BYCOMMAND, 'B', FCONTROL);
227  SetMenuItemModifier(menu, ID_SHOW_FPS, MF_BYCOMMAND, 'F', FCONTROL);
228 #endif
229 
230  HWND hwnd = CreateDialog(gHINST, MAKEINTRESOURCE(IDD_DIALOG_MAIN), NULL, IPlugAPPHost::MainDlgProc);
231 
232  if (menu)
233  {
234  SetMenu(hwnd, menu); // set the menu for the dialog to our menu (on Windows that menu is set from the .rc, but on SWELL
235  SWELL_SetDefaultModalWindowMenu(menu); // other windows will get the stock (bundle) menus
236  }
237 
238  break;
239  }
240  case SWELLAPP_ONCOMMAND:
241  // this is to catch commands coming from the system menu etc
242  if (gHWND && (parm1&0xffff))
243  SendMessage(gHWND, WM_COMMAND, parm1 & 0xffff, 0);
244  break;
245  case SWELLAPP_DESTROY:
246  if (gHWND)
247  DestroyWindow(gHWND);
248  break;
249  case SWELLAPP_PROCESSMESSAGE:
250  MSG* pMSG = (MSG*) parm1;
251  NSView* pContentView = (NSView*) pMSG->hwnd;
252  NSEvent* pEvent = (NSEvent*) parm2;
253  int etype = [pEvent type];
254 
255  bool textField = [pContentView isKindOfClass:[NSText class]];
256 
257  if (!textField && etype == NSKeyDown)
258  {
259  int flag, code = SWELL_MacKeyToWindowsKey(pEvent, &flag);
260 
261  if (!(flag&~FVIRTKEY) && (code == VK_RETURN || code == VK_ESCAPE))
262  {
263  [pContentView keyDown: pEvent];
264  return 1;
265  }
266  }
267  break;
268  }
269  return 0;
270 }
