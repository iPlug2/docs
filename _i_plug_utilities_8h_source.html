125static inline void GetVersionParts(int versionInteger, int& maj, int& min, int& pat)
126{
127 maj = (versionInteger & 0xFFFF0000) >> 16;
128 min = (versionInteger & 0x0000FF00) >> 8;
129 pat = versionInteger & 0x000000FF;
130}
131
135static inline int GetDecimalVersion(int versionInteger)
136{
137 int maj, min, pat;
138 GetVersionParts(versionInteger, maj, min, pat);
139 return 10000 * maj + 100 * min + pat;
140}

170static void ToLower(char* cDest, const char* cSrc)
171{
172 int i, n = (int)strlen(cSrc);
173 for (i = 0; i < n; ++i)
174 {
175 cDest[i] = tolower(cSrc[i]);
176 }
177 cDest[i] = '\0';

352static void MidiNoteName(double midiPitch, WDL_String& noteName, bool cents = false, bool middleCisC4 = false)
353{
354 static const char noteNames[12][3] = {"C ", "C#", "D ", "D#", "E ", "F ", "F#", "G ", "G#", "A ", "A#", "B "};
355
356 int midiPitchR = (int)std::round(midiPitch);
357 int pitchClass = midiPitchR % 12;
358 int octave = (midiPitchR / 12) - (middleCisC4 ? 1 : 2);
359
360 if (cents)
361 {
362 double frac = (midiPitch - (float)midiPitchR) * 100.;
363 noteName.SetFormatted(32, "%s%i %.0f", noteNames[pitchClass], octave, frac);
364 }
365 else
366 {
367 noteName.SetFormatted(32, "%s%i", noteNames[pitchClass], octave);
368 }
369}
