42void IPlugWeb::SendParameterValueFromUI(int paramIdx, double value)
43{
44#if WEBSOCKET_CLIENT
45 int pos = kNumMsgHeaderBytes;
46 *((int*)(mSPVFUIBuf.GetData() + pos)) = paramIdx;
47 pos += sizeof(int);
48 *((double*)(mSPVFUIBuf.GetData() + pos)) = value;
49 pos += sizeof(double);
50
51 EM_ASM(
52 {
53 var jsbuff = Module.HEAPU8.subarray($0, $0 + $1);
54 ws.send(jsbuff);
55 },
56 (int)mSPVFUIBuf.GetData(), kNumSPVFUIBytes);
57
58#else
59 val::global(mWAMCtrlrJSObjectName.Get()).call<void>("setParam", paramIdx, value);
60#endif
61 IPlugAPIBase::SendParameterValueFromUI(paramIdx, value); // call super class in order to make sure OnParamChangeUI() gets triggered
62};

64void IPlugWeb::SendMidiMsgFromUI(const IMidiMsg& msg)
65{
66#if WEBSOCKET_CLIENT
67 int pos = kNumMsgHeaderBytes;
68 mSMMFUIBuf.GetData()[pos] = msg.mStatus;
69 pos++;
70 mSMMFUIBuf.GetData()[pos] = msg.mData1;
71 pos++;
72 mSMMFUIBuf.GetData()[pos] = msg.mData2;
73 pos++;
74
75 EM_ASM(
76 {
77 var jsbuff = Module.HEAPU8.subarray($0, $0 + $1);
78 ws.send(jsbuff);
79 },
80 (int)mSMMFUIBuf.GetData(), kNumSMMFUIBytes);
81
82#else
83 WDL_String dataStr;
84 dataStr.SetFormatted(16, "%i:%i:%i", msg.mStatus, msg.mData1, msg.mData2);
85 val::global(mWAMCtrlrJSObjectName.Get()).call<void>("sendMessage", std::string("SMMFUI"), std::string(dataStr.Get()));
86#endif
87}

116void IPlugWeb::SendArbitraryMsgFromUI(int msgTag, int ctrlTag, int dataSize, const void* pData)
117{
118 mSAMFUIBuf.Resize(kNumSAMFUIBytes + dataSize);
119 int pos = kNumMsgHeaderBytes;
120
121 *((int*)(mSAMFUIBuf.GetData() + pos)) = msgTag;
122 pos += sizeof(int);
123 *((int*)(mSAMFUIBuf.GetData() + pos)) = ctrlTag;
124 pos += sizeof(int);
125 *((int*)(mSAMFUIBuf.GetData() + pos)) = dataSize;
126 pos += sizeof(int);
127
128 memcpy(mSAMFUIBuf.GetData() + pos, pData, dataSize);
129
130#if WEBSOCKET_CLIENT
131 EM_ASM(
132 {
133 var jsbuff = Module.HEAPU8.subarray($0, $0 + $1);
134 ws.send(jsbuff);
135 },
136 (int)mSAMFUIBuf.GetData(), mSAMFUIBuf.Size());
137#else
138 EM_ASM(
139 {
140 if (typeof window[Module.UTF8ToString($0)] == = 'undefined')
141 {
142 console.log("warning - SAMFUI called before controller exists");
143 }
144 else
145 {
146 window[Module.UTF8ToString($0)].sendMessage('SAMFUI', "", Module.HEAPU8.slice($1, $1 + $2).buffer);
147 }
148 },
149 mWAMCtrlrJSObjectName.Get(), (int)mSAMFUIBuf.GetData() + kNumMsgHeaderBytes, mSAMFUIBuf.Size() - kNumMsgHeaderBytes); // Non websocket doesn't need "SAMFUI" bytes at beginning
150#endif
151}
