60  EM_ASM({

67  dataStr.SetFormatted(16, "%i:%i:%i", msg.mStatus, msg.mData1, msg.mData2);
68  val::global(mWAMCtrlrJSObjectName.Get()).call<void>("sendMessage", std::string("SMMFUI"), std::string(dataStr.Get()));

99 void IPlugWeb::SendArbitraryMsgFromUI(int messageTag, int controlTag, int dataSize, const void* pData)
100 {
101  mSAMFUIBuf.Resize(kNumSAMFUIBytes + dataSize);
102  int pos = kNumMsgHeaderBytes;
103 
104  *((int*)(mSAMFUIBuf.GetData() + pos)) = messageTag; pos += sizeof(int);
105  *((int*)(mSAMFUIBuf.GetData() + pos)) = controlTag; pos += sizeof(int);
106  *((int*)(mSAMFUIBuf.GetData() + pos)) = dataSize; pos += sizeof(int);
107 
108  memcpy(mSAMFUIBuf.GetData() + pos, pData, dataSize);
109 
110 #if WEBSOCKET_CLIENT
111  EM_ASM({
112  var jsbuff = Module.HEAPU8.subarray($0, $0 + $1);
113  ws.send(jsbuff);
114  }, (int) mSAMFUIBuf.GetData(), mSAMFUIBuf.Size());
115 #else
116  EM_ASM({
117  window[Module.Pointer_stringify($0)].sendMessage('SAMFUI', "", Module.HEAPU8.slice($1, $1 + $2).buffer);
118  }, mWAMCtrlrJSObjectName.Get(), (int) mSAMFUIBuf.GetData() + kNumMsgHeaderBytes, mSAMFUIBuf.Size() - kNumMsgHeaderBytes); // Non websocket doesn't need "SAMFUI" bytes at beginning
119 #endif
120 }
