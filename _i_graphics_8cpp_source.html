487 void IGraphics::DrawBitmapedText(IBitmap& bitmap, IRECT& bounds, IText& text, IBlend* pBlend, const char* str, bool vCenter, bool multiline, int charWidth, int charHeight, int charOffset)
488 {
489  if (CStringHasContents(str))
490  {
491  int stringLength = (int) strlen(str);
492 
493  float basicYOffset = 0.;
494  float basicXOffset = 0.;
495 
496  if (vCenter)
497  basicYOffset = bounds.T + ((bounds.H() - charHeight) / 2.f);
498  else
499  basicYOffset = bounds.T;
500 
501  if (text.mAlign == IText::kAlignCenter)
502  basicXOffset = bounds.L + ((bounds.W() - (stringLength * charWidth)) / 2.f);
503  else if (text.mAlign == IText::kAlignNear)
504  basicXOffset = bounds.L;
505  else if (text.mAlign == IText::kAlignFar)
506  basicXOffset = bounds.R - (stringLength * charWidth);
507 
508  int widthAsOneLine = charWidth * stringLength;
509 
510  int nLines;
511  int stridx = 0;
512 
513  int nCharsThatFitIntoLine;
514 
515  if(multiline)
516  {
517  if (widthAsOneLine > bounds.W())
518  {
519  nCharsThatFitIntoLine = int(bounds.W() / (float)charWidth);
520  nLines = int(float(widthAsOneLine) / bounds.W()) + 1;
521  }
522  else // line is shorter than width of bounds
523  {
524  nCharsThatFitIntoLine = stringLength;
525  nLines = 1;
526  }
527  }
528  else
529  {
530  nCharsThatFitIntoLine = int(bounds.W() / (float) charWidth);
531  nLines = 1;
532  }
533 
534  for (int line=0; line<nLines; line++)
535  {
536  float yOffset = basicYOffset + line * charHeight;
537 
538  for (int linepos=0; linepos<nCharsThatFitIntoLine; linepos++)
539  {
540  if (str[stridx] == '\0') return;
541 
542  int frameOffset = (int) str[stridx++] - 31; // calculate which frame to look up
543 
544  float xOffset = ((float) linepos * ((float) charWidth + (float) charOffset)) + basicXOffset; // calculate xOffset for character we're drawing
545  IRECT charRect = IRECT(xOffset, yOffset, xOffset + charWidth, yOffset + charHeight);
546  DrawBitmap(bitmap, charRect, frameOffset, pBlend);
547  }
548  }
549  }
550 }
