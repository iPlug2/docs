89void IGraphics::Resize(int w, int h, float scale, bool needsPlatformResize)
90{
91 GetDelegate()->ConstrainEditorResize(w, h);
92
93 scale = Clip(scale, mMinScale, mMaxScale);
94
95 if (w == Width() && h == Height() && scale == GetDrawScale()) return;
96
97 //DBGMSG("resize %i, resize %i, scale %f\n", w, h, scale);
98 ReleaseMouseCapture();
99
100 mDrawScale = scale;
101 mWidth = w;
102 mHeight = h;
103
104 if (mCornerResizer)
105 mCornerResizer->OnRescale();
106
107 int windowWidth = WindowWidth() * GetPlatformWindowScale();
108 int windowHeight = WindowHeight() * GetPlatformWindowScale();
109
110 PlatformResize(GetDelegate()->EditorResizeFromUI(windowWidth, windowHeight, needsPlatformResize));
111 ForAllControls(&IControl::OnResize);
112 SetAllControlsDirty();
113 DrawResize();
114
115 if(mLayoutOnResize)
116 GetDelegate()->LayoutUI(this);
117}

137void IGraphics::RemoveControls(int fromIdx)
138{
139 int idx = NControls()-1;
140 while (idx >= fromIdx)
141 {
142 IControl* pControl = GetControl(idx);
143

145 ReleaseMouseCapture();
146
147 if(pControl == mMouseOver)
148 ClearMouseOver();
149
150 if(pControl == mInTextEntry)

152
153 if(pControl == mInPopupMenu)
154 mInPopupMenu = nullptr;
155
156 if(pControl->GetTag() > kNoTag)
157 mCtrlTags.erase(pControl->GetTag());
158
159 mControls.Delete(idx--, true);
160 }
161
162 SetAllControlsDirty();
163}

382void IGraphics::ShowBubbleControl(IControl* pCaller, float x, float y, const char* str, EDirection dir, IRECT minimumContentBounds)
383{
384 assert(mBubbleControls.GetSize() && "No bubble controls attached");
385
386 if(MultiTouchEnabled())
387 {
388 std::vector<ITouchID> touchIDsForCaller;
389 GetTouches(pCaller, touchIDsForCaller);
390 std::vector<IBubbleControl*> availableBubbleControls;
391 int nBubbleControls = mBubbleControls.GetSize();
392
393 if(touchIDsForCaller.size() == 1)
394 {
395 ITouchID touchID = touchIDsForCaller[0];
396
397 // first search to see if this touch matches existing bubble controls
398 for(int i=0;i<nBubbleControls;i++)
399 {
400 IBubbleControl* pBubbleControl = mBubbleControls.Get(i);
401 if(pBubbleControl->GetTouchID() == touchID)
402 {
403 pBubbleControl->ShowBubble(pCaller, x, y, str, dir, minimumContentBounds, touchID);
404 return;
405 }
406 else
407 availableBubbleControls.push_back(pBubbleControl);
408 }
409
410 if(availableBubbleControls.size())
411 {
412 // this works but why?
413 static int whichBubbleControl = 0;
414 availableBubbleControls[whichBubbleControl++]->ShowBubble(pCaller, x, y, str, dir, minimumContentBounds, touchID);
415 whichBubbleControl %= nBubbleControls;
416 }
417 }
418// else
419// {
420// assert(0 && "multi-touch controls with bubble controls not yet supported!");
421// }
422 }
423 else
424 mBubbleControls.Get(0)->ShowBubble(pCaller, x, y, str, dir, minimumContentBounds);
425}

603void IGraphics::PromptUserInput(IControl& control, const IRECT& bounds, int valIdx)
604{
605 assert(valIdx > kNoValIdx);
606
607 const IParam* pParam = control.GetParam(valIdx);
608
609 if(pParam)
610 {
611 IParam::EParamType type = pParam->Type();
612 const int nDisplayTexts = pParam->NDisplayTexts();
613 WDL_String currentText;
614
615 if ( type == IParam::kTypeEnum || (type == IParam::kTypeBool && nDisplayTexts))
616 {
617 pParam->GetDisplay(currentText);
618 mPromptPopupMenu.Clear();
619
620 // Fill the menu
621 for (int i = 0; i < nDisplayTexts; ++i)
622 {
623 const char* str = pParam->GetDisplayText(i);
624 // TODO: what if two parameters have the same text?
625 if (!strcmp(str, currentText.Get())) // strings are equal
626 mPromptPopupMenu.AddItem( new IPopupMenu::Item(str, IPopupMenu::Item::kChecked), -1 );
627 else // not equal
628 mPromptPopupMenu.AddItem( new IPopupMenu::Item(str), -1 );
629
630 mPromptPopupMenu.SetRootTitle(pParam->GetName());
631 }
632
633 CreatePopupMenu(control, mPromptPopupMenu, bounds, valIdx);
634 }
635 // TODO: what if there are Int/Double Params with a display text e.g. -96db = "mute"
636 else // type == IParam::kTypeInt || type == IParam::kTypeDouble
637 {
638 pParam->GetDisplay(currentText, false);
639
640 if(control.GetPromptShowsParamLabel())
641 {
642 currentText.Append(" ");
643 currentText.Append(pParam->GetLabel());
644 }
645
646 CreateTextEntry(control, control.GetText(), bounds, currentText.Get(), valIdx);
647 }
648 }
649}

673void IGraphics::DrawBitmap(const IBitmap& bitmap, const IRECT& bounds, int bmpState, const IBlend* pBlend)
674{
675 int srcX = 0;
676 int srcY = 0;
677
678 bmpState = Clip(bmpState, 1, bitmap.N());
679
680 if (bitmap.N() > 1 && bmpState > 1)
681 {
682 if (bitmap.GetFramesAreHorizontal())
683 {
684 srcX = bitmap.W() * (bmpState - 1) / bitmap.N();
685 }
686 else
687 {
688 srcY = bitmap.H() * (bmpState - 1) / bitmap.N();
689 }
690 }
691
692 return DrawBitmap(bitmap, bounds, srcX, srcY, pBlend);
693}

695void IGraphics::DrawBitmapedText(const IBitmap& bitmap, const IRECT& bounds, IText& text, IBlend* pBlend, const char* str, bool vCenter, bool multiline, int charWidth, int charHeight, int charOffset)
696{
697 if (CStringHasContents(str))
698 {
699 int stringLength = (int) strlen(str);
700
701 float basicYOffset = 0.;
702 float basicXOffset = 0.;
703
704 if (vCenter)
705 basicYOffset = bounds.T + ((bounds.H() - charHeight) / 2.f);
706 else
707 basicYOffset = bounds.T;
708
709 if (text.mAlign == EAlign::Center)
710 basicXOffset = bounds.L + ((bounds.W() - (stringLength * charWidth)) / 2.f);
711 else if (text.mAlign == EAlign::Near)
712 basicXOffset = bounds.L;
713 else if (text.mAlign == EAlign::Far)
714 basicXOffset = bounds.R - (stringLength * charWidth);
715
716 int widthAsOneLine = charWidth * stringLength;
717
718 int nLines;
719 int stridx = 0;
720
721 int nCharsThatFitIntoLine;
722
723 if(multiline)
724 {
725 if (widthAsOneLine > bounds.W())
726 {
727 nCharsThatFitIntoLine = int(bounds.W() / (float)charWidth);
728 nLines = int(float(widthAsOneLine) / bounds.W()) + 1;
729 }
730 else // line is shorter than width of bounds
731 {
732 nCharsThatFitIntoLine = stringLength;
733 nLines = 1;
734 }
735 }
736 else
737 {
738 nCharsThatFitIntoLine = int(bounds.W() / (float) charWidth);
739 nLines = 1;
740 }
741
742 for (int line=0; line<nLines; line++)
743 {
744 float yOffset = basicYOffset + line * charHeight;
745
746 for (int linepos=0; linepos<nCharsThatFitIntoLine; linepos++)
747 {
748 if (str[stridx] == '\0') return;
749
750 int frameOffset = (int) str[stridx++] - 31; // calculate which frame to look up
751
752 float xOffset = ((float) linepos * ((float) charWidth + (float) charOffset)) + basicXOffset; // calculate xOffset for character we're drawing
753 IRECT charRect = IRECT(xOffset, yOffset, xOffset + charWidth, yOffset + charHeight);
754 DrawBitmap(bitmap, charRect, frameOffset, pBlend);
755 }
756 }
757 }
758}

2023void IGraphics::ApplyLayerDropShadow(ILayerPtr& layer, const IShadow& shadow)
2024{
2025 auto GaussianBlurSwap = [](uint8_t* out, uint8_t* in, uint8_t* kernel, int width, int height,
2026 int outStride, int inStride, int kernelSize, uint32_t norm)
2027 {
2028 int repeats = 0;
2029 int fullKernelSize = kernelSize * 2 + 1;
2030 uint32_t last = 0;
2031
2032 auto RepeatCheck = [&](int idx)
2033 {
2034 repeats = last == in[idx * 4] ? std::min(repeats + 1, fullKernelSize) : 1;
2035 last = in[idx * 4];
2036
2037 return repeats == fullKernelSize;
2038 };
2039
2040 for (int i = 0; i < height; i++, in += inStride)
2041 {
2042 for (int j = 0; j < kernelSize - 1; j++)
2043 {
2044 uint32_t accum = in[j * 4] * kernel[0];
2045 for (int k = 1; k < j + 1; k++)
2046 accum += kernel[k] * in[(j - k) * 4];
2047 for (int k = 1; k < kernelSize; k++)
2048 accum += kernel[k] * in[(j + k) * 4];
2049 out[j * outStride + (i * 4)] = static_cast<uint8_t>(std::min(static_cast<uint32_t>(255), accum / norm));
2050 }
2051 for (int j = 0; j < kernelSize * 2 - 2; j++)
2052 RepeatCheck(j);
2053 for (int j = kernelSize - 1; j < (width - kernelSize) + 1; j++)
2054 {
2055 if (RepeatCheck(j + kernelSize - 1))
2056 {
2057 out[j * outStride + (i * 4)] = static_cast<uint8_t>(last);
2058 continue;
2059 }
2060
2061 uint32_t accum = in[j * 4] * kernel[0];
2062 for (int k = 1; k < kernelSize; k++)
2063 accum += kernel[k] * (in[(j - k) * 4] + in[(j + k) * 4]);
2064 out[j * outStride + (i * 4)] = static_cast<uint8_t>(std::min(static_cast<uint32_t>(255), accum / norm));
2065 }
2066 for (int j = (width - kernelSize) + 1; j < width; j++)
2067 {
2068 uint32_t accum = in[j * 4] * kernel[0];
2069 for (int k = 1; k < kernelSize; k++)
2070 accum += kernel[k] * in[(j - k) * 4];
2071 for (int k = 1; k < width - j; k++)
2072 accum += kernel[k] * in[(j + k) * 4];
2073 out[j * outStride + (i * 4)] = static_cast<uint8_t>(std::min(static_cast<uint32_t>(255), accum / norm));
2074 }
2075 }
2076 };
2077
2078 RawBitmapData temp1;
2079 RawBitmapData temp2;
2080 RawBitmapData kernel;
2081
2082 // Get bitmap in 32-bit form
2083 GetLayerBitmapData(layer, temp1);
2084
2085 if (!temp1.GetSize())
2086 return;
2087 temp2.Resize(temp1.GetSize());
2088
2089 // Form kernel (reference blurSize from zero (which will be no blur))
2090 bool flipped = FlippedBitmap();
2091 float scale = layer->GetAPIBitmap()->GetScale() * layer->GetAPIBitmap()->GetDrawScale();
2092 float blurSize = std::max(1.f, (shadow.mBlurSize * scale) + 1.f);
2093 float blurConst = 4.5f / (blurSize * blurSize);
2094 int iSize = static_cast<int>(ceil(blurSize));
2095 int width = layer->GetAPIBitmap()->GetWidth();
2096 int height = layer->GetAPIBitmap()->GetHeight();
2097 int stride1 = temp1.GetSize() / width;
2098 int stride2 = flipped ? -temp1.GetSize() / height : temp1.GetSize() / height;
2099 int stride3 = flipped ? -stride2 : stride2;
2100
2101 kernel.Resize(iSize);
2102
2103 for (int i = 0; i < iSize; i++)
2104 kernel.Get()[i] = static_cast<uint8_t>(std::round(255.f * std::expf(-(i * i) * blurConst)));
2105
2106 // Kernel normalisation
2107 int normFactor = kernel.Get()[0];
2108
2109 for (int i = 1; i < iSize; i++)
2110 normFactor += kernel.Get()[i] + kernel.Get()[i];
2111
2112 // Do blur
2113 uint8_t* asRows = temp1.Get() + AlphaChannel();
2114 uint8_t* inRows = flipped ? asRows + stride3 * (height - 1) : asRows;
2115 uint8_t* asCols = temp2.Get() + AlphaChannel();
2116
2117 GaussianBlurSwap(asCols, inRows, kernel.Get(), width, height, stride1, stride2, iSize, normFactor);
2118 GaussianBlurSwap(asRows, asCols, kernel.Get(), height, width, stride3, stride1, iSize, normFactor);
2119
2120 // Apply alphas to the pattern and recombine/replace the image
2121 ApplyShadowMask(layer, temp1, shadow);
2122}

2800#else

2806 {

2827 // Compute whether this path is a hole or a solid and set the winding direction accordingly.
