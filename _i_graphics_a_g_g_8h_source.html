31  alpha_span_generator(typename SpanGeneratorType::source_type& source, typename SpanGeneratorType::interpolator_type& interpolator, agg::cover_type a)
32  : SpanGeneratorType(source, interpolator), alpha(a) {}
33 
34  void generate(typename SpanGeneratorType::color_type* span, int x, int y, unsigned len)
35  {
36  SpanGeneratorType::generate(span, x, y, len);
37 
38  if (alpha != 255)

41  span->a = (span->a * alpha + SpanGeneratorType::base_mask) >> SpanGeneratorType::base_shift;
42  }
43  }

180  void BlendFrom(agg::rendering_buffer& renBuf, const IRECT& bounds, int srcX, int srcY, agg::comp_op_e op, agg::cover_type cover, bool preMultiplied)
181  {
182  // N.B. blend_from/rect_i is inclusive, hence -1 on each dimension here
183 
184  agg::rect_i r(srcX, srcY, srcX + std::round(bounds.W()) - 1, srcY + std::round(bounds.H()) - 1);
185  int x = std::round(bounds.L) - srcX;
186  int y = std::round(bounds.T) - srcY;
187 
188  if (preMultiplied)
189  {
190  mPixfPre.comp_op(op);
191  mRenBasePre.blend_from(PixfmtType(renBuf), &r, x, y, cover);
192  }
193  else
194  {
195  mPixf.comp_op(op);
196  mRenBase.blend_from(PixfmtType(renBuf), &r, x, y, cover);
197  }
198  }
