24static void DrawRadialMarks(IGraphics& g, const IRECT& track, int numMajor, float a1 = -135.0f, float a2 = 135.0f, float majorThickness = 2.0f, float minorThickness = 1.0f, float markSize = 10.0f, float markRatio = 0.6f)
25{
26 auto radius = track.W() / 2.0f;
27 auto a1r = DegToRad(a1);
28 auto a2r = DegToRad(a2);
29
30 auto div = (a2r-a1r) / float(numMajor-1);
31
32 for (auto i = 0; i < numMajor; ++i)
33 {
34 auto from = radius;
35 auto markWidth = majorThickness;
36
37// if (i % (numMajor / (numMajor-1)))
38// {
39// from -= (markSize * (1.0f-markRatio));
40// markWidth = minorThickness;
41// }
42
43 auto angle = a1r - (iplug::PI / 2.0f) + (i * div);
44 auto sinX = std::sin(angle);
45 auto cosX = std::cos(angle);
46 auto to = radius - markSize;
47
48 from = radius;
49 to = radius * 1.5f;
50
51 g.DrawLine(COLOR_BLACK, track.MW() + from * cosX,
52 track.MH() + from * sinX,
53 track.MW() + to * cosX,
54 track.MH() + to * sinX, 0, markWidth);
55 }
56}

59static void DrawLinearMarks(IGraphics& g, const IRECT& track, int numMajor, int numMinor, EDirection dir, const IColor& color = COLOR_BLACK, float majorThickness = 2.0f, float minorThickness = 1.0f, float majorAlpha = 0.5f, float minorAlpha = 0.25f, float markFrac = 0.15f, float distanceFromTrackCentre = 15.0f)
60{
61 const auto cellIncr = 1.0f / numMajor;
62 const auto subMarkCellIncr = cellIncr / float(numMinor+1);
63
64 const auto lineDirection = dir == EDirection::Horizontal ? EDirection::Vertical : EDirection::Horizontal;
65 const auto wholeLineRect = dir == EDirection::Horizontal ? track.GetMidVPadded(distanceFromTrackCentre) : track.GetMidHPadded(distanceFromTrackCentre);
66
67 const auto lineRect1 = wholeLineRect.FracRect(lineDirection, markFrac);
68 const auto lineRect2 = wholeLineRect.FracRect(lineDirection, markFrac, true);
69
70 for (int i = numMajor; i >= 0; i--)
71 {
72 auto majorPos = float(i) / float(numMajor);
73 g.DrawLineAcross(color.WithOpacity(majorAlpha), lineRect1, lineDirection, majorPos, nullptr, majorThickness);
74 g.DrawLineAcross(color.WithOpacity(majorAlpha), lineRect2, lineDirection, majorPos, nullptr, majorThickness);
75
76 for (int j = 1; j <= numMinor; j++)
77 {
78 auto minorPos = (cellIncr * i) + (subMarkCellIncr * j);
79 g.DrawLineAcross(color.WithOpacity(minorAlpha), lineRect1, lineDirection, minorPos, nullptr, minorThickness);
80 g.DrawLineAcross(color.WithOpacity(minorAlpha), lineRect2, lineDirection, minorPos, nullptr, minorThickness);
81 }
82 }
83}

85static void DrawRadialLabels(IGraphics& g, const IRECT& track, const std::vector<std::string>& labels, float a1 = -135.0f, float a2 = 135.0f, const IText& text = DEFAULT_VALUE_TEXT)
86{
87 auto radius = track.W() / 2.0f;
88 auto a1r = DegToRad(a1);
89 auto a2r = DegToRad(a2);
90
91 auto numMajor = labels.size();
92
93 auto div = (a2r-a1r) / float(numMajor-1);
94
95 for (auto i = 0; i < numMajor; ++i)
96 {
97 auto angle = a1r - (iplug::PI / 2.0f) + (i * div);
98 auto sinX = std::sin(angle);
99 auto cosX = std::cos(angle);
100
101 auto x = track.MW() + (radius * 1.75) * cosX;
102 auto y = track.MH() + (radius * 1.75) * sinX;
103
104 g.DrawText(text, labels[i].c_str(), x, y);
105 }
106}

108static void DrawLinearLabels(IGraphics& g, const IRECT& track, int numMajor, EDirection dir, const IText& text, float distanceFromTrackCentre = -25.0f)
109{
110 IRECT labelRect;
111
112 if (dir == EDirection::Horizontal)
113 labelRect = track.GetTranslated(-(track.W() / numMajor) / 2.0, distanceFromTrackCentre);
114 else
115 labelRect = track.GetTranslated(distanceFromTrackCentre, -(track.H() / numMajor) / 2.0);
116
117 for (int i = 0; i <= numMajor; i++)
118 {
119 WDL_FastString str;
120 str.SetFormatted(32, "%i", i);
121 g.DrawText(text, str.Get(), labelRect.SubRect(dir, numMajor, i));
122 }
123}
