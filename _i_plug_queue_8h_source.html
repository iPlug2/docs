51 bool Push(const T& item)
52 {
53 const auto currentWriteIndex = mWriteIndex.load(std::memory_order_relaxed);
54 const auto nextWriteIndex = Increment(currentWriteIndex);
55 if (nextWriteIndex != mReadIndex.load(std::memory_order_acquire))
56 {
57 mData.Get()[currentWriteIndex] = item;
58 mWriteIndex.store(nextWriteIndex, std::memory_order_release);
59 return true;
60 }
61 return false;
62 }

68 bool Pop(T& item)
69 {
70 const auto currentReadIndex = mReadIndex.load(std::memory_order_relaxed);
71 if (currentReadIndex == mWriteIndex.load(std::memory_order_acquire))
72 {
73 return false; // empty the queue
74 }
75 item = mData.Get()[currentReadIndex];
76 mReadIndex.store(Increment(currentReadIndex), std::memory_order_release);
77 return true;
78 }

82 size_t ElementsAvailable() const
83 {
84 size_t write = mWriteIndex.load(std::memory_order_acquire);
85 size_t read = mReadIndex.load(std::memory_order_relaxed);
86
87 return (read > write) ? mData.GetSize() - (read - write) : write - read;
