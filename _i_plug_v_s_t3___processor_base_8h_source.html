45 template <class T>
46 void Initialize(T* pPlug)
47 {
48 using namespace Steinberg::Vst;
49
50 Steinberg::Vst::String128 tmpStringBuf;
51
52 // TODO: move this to IPlugProcessor::MaxNBuses(x,&x) method;
53 int configWithMostInputBuses = 0;
54 int configWithMostOutputBuses = 0;
55 /*int maxNInBuses =*/MaxNBuses(ERoute::kInput, &configWithMostInputBuses);
56 int maxNOutBuses = MaxNBuses(ERoute::kOutput, &configWithMostOutputBuses);
57
58 // Add the maximum number of input buses in any ioconfig. channel count/API Bus type will be changed later if
59 // nessecary
60 {
61 const IOConfig* pConfig = GetIOConfig(configWithMostInputBuses);
62
63 int nIn = pConfig->NBuses(ERoute::kInput);
64 for (auto busIdx = 0; busIdx < nIn; busIdx++)
65 {
66 uint64_t busType = GetAPIBusTypeForChannelIOConfig(configWithMostInputBuses, ERoute::kInput, busIdx, pConfig);
67
68 int flags = busIdx == 0 ? flags = BusInfo::BusFlags::kDefaultActive : flags = 0;
69 WDL_String busName;
70 GetBusName(ERoute::kInput, busIdx, nIn, busName);
71 Steinberg::UString(tmpStringBuf, 128).fromAscii(busName.Get(), 128);
72 pPlug->addAudioInput(tmpStringBuf, busType, busIdx > 0 ? kAux : kMain, flags);
73 }
74 }
75
76 // Add the maximum number of output buses in any ioconfig. channel count/API Bus type will be changed later if
77 // nessecary
78 {
79 const IOConfig* pConfig = GetIOConfig(configWithMostOutputBuses);
80 int nOut = pConfig->NBuses(ERoute::kOutput);
81
82 for (auto busIdx = 0; busIdx < nOut; busIdx++)
83 {
84 uint64_t busType = GetAPIBusTypeForChannelIOConfig(configWithMostOutputBuses, ERoute::kOutput, busIdx, pConfig);
85 int flags = busIdx == 0 ? flags = BusInfo::BusFlags::kDefaultActive : flags = 0;
86 WDL_String busName;
87 GetBusName(ERoute::kOutput, busIdx, nOut, busName);
88 Steinberg::UString(tmpStringBuf, 128).fromAscii(busName.Get(), 128);
89 pPlug->addAudioOutput(tmpStringBuf, busType, busIdx > 0 ? kAux : kMain, flags);
90 }
91 }
92
93 if (IsMidiEffect() && maxNOutBuses == 0)
94 {
95 int flags = 0;
96 Steinberg::UString(tmpStringBuf, 128).fromAscii("Dummy Output", 128);
97 pPlug->addAudioOutput(tmpStringBuf, Steinberg::Vst::SpeakerArr::kEmpty, Steinberg::Vst::BusTypes::kMain, flags);
98 }
99
100 if (DoesMIDIIn())
101 pPlug->addEventInput(STR16("MIDI Input"), VST3_NUM_MIDI_IN_CHANS);
102
103 if (DoesMIDIOut())
104 pPlug->addEventOutput(STR16("MIDI Output"), VST3_NUM_MIDI_OUT_CHANS);
105 }

116 // Audio Processing Setup
117 template <class T>
118 bool SetBusArrangements(T* pPlug,
119 Steinberg::Vst::SpeakerArrangement* pInputBusArrangements,
120 Steinberg::int32 numInBuses,
121 Steinberg::Vst::SpeakerArrangement* pOutputBusArrangements,
122 Steinberg::int32 numOutBuses)
123 {
124 using namespace Steinberg::Vst;
125
126 // This would seem to be a bug in Ardour
127 if ((pPlug->GetHost() == kHostMixbus32C) || (pPlug->GetHost() == kHostArdour))
128 {
129 return true;
130 }
131
132 // disconnect all io pins, they will be reconnected in process
133 SetChannelConnections(ERoute::kInput, 0, MaxNChannels(ERoute::kInput), false);
134 SetChannelConnections(ERoute::kOutput, 0, MaxNChannels(ERoute::kOutput), false);
135
136 const int maxNInBuses = MaxNBuses(ERoute::kInput);
137 const int maxNOutBuses = MaxNBuses(ERoute::kOutput);
138
139 // if the host is wanting more more buses that the max we have defined in channel io configs, return false
140 if (numInBuses > maxNInBuses || numOutBuses > maxNOutBuses)
141 return false;
142
143 std::vector<int> inputBuses;
144 std::vector<int> outputBuses;
145
146 for (int i = 0; i < numInBuses; i++)
147 {
148 inputBuses.push_back(SpeakerArr::getChannelCount(pInputBusArrangements[i]));
149 }
150
151 for (int i = 0; i < numOutBuses; i++)
152 {
153 outputBuses.push_back(SpeakerArr::getChannelCount(pOutputBusArrangements[i]));
154 }
155
156 const int matchingIdx = GetIOConfigWithChanCounts(inputBuses, outputBuses);
157 Steinberg::Vst::String128 tmpStringBuf;
158
159 if (matchingIdx > -1)
160 {
161 pPlug->removeAudioBusses();
162
163 const IOConfig* pConfig = GetIOConfig(matchingIdx);
164 const int nIn = pConfig->NBuses(ERoute::kInput);
165
166 for (auto inBusIdx = 0; inBusIdx < nIn; inBusIdx++)
167 {
168 const int flags = inBusIdx == 0 ? BusInfo::BusFlags::kDefaultActive : 0;
169 SpeakerArrangement arr = GetAPIBusTypeForChannelIOConfig(matchingIdx, ERoute::kInput, inBusIdx, pConfig);
170
171 WDL_String busName;
172 GetBusName(ERoute::kInput, inBusIdx, nIn, busName);
173 Steinberg::UString(tmpStringBuf, 128).fromAscii(busName.Get(), 128);
174 pPlug->addAudioInput(tmpStringBuf, arr, (BusTypes)inBusIdx > 0, flags);
175 }
176
177 const int nOut = pConfig->NBuses(ERoute::kOutput);
178
179 for (auto outBusIdx = 0; outBusIdx < nOut; outBusIdx++)
180 {
181 int flags = outBusIdx == 0 ? BusInfo::BusFlags::kDefaultActive : 0;
182 SpeakerArrangement arr = GetAPIBusTypeForChannelIOConfig(matchingIdx, ERoute::kOutput, outBusIdx, pConfig);
183
184 WDL_String busName;
185 GetBusName(ERoute::kOutput, outBusIdx, nOut, busName);
186 Steinberg::UString(tmpStringBuf, 128).fromAscii(busName.Get(), 128);
187 pPlug->addAudioOutput(tmpStringBuf, arr, (BusTypes)outBusIdx > 0, flags);
188 }
189
190 return true;
191 }
192
193 return false;
194 }
