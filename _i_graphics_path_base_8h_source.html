453  IPattern GetSVGPattern(const NSVGpaint& paint, float opacity)
454  {
455  int alpha = std::min(255, std::max(int(roundf(opacity * 255.f)), 0));
456 
457  switch (paint.type)
458  {
459  case NSVG_PAINT_COLOR:
460  return IColor(alpha, (paint.color >> 0) & 0xFF, (paint.color >> 8) & 0xFF, (paint.color >> 16) & 0xFF);
461 
462  case NSVG_PAINT_LINEAR_GRADIENT:
463  case NSVG_PAINT_RADIAL_GRADIENT:
464  {
465  NSVGgradient* pGrad = paint.gradient;
466 
467  IPattern pattern(paint.type == NSVG_PAINT_LINEAR_GRADIENT ? kLinearPattern : kRadialPattern);
468 
469  // Set Extend Rule
470  switch (pGrad->spread)
471  {
472  case NSVG_SPREAD_PAD: pattern.mExtend = kExtendPad; break;
473  case NSVG_SPREAD_REFLECT: pattern.mExtend = kExtendReflect; break;
474  case NSVG_SPREAD_REPEAT: pattern.mExtend = kExtendRepeat; break;
475  }
476 
477  // Copy Stops
478  for (int i = 0; i < pGrad->nstops; i++)
479  {
480  unsigned int color = pGrad->stops[i].color;
481  pattern.AddStop(IColor(255, (color >> 0) & 0xFF, (color >> 8) & 0xFF, (color >> 16) & 0xFF), pGrad->stops[i].offset);
482  }
483 
484  // Copy transform
485  pattern.SetTransform(pGrad->xform[0], pGrad->xform[1], pGrad->xform[2], pGrad->xform[3], pGrad->xform[4], pGrad->xform[5]);
486 
487  return pattern;
488  }
489  default:
490  return IColor(alpha, 0, 0, 0);
491  }
492  }

494  void RenderNanoSVG(NSVGimage* pImage)
495  {
496  assert(pImage != nullptr);
497 
498  for (NSVGshape* pShape = pImage->shapes; pShape; pShape = pShape->next)
499  {
500  if (!(pShape->flags & NSVG_FLAGS_VISIBLE))
501  continue;
502 
503  for (NSVGpath* pPath = pShape->paths; pPath; pPath = pPath->next)
504  {
505  PathMoveTo(pPath->pts[0], pPath->pts[1]);
506 
507  for (int i = 0; i < pPath->npts - 1; i += 3)
508  {
509  float *p = pPath->pts + i * 2 + 2;
510  PathCurveTo(p[0], p[1], p[2], p[3], p[4], p[5]);
511  }
512 
513  if (pPath->closed)
514  PathClose();
515  }
516 
517  // Fill
518  if (pShape->fill.type != NSVG_PAINT_NONE)
519  {
520  IFillOptions options;
521 
522  if (pShape->fillRule == NSVG_FILLRULE_EVENODD)
523  options.mFillRule = kFillEvenOdd;
524  else
525  options.mFillRule = kFillWinding;
526 
527  options.mPreserve = pShape->stroke.type != NSVG_PAINT_NONE;
528  PathFill(GetSVGPattern(pShape->fill, pShape->opacity), options, nullptr);
529  }
530 
531  // Stroke
532  if (pShape->stroke.type != NSVG_PAINT_NONE)
533  {
534  IStrokeOptions options;
535 
536  options.mMiterLimit = pShape->miterLimit;
537 
538  switch (pShape->strokeLineCap)
539  {
540  case NSVG_CAP_BUTT: options.mCapOption = kCapButt; break;
541  case NSVG_CAP_ROUND: options.mCapOption = kCapRound; break;
542  case NSVG_CAP_SQUARE: options.mCapOption = kCapSquare; break;
543  }
544 
545  switch (pShape->strokeLineJoin)
546  {
547  case NSVG_JOIN_MITER: options.mJoinOption = kJoinMiter; break;
548  case NSVG_JOIN_ROUND: options.mJoinOption = kJoinRound; break;
549  case NSVG_JOIN_BEVEL: options.mJoinOption = kJoinBevel; break;
550  }
551 
552  options.mDash.SetDash(pShape->strokeDashArray, pShape->strokeDashOffset, pShape->strokeDashCount);
553 
554  PathStroke(GetSVGPattern(pShape->stroke, pShape->opacity), pShape->strokeWidth, options, nullptr);
555  }
556  }
