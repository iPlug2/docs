131 void IGraphicsAGG::Rasterizer::Rasterize(const IPattern& pattern, agg::comp_op_e op, float opacity, EFillRule rule)
132 {
133  mRasterizer.filling_rule(rule == kFillWinding ? agg::fill_non_zero : agg::fill_even_odd );
134 
135  switch (pattern.mType)
136  {
137  case kSolidPattern:
138  {
139  // Rasterize
140 
141  Rasterize(AGGColor(pattern.GetStop(0).mColor, opacity), op);
142  }
143  break;
144 
145  case kLinearPattern:
146  case kRadialPattern:
147  {
148  // Common gradient objects
149 
150  const IMatrix& m = pattern.mTransform;
151 
152  agg::trans_affine gradientMTX(m.mXX, m.mYX , m.mXY, m.mYY, m.mTX, m.mTY);
153  agg::gradient_lut<agg::color_interpolator<agg::rgba8>, 512> colorArray;
154 
155  // Scaling
156 
157  gradientMTX = (agg::trans_affine() / mGraphics.mTransform) * gradientMTX * agg::trans_affine_scaling(512.0);
158 
159  // Make gradient lut
160 
161  colorArray.remove_all();
162 
163  for (int i = 0; i < pattern.NStops(); i++)
164  {
165  const IColorStop& stop = pattern.GetStop(i);
166  float offset = stop.mOffset;
167  colorArray.add_color(offset, AGGColor(stop.mColor, opacity));
168  }
169 
170  colorArray.build_lut();
171 
172  // Rasterize
173 
174  if (pattern.mType == kLinearPattern)
175  {
176  GradientRasterizeAdapt(*this, pattern.mExtend, agg::gradient_y(), gradientMTX, colorArray, op);
177  }
178  else
179  {
180  GradientRasterizeAdapt(*this, pattern.mExtend, agg::gradient_radial_d(), gradientMTX, colorArray, op);
181  }
182  }
183  break;
184  }
185 }

282 bool CheckTransform(const agg::trans_affine& mtx)
283 {
284  if (!agg::is_equal_eps(mtx.tx - std::round(mtx.tx), 0.0, 1e-3))
285  return false;
286  if (!agg::is_equal_eps(mtx.ty - std::round(mtx.ty), 0.0, 1e-3))

295 void IGraphicsAGG::DrawBitmap(IBitmap& bitmap, const IRECT& dest, int srcX, int srcY, const IBlend* pBlend)
296 {
297  bool preMultiplied = static_cast<AGGBitmap*>(bitmap.GetAPIBitmap())->IsPreMultiplied();
298  IRECT bounds = mClipRECT.Empty() ? dest : mClipRECT.Intersect(dest);
299  bounds.Scale(GetBackingPixelScale());
300 
301  APIBitmap* pAPIBitmap = dynamic_cast<AGGBitmap*>(bitmap.GetAPIBitmap());
302  agg::pixel_map* pSource = pAPIBitmap->GetBitmap();
303  agg::rendering_buffer src(pSource->buf(), pSource->width(), pSource->height(), pSource->row_bytes());;
304  const double scale = GetScreenScale() / (pAPIBitmap->GetScale() * pAPIBitmap->GetDrawScale());
305 
306  agg::trans_affine srcMtx;
307  srcMtx /= mTransform;
308  srcMtx *= agg::trans_affine_translation((srcX * scale) - dest.L, (srcY * scale) - dest.T);
309  srcMtx *= agg::trans_affine_scaling(bitmap.GetScale() * bitmap.GetDrawScale());
310 
311  if (bounds.IsPixelAligned() && CheckTransform(srcMtx))
312  {
313  double offsetScale = scale * GetScreenScale();
314  IRECT destScaled = dest.GetScaled(GetBackingPixelScale());
315  srcX = std::round(srcX * offsetScale + std::max(0.f, bounds.L - destScaled.L));
316  srcY = std::round(srcY * offsetScale + std::max(0.f, bounds.T - destScaled.T));
317  bounds.Translate(mTransform.tx, mTransform.ty);
318 
319  mRasterizer.BlendFrom(src, bounds, srcX, srcY, AGGBlendMode(pBlend), AGGCover(pBlend), preMultiplied);
320  }
321  else
322  {
323  agg::rounded_rect rect(dest.L, dest.T, dest.R, dest.B, 0);
324  agg::conv_transform<agg::rounded_rect> tr(rect, mTransform);
325 
326  if (preMultiplied)
327  {
328  PixfmtPreType fmtSrc(src);
329  mRasterizer.Rasterize(fmtSrc, tr, srcMtx, AGGBlendMode(pBlend), AGGCover(pBlend));
330  }
331  else
332  {
333  PixfmtType fmtSrc(src);
334  mRasterizer.Rasterize(fmtSrc, tr, srcMtx, AGGBlendMode(pBlend), AGGCover(pBlend));
335  }
336  }
337 }

410 void IGraphicsAGG::PathStroke(const IPattern& pattern, float thickness, const IStrokeOptions& options, const IBlend* pBlend)
411 {
412  typedef agg::conv_curve<agg::path_storage> CPType;
413  typedef agg::conv_transform<CPType> S1Type;
414  typedef agg::conv_stroke<S1Type> S2Type;
415  typedef agg::conv_transform<S2Type> S3Type;
416  typedef agg::conv_dash<S1Type> D2Type;
417  typedef agg::conv_stroke<D2Type> D3Type;
418  typedef agg::conv_transform<D3Type> D4Type;
419 
420  agg::trans_affine tranform(mTransform);
421  CPType curvedPath(mPath);
422  S1Type basePath(curvedPath, tranform.invert());
423 
424  if (options.mDash.GetCount())
425  {
426  D2Type dashedPath(basePath);
427  D3Type strokedDashedPath(dashedPath);
428  D4Type finalPath(strokedDashedPath, mTransform);
429 
430  // Set the dashes (N.B. - for odd counts the array is read twice)
431 
432  int dashCount = options.mDash.GetCount();
433  int dashMax = dashCount & 1 ? dashCount * 2 : dashCount;
434  const float* dashArray = options.mDash.GetArray();
435 
436  dashedPath.remove_all_dashes();
437  dashedPath.dash_start(options.mDash.GetOffset());
438 
439  for (int i = 0; i < dashMax; i += 2)
440  dashedPath.add_dash(dashArray[i % dashCount], dashArray[(i + 1) % dashCount]);
441 
442  StrokeOptions(strokedDashedPath, thickness, options);
443  mRasterizer.Rasterize(finalPath, pattern, AGGBlendMode(pBlend), BlendWeight(pBlend));
444  }
445  else
446  {
447  S2Type strokedPath(basePath);
448  S3Type finalPath(strokedPath, mTransform);
449 
450  StrokeOptions(strokedPath, thickness, options);
451  mRasterizer.Rasterize(finalPath, pattern, AGGBlendMode(pBlend), BlendWeight(pBlend));
452  }
453 
454  if (!options.mPreserve)
455  mPath.remove_all();
456 }

502 APIBitmap* IGraphicsAGG::ScaleAPIBitmap(const APIBitmap* pBitmap, int scale)
503 {

512  PixfmtType imgPixfDest(dest);

516 
517  agg::rasterizer_scanline_aa<> rasterizer;
518  agg::trans_affine_scaling srcMtx(pBitmap->GetScale() / (double) scale);

520  imgSourceType imgSrc(imgPixfSrc);

522  agg::scanline_u8 scanline;
523  SpanGeneratorType spanGenerator(imgSrc, interpolator);
524  BitmapRenderType renderer(renBase, spanAllocator, spanGenerator);
525 
526  agg::rounded_rect bounds(0, 0, destW, destH, 0);
527  rasterizer.add_path(bounds);
528  agg::render_scanlines(rasterizer, scanline, renderer);
529 
530  return new AGGBitmap(pCopy, scale, pBitmap->GetDrawScale(), false);

560  agg::pixel_map* pPixMap = pBitmap->GetBitmap();

580  IBlend blend1(kBlendSourceIn, 1.0);

588  PathTransformRestore();
589  }

654 bool IGraphicsAGG::DoDrawMeasureText(const IText& text, const char* str, IRECT& bounds, const IBlend* pBlend, bool measure)
655 {
656  if (!str || str[0] == '\0')
657  {
658  return true;
659  }
660 
661  float weight = 0.0;
662  const bool kerning = false;
663  const bool hinting = false;
664 
665  mFontContour.width(-weight * (text.mSize * 0.05));
666 
667  agg::font* pFontData = FindFont(text.mFont, text.mSize);
668 
669  if (!pFontData)
670  pFontData = FindFont("Arial", text.mSize);
671 
672  assert(pFontData);
673 
674  if (pFontData != 0 && mFontEngine.load_font("", 0, agg::glyph_ren_outline, pFontData->buf(), pFontData->size()))
675  {
676  mFontEngine.hinting(hinting);
677  mFontEngine.height(text.mSize);
678  mFontEngine.width(text.mSize);
679  mFontEngine.flip_y(true);
680 
681  WDL_TypedBuf<LineInfo> lines;
682  CalculateTextLines(&lines, bounds, str, mFontManager);
683  LineInfo * pLines = lines.Get();
684 
685  double x = bounds.L;
686  double y = bounds.T + (text.mSize);
687 
688  switch (text.mVAlign)
689  {
690  case IText::kVAlignTop:
691  break;
692  case IText::kVAlignMiddle:
693  y += ((bounds.H() - (lines.GetSize() * text.mSize)) / 2.0);
694  break;
695  case IText::kVAlignBottom:
696  y = bounds.B;
697  break;
698  }
699 
700  if (measure)
701  {
702  double width = 0.0;
703 
704  for (int i = 0; i < lines.GetSize(); ++i, ++pLines)
705  width = std::max(width, pLines->mWidth);
706 
707  bounds.B = bounds.T + lines.GetSize() * text.mSize;
708  bounds.R = bounds.L + width;
709  }
710  else
711  {
712  for (int i = 0; i < lines.GetSize(); ++i, ++pLines)
713  {
714  switch (text.mAlign)
715  {
716  case IText::kAlignNear:
717  x = bounds.L;
718  break;
719  case IText::kAlignCenter:
720  x = bounds.L + ((bounds.W() - pLines->mWidth) / 2.0);
721  break;
722  case IText::kAlignFar:
723  x = bounds.L + (bounds.W() - pLines->mWidth);
724  break;
725  }
726 
727  for (size_t c = pLines->mStartChar; c < pLines->mEndChar; c++)
728  {
729  const agg::glyph_cache* pGlyph = mFontManager.glyph(str[c]);
730 
731  if (pGlyph)
732  {
733  if (kerning)
734  {
735  mFontManager.add_kerning(&x, &y);
736  }
737 
738  mFontManager.init_embedded_adaptors(pGlyph, x, y);
739  agg::rgba8 color(AGGColor(text.mFGColor, BlendWeight(pBlend)));
740 
741  if (fabs(weight) <= 0.01)
742  {
743  //for the sake of efficiency skip the contour converter if the weight is about zero.
744  mRasterizer.Rasterize(mFontCurvesTransformed, color, AGGBlendMode(pBlend));
745  }
746  else
747  {
748  mRasterizer.Rasterize(mFontContourTransformed, color, AGGBlendMode(pBlend));
749  }
750  }
751  x += pGlyph->advance_x;
752  y += pGlyph->advance_y;
753  }
754  y += text.mSize;
755  }
756  }
757  }
758  return false;
759 }
