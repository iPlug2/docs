415  if(pCellRect == mMouseCellBounds)

422  {
423  MenuPanel* pMenuPanel = mMenuPanels.Get(mr);
424

435  mActiveMenuPanel->mHighlightedCell = pCellRect;

439  // There is no MenuPanel for this menu, make a new one
440  if(pMenuPanelForThisMenu == nullptr) {
441  pMenuPanelForThisMenu = mMenuPanels.Add(new MenuPanel(*this, *pSubMenu, pCellRect->R + PAD, pCellRect->T, mMenuPanels.Find(mActiveMenuPanel)));
442  }
443 
444  for (auto mr = 0; mr < mMenuPanels.GetSize(); mr++)
445  {
446  MenuPanel* pMenuPanel = mMenuPanels.Get(mr);
447 
448  if(pMenuPanel->mShouldDraw)
449  {

475  }

537 {
538  IPopupMenu* pClickedMenu = &mActiveMenuPanel->mMenu;
539 
540  pClickedMenu->SetChosenItemIdx(-1);
541 
542  for(auto i = 0; i < mActiveMenuPanel->mCellBounds.GetSize(); i++)
543  {

545

547  {

550 
551  if(pItem->GetIsChoosable())

553  pClickedMenu->SetChosenItemIdx(itemChosen);
554  mActiveMenuPanel->mClickedCell = pR;
555  }
556  }

559  if(pClickedMenu->GetFunction())
560  pClickedMenu->ExecFunction();
561 
562  if(mCaller)
563  {
564  if(mIsContextMenu)

566  else
567  {

569  mCaller->OnPopupMenuSelection(nullptr);
570  else
571  mCaller->OnPopupMenuSelection(pClickedMenu);
572  }
573 
574  mIsContextMenu = false;
575  }
576 
577  mActiveMenuPanel = nullptr;
578 
579  mState = kFlickering;
580  Hide(true);
581  SetDirty(true); // triggers animation
582 }

584 void IPopupMenuControl::OnEndAnimation()
585 {
586 // DBGMSG("state %i\n", mState);
587 
588  if(mState == kExpanding)
589  {
590  for (auto i = 0; i < mMenuPanels.GetSize(); i++) {
591  mMenuPanels.Get(i)->mBlend.mWeight = mOpacity;
592  }
593 
594  mState = kExpanded;
595  }
596  else if (mState == kFlickering)
597  {
598  mState = kCollapsing;
599  SetDirty(true); // triggers animation again
600  return; // don't cancel animation
601  }
602  else if (mState == kSubMenuAppearing)
603  {
604  for (auto i = 0; i < mMenuPanels.GetSize(); i++) {
605  mMenuPanels.Get(i)->mBlend.mWeight = mOpacity;
606  }
607 
608  mState = kExpanded;
609  }
610  else if(mState == kCollapsing)
611  {
612  mTargetRECT = mSpecifiedCollapsedBounds;
613 
614  for (auto i = 0; i < mMenuPanels.GetSize(); i++) {
615  mMenuPanels.Get(i)->mBlend.mWeight = 0.;
616  }
617 
618  mState = kIdling;
619  mMouseCellBounds = nullptr;
620  mOriginalBounds = IRECT();
621 
622  SetDirty(true); // triggers animation again
623  return; // don't cancel animation
624  }
625  else if(mState == kIdling)
626  {
627  GetUI()->UpdateTooltips(); // will enable the tooltips
628 
629  mMenuPanels.Empty(true);
630  mRECT = mSpecifiedCollapsedBounds;
631  mState = kCollapsed;
632  }
633 
634  IControl::OnEndAnimation();
635 }
