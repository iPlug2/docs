19static int VSTSpkrArrType(int nchan)
20{
21 if (!nchan)
22 return kSpeakerArrEmpty;

30static int AsciiToVK(int ascii)
31{
32#ifdef OS_WIN
33 HKL layout = GetKeyboardLayout(0);
34 return VkKeyScanExA((CHAR)ascii, layout);
35#else
36 // Numbers and uppercase alpha chars map directly to VK
37 if ((ascii >= 0x30 && ascii <= 0x39) || (ascii >= 0x41 && ascii <= 0x5A))
38 {
39 return ascii;
40 }
41
42 // Lowercase alpha chars map to VK but need shifting
43 if (ascii >= 0x61 && ascii <= 0x7A)
44 {
45 return ascii - 0x20;
46 }
47
48 return kVK_NONE;
49#endif
50}

125IPlugVST2::IPlugVST2(const InstanceInfo& info, const Config& config)
126: IPlugAPIBase(config, kAPIVST2)
127, IPlugProcessor(config, kAPIVST2)
128, mHostCallback(info.mVSTHostCallback)
129{
130 Trace(TRACELOC, "%s", config.pluginName);
131
132 mHasVSTExtensions = VSTEXT_NONE;
133
134 int nInputs = MaxNChannels(ERoute::kInput), nOutputs = MaxNChannels(ERoute::kOutput);
135
136 memset(&mAEffect, 0, sizeof(AEffect));
137 mAEffect.object = this;
138 mAEffect.magic = kEffectMagic;
139 mAEffect.dispatcher = VSTDispatcher;
140 mAEffect.getParameter = VSTGetParameter;
141 mAEffect.setParameter = VSTSetParameter;
142 mAEffect.numPrograms = config.nPresets;
143 mAEffect.numParams = config.nParams;
144 mAEffect.numInputs = nInputs;
145 mAEffect.numOutputs = nOutputs;
146 mAEffect.uniqueID = config.uniqueID;
147 mAEffect.version = GetPluginVersion(true);
148 mAEffect.__ioRatioDeprecated = 1.0f;
149 mAEffect.__processDeprecated = VSTProcess;
150 mAEffect.processReplacing = VSTProcessReplacing;
151 mAEffect.processDoubleReplacing = VSTProcessDoubleReplacing;
152 mAEffect.initialDelay = config.latency;
153 mAEffect.flags = effFlagsCanReplacing | effFlagsCanDoubleReplacing;
154
155 if (config.plugDoesChunks)
156 {
157 mAEffect.flags |= effFlagsProgramChunks;
158 }
159 if (LegalIO(1, -1))
160 {
161 mAEffect.flags |= __effFlagsCanMonoDeprecated;
162 }
163 if (config.plugType == EIPlugPluginType::kInstrument)
164 {
165 mAEffect.flags |= effFlagsIsSynth;
166 }
167
168 memset(&mEditRect, 0, sizeof(ERect));
169 memset(&mInputSpkrArr, 0, sizeof(VstSpeakerArrangement));
170 memset(&mOutputSpkrArr, 0, sizeof(VstSpeakerArrangement));
171 mInputSpkrArr.numChannels = nInputs;
172 mOutputSpkrArr.numChannels = nOutputs;
173 mInputSpkrArr.type = VSTSpkrArrType(nInputs);
174 mOutputSpkrArr.type = VSTSpkrArrType(nOutputs);
175
176 // Default everything to connected, then disconnect pins if the host says to.
177 SetChannelConnections(ERoute::kInput, 0, nInputs, true);
178 SetChannelConnections(ERoute::kOutput, 0, nOutputs, true);
179
180 SetBlockSize(DEFAULT_BLOCK_SIZE);
181
182 if (config.plugHasUI)
183 {
184 mAEffect.flags |= effFlagsHasEditor;
185 UpdateEditRect();
186 }
187
188 CreateTimer();
189}

231bool IPlugVST2::SendVSTEvent(VstEvent& event)
232{
233 // It would be more efficient to bundle these and send at the end of a processed block,
234 // but that would require writing OnBlockEnd and making sure it always gets called,
235 // and who cares anyway, midi events aren't that dense.
236 VstEvents events;
237 memset(&events, 0, sizeof(VstEvents));
238 events.numEvents = 1;
239 events.events[0] = &event;
240 return (mHostCallback(&mAEffect, audioMasterProcessEvents, 0, 0, &events, 0.0f) == 1);
241}

243bool IPlugVST2::SendMidiMsg(const IMidiMsg& msg)
244{
245 VstMidiEvent midiEvent;
246 memset(&midiEvent, 0, sizeof(VstMidiEvent));
247
248 midiEvent.type = kVstMidiType;
249 midiEvent.byteSize = sizeof(VstMidiEvent); // Should this be smaller?
250 midiEvent.deltaFrames = msg.mOffset;
251 midiEvent.midiData[0] = msg.mStatus;
252 midiEvent.midiData[1] = msg.mData1;
253 midiEvent.midiData[2] = msg.mData2;
254
255 return SendVSTEvent((VstEvent&)midiEvent);
256}

258bool IPlugVST2::SendSysEx(const ISysEx& msg)
259{
260 VstMidiSysexEvent sysexEvent;
261 memset(&sysexEvent, 0, sizeof(VstMidiSysexEvent));
262
263 sysexEvent.type = kVstSysExType;
264 sysexEvent.byteSize = sizeof(VstMidiSysexEvent);
265 sysexEvent.deltaFrames = msg.mOffset;
266 sysexEvent.dumpBytes = msg.mSize;
267 sysexEvent.sysexDump = (char*)msg.mData;
268
269 return SendVSTEvent((VstEvent&)sysexEvent);

291VstIntPtr VSTCALLBACK IPlugVST2::VSTDispatcher(AEffect* pEffect, VstInt32 opCode, VstInt32 idx, VstIntPtr value, void* ptr, float opt)
292{
293 // VSTDispatcher is an IPlugVST class member, we can access anything in IPlugVST from here.
294 IPlugVST2* _this = (IPlugVST2*)pEffect->object;
295 if (!_this)
296 {
297 return 0;
298 }
299
300 // Handle a couple of opcodes here to make debugging easier.
301 // switch (opCode)
302 // {
303 // case effEditIdle:
304 // case __effIdleDeprecated:
305 // #ifdef USE_IDLE_CALLS
306 // _this->OnIdle();
307 // #endif
308 // return 0;
309 // }
310
311 Trace(TRACELOC, "%d(%s):%d:%d", opCode, VSTOpcodeStr(opCode), idx, (int)value);
312
313 switch (opCode)
314 {
315 case effOpen:
316 {
317 if (_this->GetHost() == kHostUninit)
318 {
319 char productStr[256];
320 productStr[0] = '\0';
321 int version = 0;
322 _this->mHostCallback(&_this->mAEffect, audioMasterGetProductString, 0, 0, productStr, 0.0f);
323
324 if (CStringHasContents(productStr))
325 {
326 int decVer = (int)_this->mHostCallback(&_this->mAEffect, audioMasterGetVendorVersion, 0, 0, 0, 0.0f);
327 int ver = decVer / 10000;
328 int rmaj = (decVer - 10000 * ver) / 100;
329 int rmin = (decVer - 10000 * ver - 100 * rmaj);
330 version = (ver << 16) + (rmaj << 8) + rmin;
331 }
332
333 _this->SetHost(productStr, version);
334 }
335 _this->OnParamReset(kReset);
336 return 0;
337 }
338 case effClose:
339 {
340 delete _this;
341 return 0;
342 }
343 case effGetParamLabel:
344 {
345 if (idx >= 0 && idx < _this->NParams())
346 {
347 ENTER_PARAMS_MUTEX_STATIC
348 strcpy((char*)ptr, _this->GetParam(idx)->GetLabel());
349 LEAVE_PARAMS_MUTEX_STATIC
350 }
351 return 0;
352 }
353 case effGetParamDisplay:
354 {
355 if (idx >= 0 && idx < _this->NParams())
356 {
357 ENTER_PARAMS_MUTEX_STATIC
358 _this->GetParam(idx)->GetDisplay(_this->mParamDisplayStr);
359 LEAVE_PARAMS_MUTEX_STATIC
360 strcpy((char*)ptr, _this->mParamDisplayStr.Get());
361 }
362 return 0;
363 }
364 case effGetParamName:
365 {
366 if (idx >= 0 && idx < _this->NParams())
367 {
368 ENTER_PARAMS_MUTEX_STATIC
369 strcpy((char*)ptr, _this->GetParam(idx)->GetName());
370 LEAVE_PARAMS_MUTEX_STATIC
371 }
372 return 0;
373 }
374 case effGetParameterProperties:
375 {
376 if (idx >= 0 && idx < _this->NParams())
377 {
378 VstParameterProperties* props = (VstParameterProperties*)ptr;
379
380 ENTER_PARAMS_MUTEX_STATIC
381 IParam* pParam = _this->GetParam(idx);
382 switch (pParam->Type())
383 {
384 case IParam::kTypeInt:
385 case IParam::kTypeEnum:
386 props->flags = kVstParameterUsesIntStep | kVstParameterUsesIntegerMinMax;
387 props->minInteger = (int)pParam->GetMin();
388 props->maxInteger = (int)pParam->GetMax();
389 props->stepInteger = props->largeStepInteger = 1;
390 break;
391 case IParam::kTypeBool: props->flags = kVstParameterIsSwitch; break;
392 case IParam::kTypeDouble:
393 default:
394 props->flags = kVstParameterUsesFloatStep;
395 props->largeStepFloat = props->smallStepFloat = props->stepFloat = (float)pParam->GetStep();
396 break;
397 }
398
399 strcpy(props->label, pParam->GetLabel());
400 LEAVE_PARAMS_MUTEX_STATIC
401
402 return 1;
403 }
404 return 0;
405 }
406 case effString2Parameter:
407 {
408 if (idx >= 0 && idx < _this->NParams())
409 {
410 if (ptr)
411 {
412 ENTER_PARAMS_MUTEX_STATIC
413 IParam* pParam = _this->GetParam(idx);
414 const double v = pParam->StringToValue((const char*)ptr);
415 pParam->Set(v);
416 _this->SendParameterValueFromAPI(idx, v, false);
417 _this->OnParamChange(idx, kHost);
418 LEAVE_PARAMS_MUTEX_STATIC
419 }
420 return 1;
421 }
422 return 0;
423 }
424 case effSetSampleRate:
425 {
426 _this->SetSampleRate(opt);
427 _this->OnReset();
428 return 0;
429 }
430 case effSetBlockSize:
431 {
432 _this->SetBlockSize((int)value);
433 _this->OnReset();
434 return 0;
435 }
436 case effMainsChanged:
437 {
438 if (!value)
439 {
440 _this->OnActivate(false);
441 _this->OnReset();
442 }
443 else
444 {
445 _this->OnActivate(true);
446 }
447 return 0;
448 }
449 case effEditGetRect:
450 {
451 if (ptr && _this->HasUI())
452 {
453 _this->UpdateEditRect();
454 *(ERect**)ptr = &(_this->mEditRect);
455 return 1;
456 }
457 ptr = 0;
458 return 0;
459 }
460 case effEditOpen:
461 {
462#if defined OS_WIN || defined ARCH_64BIT
463 if (_this->OpenWindow(ptr))
464 {
465 return 1;
466 }
467#else // OSX 32 bit, check if we are in a Cocoa VST host, otherwise tough luck
468 bool iscocoa = (_this->mHasVSTExtensions & VSTEXT_COCOA);
469 if (iscocoa && _this->OpenWindow(ptr))
470 {
471 return 1; // cocoa supported open cocoa
472 }
473#endif
474 return 0;
475 }
476 case effEditClose:
477 {
478 if (_this->HasUI())
479 {
480 _this->CloseWindow();
481 return 1;
482 }
483 return 0;
484 }
485 case __effIdentifyDeprecated:
486 {
487 return 'NvEf'; // Random deprecated magic.
488 }
489 case effGetChunk:
490 {
491 uint8_t** ppData = (uint8_t**)ptr;
492 if (ppData)
493 {
494 bool isBank = (!idx);
495 IByteChunk& chunk = (isBank ? _this->mBankState : _this->mState);
496 IByteChunk::InitChunkWithIPlugVer(chunk);
497 bool savedOK = true;
498
499 if (isBank)
500 {
501 _this->ModifyCurrentPreset();
502 savedOK = static_cast<IPluginBase*>(_this)->SerializePresets(chunk);
503 }
504 else
505 {
506 savedOK = _this->SerializeState(chunk);
507 }
508
509 if (savedOK && chunk.Size())
510 {
511 *ppData = chunk.GetData();
512 return chunk.Size();
513 }
514 }
515 return 0;
516 }
517 case effSetChunk:
518 {
519 if (ptr)
520 {
521 bool isBank = (!idx);
522 IByteChunk& chunk = (isBank ? _this->mBankState : _this->mState);
523 chunk.Resize((int)value);
524 memcpy(chunk.GetData(), ptr, value);
525 int pos = 0;
526 int iplugVer = IByteChunk::GetIPlugVerFromChunk(chunk, pos);
527 isBank &= (iplugVer >= 0x010000);
528
529 if (isBank)
530 {
531 pos = static_cast<IPluginBase*>(_this)->UnserializePresets(chunk, pos);
532 }
533 else
534 {
535 pos = _this->UnserializeState(chunk, pos);
536 _this->ModifyCurrentPreset();
537 }
538
539 if (pos >= 0)
540 {
541 _this->OnRestoreState();
542 return 1;
543 }
544 }
545 return 0;
546 }
547 case effProcessEvents:
548 {
549 VstEvents* pEvents = (VstEvents*)ptr;
550 if (pEvents)
551 {
552 for (int i = 0; i < pEvents->numEvents; ++i)
553 {
554 VstEvent* pEvent = pEvents->events[i];
555 if (pEvent)
556 {
557 if (pEvent->type == kVstMidiType)
558 {
559 VstMidiEvent* pME = (VstMidiEvent*)pEvent;
560 IMidiMsg msg(pME->deltaFrames, pME->midiData[0], pME->midiData[1], pME->midiData[2]);
561 _this->ProcessMidiMsg(msg);
562 _this->mMidiMsgsFromProcessor.Push(msg);
563
564 // #ifdef TRACER_BUILD
565 // msg.LogMsg();
566 // #endif
567 }
568 else if (pEvent->type == kVstSysExType)
569 {
570 VstMidiSysexEvent* pSE = (VstMidiSysexEvent*)pEvent;
571 ISysEx sysex(pSE->deltaFrames, (const uint8_t*)pSE->sysexDump, pSE->dumpBytes);
572 _this->ProcessSysEx(sysex);
573 }
574 }
575 }
576 return 1;
577 }
578 return 0;
579 }
580 case effCanBeAutomated:
581 {
582 if (idx >= 0 && idx < _this->NParams())
583 {
584 return _this->GetParam(idx)->GetCanAutomate();
585 }
586 }
587 case effGetInputProperties:
588 {
589 if (ptr && idx >= 0 && idx < _this->MaxNChannels(ERoute::kInput))
590 {
591 VstPinProperties* pp = (VstPinProperties*)ptr;
592 pp->flags = kVstPinIsActive;
593
594 if (!(idx % 2) && idx < _this->MaxNChannels(ERoute::kInput) - 1)
595 pp->flags |= kVstPinIsStereo;
596
597 if (_this->GetChannelLabel(ERoute::kInput, idx).GetLength())
598 sprintf(pp->label, "%s", _this->GetChannelLabel(ERoute::kInput, idx).Get());
599 else
600 sprintf(pp->label, "Input %d", idx + 1);
601
602 return 1;
603 }
604 return 0;
605 }
606 case effGetOutputProperties:
607 {
608 if (ptr && idx >= 0 && idx < _this->MaxNChannels(ERoute::kOutput))
609 {
610 VstPinProperties* pp = (VstPinProperties*)ptr;
611 pp->flags = kVstPinIsActive;
612
613 if (!(idx % 2) && idx < _this->MaxNChannels(ERoute::kOutput) - 1)
614 pp->flags |= kVstPinIsStereo;
615
616 if (_this->GetChannelLabel(ERoute::kOutput, idx).GetLength())
617 sprintf(pp->label, "%s", _this->GetChannelLabel(ERoute::kOutput, idx).Get());
618 else
619 sprintf(pp->label, "Output %d", idx + 1);
620
621 return 1;
622 }
623 return 0;
624 }
625 case effGetPlugCategory:
626 {
627 if (_this->IsInstrument())
628 return kPlugCategSynth;
629 return kPlugCategEffect;
630 }
631 case effProcessVarIo:
632 {
633 // VstVariableIo* pIO = (VstVariableIo*) ptr; // For offline processing (of audio files?)
634 return 0;
635 }
636 case effSetSpeakerArrangement:
637 {
638 VstSpeakerArrangement* pInputArr = (VstSpeakerArrangement*)value;
639 VstSpeakerArrangement* pOutputArr = (VstSpeakerArrangement*)ptr;
640
641 if (pInputArr)
642 {
643 int n = pInputArr->numChannels;
644
645 // For a mono-in plug-in in Reaper reject effSetSpeakerArrangement passed due to wantsChannelCountNotifications
646 if (n > _this->mAEffect.numInputs)
647 return 0;
648
649 _this->SetChannelConnections(ERoute::kInput, 0, n, true);
650 _this->SetChannelConnections(ERoute::kInput, n, _this->MaxNChannels(ERoute::kInput) - n, false);
651 }
652 if (pOutputArr)
653 {
654 int n = pOutputArr->numChannels;
655
656 // For a mono-out plug-in in Reaper reject effSetSpeakerArrangement passed due to wantsChannelCountNotifications
657 if (n > _this->mAEffect.numOutputs)
658 return 0;
659
660 _this->SetChannelConnections(ERoute::kOutput, 0, n, true);
661 _this->SetChannelConnections(ERoute::kOutput, n, _this->MaxNChannels(ERoute::kOutput) - n, false);
662 }
663 return 1;
664 }
665 case effGetSpeakerArrangement:
666 {
667 VstSpeakerArrangement** ppInputArr = (VstSpeakerArrangement**)value;
668 VstSpeakerArrangement** ppOutputArr = (VstSpeakerArrangement**)ptr;
669 if (ppInputArr)
670 {
671 *ppInputArr = &(_this->mInputSpkrArr);
672 }
673 if (ppOutputArr)
674 {
675 *ppOutputArr = &(_this->mOutputSpkrArr);
676 }
677 return 1;
678 }
679 case effGetEffectName:
680 {
681 if (ptr)
682 {
683 strcpy((char*)ptr, _this->GetPluginName());
684 return 1;
685 }
686 return 0;
687 }
688 case effGetProductString:
689 {
690 if (ptr)
691 {
692 strcpy((char*)ptr, _this->GetProductName());
693 return 1;
694 }
695 return 0;
696 }
697 case effGetVendorString:
698 {
699 if (ptr)
700 {
701 strcpy((char*)ptr, _this->GetMfrName());
702 return 1;
703 }
704 return 0;
705 }
706 case effGetVendorVersion:
707 {
708 return _this->GetPluginVersion(true);
709 }
710 case effCanDo:
711 {
712 if (ptr)
713 {
714 Trace(TRACELOC, "VSTCanDo(%s)", (char*)ptr);
715 if (!strcmp((char*)ptr, "receiveVstTimeInfo"))
716 {
717 return 1;
718 }
719 if (_this->DoesMIDIIn())
720 {
721 if (!strcmp((char*)ptr, "receiveVstEvents") || !strcmp((char*)ptr, "receiveVstMidiEvent"))
722 {
723 return 1;
724 }
725 }
