57 struct IByteGetter
58 {
59  static inline int GetBytes(const uint8_t* pData, int dataSize, void* pBuf, int size, int startPos)
60  {
61  int endPos = startPos + size;
62  if (startPos >= 0 && endPos <= dataSize)
63  {
64  memcpy(pBuf, pData + startPos, size);
65  return endPos;
66  }
67  return -1;
68  }
69 
70  static inline int GetStr(const uint8_t* pData, int dataSize, WDL_String& str, int startPos)
71  {
72  int len;
73  int strStartPos = GetBytes(pData, dataSize, &len, sizeof(len), startPos);
74  if (strStartPos >= 0)
75  {
76  int strEndPos = strStartPos + len;
77  if (strEndPos <= dataSize)
78  {
79  if (len > 0)
80  str.Set((char*) (pData + strStartPos), len);
81  else
82  str.Set("");
83  }
84  return strEndPos;
85  }
86  return -1;
87  }
