66struct IByteGetter
67{
76 static inline int GetBytes(const uint8_t* pSrc, int srcSize, void* pDst, int nBytesToCopy, int startPos)
77 {
78 int endPos = startPos + nBytesToCopy;
79 if (startPos >= 0 && endPos <= srcSize)
80 {
81 memcpy(pDst, pSrc + startPos, nBytesToCopy);
82 return endPos;
83 }
84 return -1;
85 }
86
94 static inline int GetStr(const uint8_t* pSrc, int srcSize, WDL_String& str, int startPos)
95 {
96 int len;
97 int strStartPos = GetBytes(pSrc, srcSize, &len, sizeof(len), startPos);
98 if (strStartPos >= 0)
99 {
100 int strEndPos = strStartPos + len;
101 if (strEndPos <= srcSize)
102 {
103 if (len > 0)
104 str.Set((char*)(pSrc + strStartPos), len);
105 else
106 str.Set("");
107 }
108 return strEndPos;
109 }
110 return -1;
111 }
112};

137 static int GetIPlugVerFromChunk(const IByteChunk& chunk, int& position)
138 {
139 int magic = 0, ver = 0;
140 int magicpos = chunk.Get(&magic, position);
141
142 if (magicpos > position && magic == IPLUG_VERSION_MAGIC)
143 position = chunk.Get(&ver, magicpos);
144
145 return ver;
146 }

152 inline int PutBytes(const void* pSrc, int nBytesToCopy)
153 {
154 int n = mBytes.GetSize();
155 mBytes.Resize(n + nBytesToCopy);
156 memcpy(mBytes.Get() + n, pSrc, nBytesToCopy);
157 return mBytes.GetSize();
158 }
159
166 inline int GetBytes(void* pDst, int nBytesToCopy, int startPos) const
167 {
168 return IByteGetter::GetBytes(mBytes.Get(), Size(), pDst, nBytesToCopy, startPos);
169 }
170
175 template <class T>
176 inline int Put(const T* pVal)
177 {
178 return PutBytes(pVal, sizeof(T));
179 }
180
187 template <class T>
188 inline int Get(T* pDst, int startPos) const
189 {
190 return GetBytes(pDst, sizeof(T), startPos);
191 }
192
196 inline int PutStr(const char* str)
197 {
198 int slen = (int)strlen(str);
199 Put(&slen);
200 return PutBytes(str, slen);
201 }
202
208 inline int GetStr(WDL_String& str, int startPos) const
209 {
210 return IByteGetter::GetStr(mBytes.Get(), Size(), str, startPos);
211 }
212
216 inline int PutChunk(const IByteChunk* pRHS) { return PutBytes(pRHS->GetData(), pRHS->Size()); }
217
219 inline void Clear() { mBytes.Resize(0); }
220
223 inline int Size() const { return mBytes.GetSize(); }
224
228 inline int Resize(int newSize)
229 {
230 int n = mBytes.GetSize();
231 mBytes.Resize(newSize);
232 if (newSize > n)
233 {
234 memset(mBytes.Get() + n, 0, (newSize - n));
235 }
236 return n;
237 }
238
241 inline uint8_t* GetData() { return mBytes.Get(); }
242
245 inline const uint8_t* GetData() const { return mBytes.Get(); }
246
250 inline bool IsEqual(IByteChunk& otherChunk) const
251 {
252 return (otherChunk.Size() == Size() && !memcmp(otherChunk.mBytes.Get(), mBytes.Get(), Size()));
253 }
254
255private:
256 WDL_TypedBuf<uint8_t> mBytes;

486struct IOConfig
487{
488 WDL_PtrList<IBusInfo> mBusInfo[2]; // A particular valid io config may have multiple input buses or output busses
489

491 {
492 mBusInfo[0].Empty(true);

500 void AddBusInfo(ERoute direction, int NChans) { mBusInfo[direction].Add(new IBusInfo(direction, NChans)); }
501
506 const IBusInfo* GetBusInfo(ERoute direction, int index) const
507 {
508 assert(index >= 0 && index < mBusInfo[direction].GetSize());
509 return mBusInfo[direction].Get(index);
510 }
511
516 int NChansOnBusSAFE(ERoute direction, int index) const
517 {
518 int NChans = 0;
519
520 if (index >= 0 && index < mBusInfo[direction].GetSize())
521 NChans = mBusInfo[direction].Get(index)->NChans();
522
523 return NChans;
524 }
525
529 int NBuses(ERoute direction) const { return mBusInfo[direction].GetSize(); }
530
534 int GetTotalNChannels(ERoute direction) const
535 {
536 int total = 0;
537
538 for (int i = 0; i < mBusInfo[direction].GetSize(); i++)
539 total += mBusInfo[direction].Get(i)->NChans();
540
541 return total;
542 }
543
548 bool ContainsWildcard(ERoute direction) const
549 {
550 for (auto i = 0; i < mBusInfo[direction].GetSize(); i++)
551 {
552 if (mBusInfo[direction].Get(i)->NChans() < 0)
553 return true;
554 }
555
556 return false;
557 }
