159 void IGraphicsCanvas::SetCanvasSourcePattern(val& context, const IPattern& pattern, const IBlend* pBlend)
160 {
161  SetCanvasBlendMode(pBlend);
162 
163  switch (pattern.mType)
164  {
165  case kSolidPattern:
166  {
167  const IColor color = pattern.GetStop(0).mColor;
168  std::string colorString = CanvasColor(color, BlendWeight(pBlend));
169 
170  context.set("fillStyle", colorString);
171  context.set("strokeStyle", colorString);
172  }
173  break;
174 
175  case kLinearPattern:
176  case kRadialPattern:
177  {
178  double x, y;
179  IMatrix m = IMatrix(pattern.mTransform).Invert();
180  m.TransformPoint(x, y, 0.0, 1.0);
181 
182  val gradient = (pattern.mType == kLinearPattern) ?
183  context.call<val>("createLinearGradient", m.mTX, m.mTY, x, y) :
184  context.call<val>("createRadialGradient", m.mTX, m.mTY, 0.0, m.mTX, m.mTY, m.mXX);
185 
186  /*
187  switch (pattern.mExtend)
188  {
189  case kExtendNone: cairo_pattern_set_extend(cairoPattern, CAIRO_EXTEND_NONE); break;
190  case kExtendPad: cairo_pattern_set_extend(cairoPattern, CAIRO_EXTEND_PAD); break;
191  case kExtendReflect: cairo_pattern_set_extend(cairoPattern, CAIRO_EXTEND_REFLECT); break;
192  case kExtendRepeat: cairo_pattern_set_extend(cairoPattern, CAIRO_EXTEND_REPEAT); break;
193  }*/
194 
195  for (int i = 0; i < pattern.NStops(); i++)
196  {
197  const IColorStop& stop = pattern.GetStop(i);
198  gradient.call<void>("addColorStop", stop.mOffset, CanvasColor(stop.mColor));
199  }
200 
201  context.set("fillStyle", gradient);
202  context.set("strokeStyle", gradient);
203  }
204  break;
205  }

235 bool IGraphicsCanvas::DoDrawMeasureText(const IText& text, const char* str, IRECT& bounds, const IBlend* pBlend, bool measure)
236 {
237  // TODO: orientation
238  val context = GetContext();
239  std::string textString(str);
240 
241  char fontString[FONT_LEN + 64];
242  const char* styles[] = { "normal", "bold", "italic" };
243  context.set("textBaseline", std::string("top"));
244  val font = context["font"];
245  sprintf(fontString, "%s %dpx %s", styles[text.mStyle], text.mSize, text.mFont);
246  context.set("font", std::string(fontString));
247  double textWidth = context.call<val>("measureText", textString)["width"].as<double>();
248  double textHeight = EM_ASM_DOUBLE({
249  return parseFloat(document.getElementById("canvas").getContext("2d").font);
250  });
251 
252  if (measure)
253  {
254  bounds = IRECT(0, 0, (float) textWidth, (float) textHeight);
255  return true;
256  }
257  else
258  {
259  double x = bounds.L;
260  double y = bounds.T;
261 
262  switch (text.mAlign)
263  {
264  case IText::kAlignNear: break;
265  case IText::kAlignCenter: x = bounds.MW() - (textWidth / 2.0); break;
266  case IText::kAlignFar: x = bounds.R - textWidth; break;
267  }
268 
269  switch (text.mVAlign)
270  {
271  case IText::EVAlign::kVAlignTop: y = bounds.T; break;
272  case IText::EVAlign::kVAlignMiddle: y = bounds.MH() - (textHeight/2.); break;
273  case IText::EVAlign::kVAlignBottom: y = bounds.B - textHeight; break;
274  default: break;
275  }
276 
277  context.call<void>("save");
278  PathRect(bounds);
279  context.call<void>("clip");
280  PathClear();
281  SetCanvasSourcePattern(context, text.mFGColor, pBlend);
282  context.call<void>("fillText", textString, x, y);
283  context.call<void>("restore");
284  }
285 
286  return true;
287 }

379 void IGraphicsCanvas::ApplyShadowMask(ILayerPtr& layer, RawBitmapData& mask, const IShadow& shadow)
380 {
381  const APIBitmap* pBitmap = layer->GetAPIBitmap();
382  int size = pBitmap->GetWidth() * pBitmap->GetHeight() * 4;
383 
384  if (mask.GetSize() >= size)
385  {
386  int width = pBitmap->GetWidth();
387  int height = pBitmap->GetHeight();
388  double scale = pBitmap->GetScale() * pBitmap->GetDrawScale();
389  double x = shadow.mXOffset * scale;
390  double y = shadow.mYOffset * scale;
391  val layerCanvas = *pBitmap->GetBitmap();
392  val layerContext = layerCanvas.call<val>("getContext", std::string("2d"));
393  layerContext.call<void>("setTransform");
394 
395  if (!shadow.mDrawForeground)
396  {
397  layerContext.call<void>("clearRect", 0, 0, width, height);
398  }
399 
400  CanvasBitmap localBitmap(width, height, pBitmap->GetScale(), pBitmap->GetDrawScale());
401  val localCanvas = *localBitmap.GetBitmap();
402  val localContext = localCanvas.call<val>("getContext", std::string("2d"));
403  val imageData = localContext.call<val>("createImageData", width, height);
404  val pixelData = imageData["data"];
405  unsigned char* in = mask.Get();
406 
407  for (auto i = 0; i < size; i++)
408  pixelData.set(i, in[i]);
409 
410  localContext.call<void>("putImageData", imageData, 0, 0);
411  IBlend blend(kBlendNone, shadow.mOpacity);
412  localContext.call<void>("rect", 0, 0, width, height);
413  localContext.call<void>("scale", scale, scale);
414  localContext.call<void>("translate", -(layer->Bounds().L + shadow.mXOffset), -(layer->Bounds().T + shadow.mYOffset));
415  SetCanvasSourcePattern(localContext, shadow.mPattern, &blend);
416  localContext.set("globalCompositeOperation", "source-in");
417  localContext.call<void>("fill");
418 
419  layerContext.set("globalCompositeOperation", "destination-over");
420  layerContext.call<void>("drawImage", localCanvas, 0, 0, width, height, x, y, width, height);
421  }
