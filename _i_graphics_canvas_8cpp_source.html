103void IGraphicsCanvas::DrawBitmap(const IBitmap& bitmap, const IRECT& bounds, int srcX, int srcY, const IBlend* pBlend)
104{
105 val context = GetContext();
106 val img = *bitmap.GetAPIBitmap()->GetBitmap();
107 context.call<void>("save");
108 SetCanvasBlendMode(context, pBlend);
109 context.set("globalAlpha", BlendWeight(pBlend));
110
111 const float bs = bitmap.GetScale();
112 IRECT sr = bounds;
113 sr.Scale(bs * bitmap.GetDrawScale());
114
115 PathRect(bounds);
116 context.call<void>("clip");
117 context.call<void>("drawImage", img, srcX * bs, srcY * bs, sr.W(), sr.H(), bounds.L, bounds.T, bounds.W(), bounds.H());
118 GetContext().call<void>("restore");
119 PathClear();
120}

139void IGraphicsCanvas::PathStroke(const IPattern& pattern, float thickness, const IStrokeOptions& options, const IBlend* pBlend)
140{
141 val context = GetContext();
142
143 switch (options.mCapOption)
144 {
145 case ELineCap::Butt: context.set("lineCap", "butt"); break;
146 case ELineCap::Round: context.set("lineCap", "round"); break;
147 case ELineCap::Square: context.set("lineCap", "square"); break;
148 }
149
150 switch (options.mJoinOption)
151 {
152 case ELineJoin::Miter: context.set("lineJoin", "miter"); break;
153 case ELineJoin::Round: context.set("lineJoin", "round"); break;
154 case ELineJoin::Bevel: context.set("lineJoin", "bevel"); break;
155 }
156
157 context.set("miterLimit", options.mMiterLimit);
158
159 val dashArray = val::array();
160
161 for (int i = 0; i < options.mDash.GetCount(); i++)
162 dashArray.call<void>("push", val(*(options.mDash.GetArray() + i)));
163
164 context.call<void>("setLineDash", dashArray);
165 context.set("lineDashOffset", options.mDash.GetOffset());
166 context.set("lineWidth", thickness);
167
168 SetCanvasSourcePattern(context, pattern, pBlend);
169
170 context.call<void>("stroke");
171
172 if (!options.mPreserve)
173 PathClear();
174}

189void IGraphicsCanvas::SetCanvasSourcePattern(val& context, const IPattern& pattern, const IBlend* pBlend)
190{
191 SetCanvasBlendMode(context, pBlend);
192
193 switch (pattern.mType)
194 {
195 case EPatternType::Solid:
196 {
197 const IColor color = pattern.GetStop(0).mColor;
198 std::string colorString = CanvasColor(color, BlendWeight(pBlend));
199
200 context.set("fillStyle", colorString);
201 context.set("strokeStyle", colorString);
202 }
203 break;
204
205 case EPatternType::Linear:
206 case EPatternType::Radial:
207 {
208 double x, y;
209 IMatrix m = IMatrix(pattern.mTransform).Invert();
210 m.TransformPoint(x, y, 0.0, 1.0);
211
212 val gradient =
213 (pattern.mType == EPatternType::Linear) ? context.call<val>("createLinearGradient", m.mTX, m.mTY, x, y) : context.call<val>("createRadialGradient", m.mTX, m.mTY, 0.0, m.mTX, m.mTY, m.mXX);
214
215 for (int i = 0; i < pattern.NStops(); i++)
216 {
217 const IColorStop& stop = pattern.GetStop(i);
218 gradient.call<void>("addColorStop", stop.mOffset, CanvasColor(stop.mColor));
219 }
220
221 context.set("fillStyle", gradient);
222 context.set("strokeStyle", gradient);
223 }
224 break;
225 default: break;
226 }
227}

230{

240 case EBlend::DstOver: context.set("globalCompositeOperation", "destination-over"); break;
241 case EBlend::DstIn: context.set("globalCompositeOperation", "destination-in"); break;
242 case EBlend::DstOut: context.set("globalCompositeOperation", "destination-out"); break;

249void IGraphicsCanvas::PrepareAndMeasureText(const IText& text, const char* str, IRECT& r, double& x, double& y) const
250{
251 StaticStorage<Font>::Accessor storage(sFontCache);
252 Font* pFont = storage.Find(text.mFont);
253
254 assert(pFont && "No font found - did you forget to load it?");
255
256 FontDescriptor descriptor = &pFont->mDescriptor;
257 val context = GetContext();
258 std::string fontString = GetFontString(descriptor->first.Get(), descriptor->second.Get(), text.mSize * pFont->mEMRatio);
259
260 context.set("font", fontString);
261
262 const double textWidth = context.call<val>("measureText", std::string(str))["width"].as<double>();
263 const double textHeight = text.mSize;
264 const double ascender = pFont->mAscenderRatio * textHeight;
265 const double descender = -(1.0 - pFont->mAscenderRatio) * textHeight;
266
267 switch (text.mAlign)
268 {
269 case EAlign::Near: x = r.L; break;
270 case EAlign::Center: x = r.MW() - (textWidth / 2.0); break;
271 case EAlign::Far: x = r.R - textWidth; break;
272 }
273
274 switch (text.mVAlign)
275 {
276 case EVAlign::Top: y = r.T + ascender; break;
277 case EVAlign::Middle: y = r.MH() + descender + (textHeight / 2.0); break;
278 case EVAlign::Bottom: y = r.B + descender; break;
279 }
280
281 r = IRECT((float)x, (float)(y - ascender), (float)(x + textWidth), (float)(y + textHeight - ascender));
282}

339APIBitmap* IGraphicsCanvas::LoadAPIBitmap(const char* name, const void* pData, int dataSize, int scale)
340{
341 int width = 0;
342 int height = 0;
343 int channels = 0;
344 uint8_t* pRGBA;
345
346 pRGBA = stbi_load_from_memory((const uint8_t*)pData, dataSize, &width, &height, &channels, 4);
347 if (!pRGBA)
348 {
349 return nullptr;
350 }
351
352 DBGMSG("Size: %d x %d\n", width, height);
353 // Now that we've decoded the data, put it on a canvas
354 int rgbaSize = width * height * channels;
355 val rgbaArray = val::global("Uint8ClampedArray").new_(val(emscripten::typed_memory_view(rgbaSize, pRGBA)));
356
357 val imgData = val::global("ImageData").new_(rgbaArray, val(width), val(height));
358 val canvas = val::global("document").call<val>("createElement", std::string("canvas"));
359 canvas.set("width", width);
360 canvas.set("height", height);
361
362 val ctx = canvas.call<val>("getContext", std::string("2d"));
363 ctx.call<void>("putImageData", imgData, 0, 0);
364
365 stbi_image_free(pRGBA);
366
367 return new Bitmap(canvas, name, scale);
368}

372void IGraphicsCanvas::GetFontMetrics(const char* font, const char* style, double& ascenderRatio, double& EMRatio)
373{
374 // Provides approximate font metrics for a system font (until text metrics are properly supported)
375 int size = 1000;
376 std::string fontString = GetFontString(font, style, size);
377
378 val document = val::global("document");
379 val textSpan = document.call<val>("createElement", std::string("span"));
380 textSpan.set("innerHTML", std::string("M"));
381 textSpan["style"].set("font", fontString);
382
383 val block = document.call<val>("createElement", std::string("div"));
384 block["style"].set("display", std::string("inline-block"));
385 block["style"].set("width", std::string("1px"));
386 block["style"].set("height", std::string("0px"));
387
388 val div = document.call<val>("createElement", std::string("div"));
389 div.call<void>("appendChild", textSpan);
390 div.call<void>("appendChild", block);
391 document["body"].call<void>("appendChild", div);
392
393 block["style"].set("vertical-align", std::string("baseline"));
394 double ascent = block["offsetTop"].as<double>() - textSpan["offsetTop"].as<double>();
395 double height = textSpan.call<val>("getBoundingClientRect")["height"].as<double>();
396 document["body"].call<void>("removeChild", div);
397
398 EMRatio = size / height;
399 ascenderRatio = ascent / height;
400}

402bool IGraphicsCanvas::CompareFontMetrics(const char* style, const char* font1, const char* font2)
403{
404 WDL_String fontCombination;
405 fontCombination.SetFormatted(FONT_LEN * 2 + 2, "%s, %s", font1, font2);
406 val context = GetContext();
407 std::string textString("@BmwdWMoqPYyzZr1234567890.+-=_~'");
408 const int size = 72;
409
410 context.set("font", GetFontString(font2, style, size));
411 val metrics1 = context.call<val>("measureText", textString);
412
413 context.set("font", GetFontString(fontCombination.Get(), style, size));
414 val metrics2 = context.call<val>("measureText", textString);
415
416 return metrics1["width"].as<double>() == metrics2["width"].as<double>();
417}

424bool IGraphicsCanvas::LoadAPIFont(const char* fontID, const PlatformFontPtr& font)
425{
426 StaticStorage<Font>::Accessor storage(sFontCache);
427
428 if (storage.Find(fontID))
429 return true;
430
431 if (!font->IsSystem())
432 {
433 IFontDataPtr data = font->GetFontData();
434
435 if (data->IsValid())
436 {
437 // Load the font from data
438
439 val fontData = val(typed_memory_view(data->GetSize(), data->Get()));
440 val fontFace = val::global("FontFace").new_(std::string(fontID), fontData);
441
442 FontDescriptor descriptor = font->GetDescriptor();
443 const double ascenderRatio = data->GetAscender() / static_cast<double>(data->GetAscender() - data->GetDescender());
444 const double EMRatio = data->GetHeightEMRatio();
445 storage.Add(new Font({descriptor->first, descriptor->second}, ascenderRatio, EMRatio), fontID);
446
447 // Add to store and request load immediately
448
449 fontFace.call<val>("load");
450 val::global("document")["fonts"].call<void>("add", fontFace);
451 mLoadingFonts.push_back(fontFace);
452
453 return true;
454 }
455 }
456 else
457 {
458 FontDescriptor descriptor = font->GetDescriptor();
459 const char* fontName = descriptor->first.Get();
460 const char* styleName = descriptor->second.Get();
461
462 if (FontExists(fontName, styleName))
463 {
464 double ascenderRatio, EMRatio;
465
466 GetFontMetrics(descriptor->first.Get(), descriptor->second.Get(), ascenderRatio, EMRatio);
467 storage.Add(new Font({descriptor->first, descriptor->second}, ascenderRatio, EMRatio), fontID);
468 return true;
469 }
470 }
471
472 return false;
473}

512void IGraphicsCanvas::ApplyShadowMask(ILayerPtr& layer, RawBitmapData& mask, const IShadow& shadow)
513{
514 const APIBitmap* pBitmap = layer->GetAPIBitmap();
515 int size = pBitmap->GetWidth() * pBitmap->GetHeight() * 4;
516
517 if (mask.GetSize() >= size)
518 {
519 int width = pBitmap->GetWidth();
520 int height = pBitmap->GetHeight();
521 double scale = pBitmap->GetScale() * pBitmap->GetDrawScale();
522 double x = shadow.mXOffset * scale;
523 double y = shadow.mYOffset * scale;
524 val layerCanvas = *pBitmap->GetBitmap();
525 val layerContext = layerCanvas.call<val>("getContext", std::string("2d"));
526 layerContext.call<void>("setTransform");
527
528 if (!shadow.mDrawForeground)
529 {
530 layerContext.call<void>("clearRect", 0, 0, width, height);
531 }
532
533 Bitmap localBitmap(width, height, pBitmap->GetScale(), pBitmap->GetDrawScale());
534 val localCanvas = *localBitmap.GetBitmap();
535 val localContext = localCanvas.call<val>("getContext", std::string("2d"));
536 val imageData = localContext.call<val>("createImageData", width, height);
537 val pixelData = imageData["data"];
538 unsigned char* in = mask.Get();
539
540 for (auto i = 0; i < size; i++)
541 pixelData.set(i, in[i]);
542
543 localContext.call<void>("putImageData", imageData, 0, 0);
544 IBlend blend(EBlend::SrcIn, shadow.mOpacity);
545 localContext.call<void>("rect", 0, 0, width, height);
546 localContext.call<void>("scale", scale, scale);
547 localContext.call<void>("translate", -(layer->Bounds().L + shadow.mXOffset), -(layer->Bounds().T + shadow.mYOffset));
548 SetCanvasSourcePattern(localContext, shadow.mPattern, &blend);
549 localContext.call<void>("fill");
550
551 layerContext.set("globalCompositeOperation", "destination-over");
552 layerContext.call<void>("drawImage", localCanvas, 0, 0, width, height, x, y, width, height);
553 }
