33// Unicode helpers
34void UTF8ToUTF16(wchar_t* utf16Str, const char* utf8Str, int maxLen)
35{
36 int requiredSize = MultiByteToWideChar(CP_UTF8, 0, utf8Str, -1, NULL, 0);
37
38 if (requiredSize > 0 && requiredSize <= maxLen)
39 {
40 MultiByteToWideChar(CP_UTF8, 0, utf8Str, -1, utf16Str, requiredSize);
41 return;
42 }
43
44 utf16Str[0] = 0;
45}

47void UTF16ToUTF8(WDL_String& utf8Str, const wchar_t* utf16Str)
48{
49 int requiredSize = WideCharToMultiByte(CP_UTF8, 0, utf16Str, -1, NULL, 0, NULL, NULL);
50
51 if (requiredSize > 0 && utf8Str.SetLen(requiredSize))
52 {
53 WideCharToMultiByte(CP_UTF8, 0, utf16Str, -1, utf8Str.Get(), requiredSize, NULL, NULL);
54 return;
55 }
56
57 utf8Str.Set("");
58}

69static void GetModulePath(HMODULE hModule, WDL_String& path)
70{
71 path.Set("");
72 char pathCStr[MAX_WIN32_PATH_LEN];
73 pathCStr[0] = '\0';
74 if (GetModuleFileName(hModule, pathCStr, MAX_WIN32_PATH_LEN))
75 {
76 int s = -1;
77 for (int i = 0; i < strlen(pathCStr); ++i)
78 {
79 if (pathCStr[i] == '\\')
80 {
81 s = i;
82 }
83 }
84 if (s >= 0 && s + 1 < strlen(pathCStr))
85 {
86 path.Set(pathCStr, s + 1);
87 }
88 }
89}

146static BOOL EnumResNameProc(HANDLE module, LPCTSTR type, LPTSTR name, LONG_PTR param)
147{
148 if (IS_INTRESOURCE(name))
149 return true; // integer resources not wanted
150 else
151 {
152 WDL_String* search = (WDL_String*)param;
153 if (search != 0 && name != 0)
154 {
155 // strip off extra quotes
156 WDL_String strippedName(strlwr(name + 1));
157 strippedName.SetLen(strippedName.GetLength() - 1);
158
159 if (strcmp(strlwr(search->Get()), strippedName.Get()) == 0) // if we are looking for a resource with this name
160 {
161 search->SetFormatted(strippedName.GetLength() + 7, "found: %s", strippedName.Get());
162 return false;
163 }
164 }
165 }
166
167 return true; // keep enumerating
168}

170EResourceLocation LocateResource(const char* name, const char* type, WDL_String& result, const char*, void* pHInstance,
171 const char*)
172{
173 if (CStringHasContents(name))
174 {
175 WDL_String search(name);
176 WDL_String typeUpper(type);
177
178 HMODULE hInstance = static_cast<HMODULE>(pHInstance);
179
180 EnumResourceNames(hInstance, _strupr(typeUpper.Get()), (ENUMRESNAMEPROC)EnumResNameProc, (LONG_PTR)&search);
181
182 if (strstr(search.Get(), "found: ") != 0)
183 {
184 result.SetFormatted(MAX_PATH, "\"%s\"", search.Get() + 7, search.GetLength() - 7); // 7 = strlen("found: ")
185 return EResourceLocation::kWinBinary;
186 }
187 else
188 {
189 if (PathFileExists(name))
190 {
191 result.Set(name);
192 return EResourceLocation::kAbsolutePath;
193 }
194 }
195 }
196 return EResourceLocation::kNotFound;
197}

199const void* LoadWinResource(const char* resid, const char* type, int& sizeInBytes, void* pHInstance)
200{
201 WDL_String typeUpper(type);
202
203 HMODULE hInstance = static_cast<HMODULE>(pHInstance);
204
205 HRSRC hResource = FindResource(hInstance, resid, _strupr(typeUpper.Get()));
206
207 if (!hResource)
208 return NULL;
209
210 DWORD size = SizeofResource(hInstance, hResource);
211
212 if (size < 8)
213 return NULL;
214
215 HGLOBAL res = LoadResource(hInstance, hResource);
216
217 const void* pResourceData = LockResource(res);
218
219 if (!pResourceData)
220 {
221 sizeInBytes = 0;
222 return NULL;
223 }
224 else
225 {
226 sizeInBytes = size;
227 return pResourceData;
228 }
229}

249EResourceLocation LocateResource(const char* name, const char* type, WDL_String& result, const char*, void*,
250 const char*)
251{
252 if (CStringHasContents(name))
253 {
254 WDL_String plusSlash;
255 WDL_String path(name);
256 const char* file = path.get_filepart();
257
258 bool foundResource = false;
259
260 // TODO: FindResource is not sufficient here
261
262 if (strcmp(type, "png") == 0)
263 { // TODO: lowercase/uppercase png
264 plusSlash.SetFormatted(strlen("/resources/img/") + strlen(file) + 1, "/resources/img/%s", file);
265 foundResource =
266 emscripten::val::global("preloadedImages").call<bool>("hasOwnProperty", std::string(plusSlash.Get()));
267 }
268 else if (strcmp(type, "ttf") == 0)
269 { // TODO: lowercase/uppercase ttf
270 plusSlash.SetFormatted(strlen("/resources/fonts/") + strlen(file) + 1, "/resources/fonts/%s", file);
271 foundResource = true; // TODO: check ttf
272 }
273 else if (strcmp(type, "svg") == 0)
274 { // TODO: lowercase/uppercase svg
275 plusSlash.SetFormatted(strlen("/resources/img/") + strlen(file) + 1, "/resources/img/%s", file);
276 foundResource = true; // TODO: check svg
277 }
278
279 if (foundResource)
280 {
281 result.Set(plusSlash.Get());
282 return EResourceLocation::kAbsolutePath;
283 }
284 }
285 return EResourceLocation::kNotFound;
286}
