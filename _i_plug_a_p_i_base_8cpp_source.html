68bool IPlugAPIBase::CompareState(const uint8_t* pIncomingState, int startPos) const
69{
70 bool isEqual = true;
71
72 const double* data = (const double*)pIncomingState + startPos;
73
74 // dirty hack here because protools treats param values as 32 bit int and in IPlug they are 64bit float
75 // if we memcmp() the incoming state with the current they may have tiny differences due to the quantization
76 for (int i = 0; i < NParams(); i++)
77 {
78 float v = (float)GetParam(i)->Value();
79 float vi = (float)*(data++);
80
81 isEqual &= (std::fabs(v - vi) < 0.00001);
82 }
83
84 return isEqual;
85}

137void IPlugAPIBase::OnTimer(Timer& t)
138{
139 if (HasUI())
140 {
141// VST3 ********************************************************************************
142#if defined VST3P_API || defined VST3_API
143 while (mMidiMsgsFromProcessor.ElementsAvailable())
144 {
145 IMidiMsg msg;
146 mMidiMsgsFromProcessor.Pop(msg);
147 #ifdef VST3P_API // distributed
148 TransmitMidiMsgFromProcessor(msg);
149 #else
150 SendMidiMsgFromDelegate(msg);
151 #endif
152 }
153
154 while (mSysExDataFromProcessor.ElementsAvailable())
155 {
156 SysExData msg;
157 mSysExDataFromProcessor.Pop(msg);
158 #ifdef VST3P_API // distributed
159 TransmitSysExDataFromProcessor(msg);
160 #else
161 SendSysexMsgFromDelegate({msg.mOffset, msg.mData, msg.mSize});
162 #endif
163 }
164// !VST3 ******************************************************************************
165#else
166 while (mParamChangeFromProcessor.ElementsAvailable())
167 {
168 ParamTuple p;
169 mParamChangeFromProcessor.Pop(p);
170 SendParameterValueFromDelegate(p.idx, p.value, false);
171 }
172
173 while (mMidiMsgsFromProcessor.ElementsAvailable())
174 {
175 IMidiMsg msg;
176 mMidiMsgsFromProcessor.Pop(msg);
177 SendMidiMsgFromDelegate(msg);
178 }
179
180 while (mSysExDataFromProcessor.ElementsAvailable())
181 {
182 SysExData msg;
183 mSysExDataFromProcessor.Pop(msg);
184 SendSysexMsgFromDelegate({msg.mOffset, msg.mData, msg.mSize});
185 }
186#endif
187 }
188
189 OnIdle();
190}
