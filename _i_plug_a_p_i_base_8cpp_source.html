69 bool IPlugAPIBase::CompareState(const uint8_t* pIncomingState, int startPos) const
70 {
71  bool isEqual = true;
72 
73  const double* data = (const double*) pIncomingState + startPos;
74 
75  // dirty hack here because protools treats param values as 32 bit int and in IPlug they are 64bit float
76  // if we memcmp() the incoming state with the current they may have tiny differences due to the quantization
77  for (int i = 0; i < NParams(); i++)
78  {
79  float v = (float) GetParam(i)->Value();
80  float vi = (float) *(data++);
81 
82  isEqual &= (fabsf(v - vi) < 0.00001);
83  }
84 
85  return isEqual;
86 }

138 void IPlugAPIBase::OnTimer(Timer& t)
139 {
140  if(HasUI())
141  {
142  // in distributed VST 3, parameter changes are managed by the host
143  #if !defined VST3C_API && !defined VST3P_API
144  while(mParamChangeFromProcessor.ElementsAvailable())
145  {
146  IParamChange p;
147  mParamChangeFromProcessor.Pop(p);
148  SendParameterValueFromDelegate(p.paramIdx, p.value, p.normalized); // TODO: if the parameter hasn't changed maybe we shouldn't do anything?
149  }
150 
151  while (mMidiMsgsFromProcessor.ElementsAvailable())
152  {
153  IMidiMsg msg;
154  mMidiMsgsFromProcessor.Pop(msg);
155  SendMidiMsgFromDelegate(msg);
156  }
157 
158  while (mSysExDataFromProcessor.ElementsAvailable())
159  {
160  SysExData msg;
161  mSysExDataFromProcessor.Pop(msg);
162  SendSysexMsgFromDelegate({msg.mOffset, msg.mData, msg.mSize});
163  }
164  #endif
165 
166  // Midi messages from the processor to the controller, are sent as IMessages and SendMidiMsgFromDelegate gets triggered on the other side's notify
167  #if defined VST3P_API
168  while (mMidiMsgsFromProcessor.ElementsAvailable())
169  {
170  IMidiMsg msg;
171  mMidiMsgsFromProcessor.Pop(msg);
172  TransmitMidiMsgFromProcessor(msg);
173  }
174 
175  while (mSysExDataFromProcessor.ElementsAvailable())
176  {
177  SysExData data;
178  mSysExDataFromProcessor.Pop(data);
179  TransmitSysExDataFromProcessor(data);
180  }
181  #endif
182  }
183 
184  OnIdle();
185 }
