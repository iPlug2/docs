96 bool IPlugAPPHost::InitState()
97 {

104 #else
105  #error NOT IMPLEMENTED

107 
108  struct stat st;
109 
110  if(stat(mINIPath.Get(), &st) == 0) // if directory exists
111  {
112  mINIPath.Append("settings.ini"); // add file name to path
113 
114  char buf[STRBUFSZ];
115

142 
143  // if settings file doesn't exist, populate with default values, otherwise overrwrite
144  UpdateINI();
145  }
146  else // folder doesn't exist - make folder and make file

148 #if defined OS_WIN

161  UpdateINI(); // will write file if doesn't exist

163  else
164  {
165  return false;
166  }

170  }

267 void IPlugAPPHost::ProbeAudioIO()
268 {
269  std::cout << "\nRtAudio Version " << RtAudio::getVersion() << std::endl;
270 
271  RtAudio::DeviceInfo info;
272 
273  mAudioInputDevs.clear();
274  mAudioOutputDevs.clear();
275  mAudioIDDevNames.clear();
276 
277  uint32_t nDevices = mDAC->getDeviceCount();
278 
279  for (int i=0; i<nDevices; i++)
280  {
281  info = mDAC->getDeviceInfo(i);
282  std::string deviceName = info.name;
283 
284 #ifdef OS_MAC
285  size_t colonIdx = deviceName.rfind(": ");
286 
287  if(colonIdx != std::string::npos && deviceName.length() >= 2)
288  deviceName = deviceName.substr(colonIdx + 2, deviceName.length() - colonIdx - 2);
289 
290 #endif
291 
292  mAudioIDDevNames.push_back(deviceName);
293 
294  if ( info.probed == false )
295  std::cout << deviceName << ": Probe Status = Unsuccessful\n";
296  else if ( !strcmp("Generic Low Latency ASIO Driver", deviceName.c_str() ))
297  std::cout << deviceName << ": Probe Status = Unsuccessful\n";
298  else
299  {
300  if(info.inputChannels > 0)
301  mAudioInputDevs.push_back(i);
302 
303  if(info.outputChannels > 0)
304  mAudioOutputDevs.push_back(i);
305 
306  if (info.isDefaultInput)
307  mDefaultInputDev = i;
308 
309  if (info.isDefaultOutput)
310  mDefaultOutputDev = i;
311  }
312  }

565 bool IPlugAPPHost::InitAudio(uint32_t inId, uint32_t outId, uint32_t sr, uint32_t iovs)
566 {
567  if (mDAC->isStreamOpen())
568  {
569  if (mDAC->isStreamRunning())
570  {
571  try
572  {
573  mDAC->abortStream();
574  }
575  catch (RtAudioError& e)
576  {
577  e.printMessage();
578  }
579  }
580 
581  mDAC->closeStream();
582  }
583 
584  RtAudio::StreamParameters iParams, oParams;
585  iParams.deviceId = inId;
586  iParams.nChannels = 2; // TODO: flexible channel count
587  iParams.firstChannel = 0; // TODO: flexible channel count
588 
589  oParams.deviceId = outId;
590  oParams.nChannels = 2; // TODO: flexible channel count
591  oParams.firstChannel = 0; // TODO: flexible channel count
592 
593  mBufferSize = iovs; // mBufferSize may get changed by stream
594 
595  DBGMSG("\ntrying to start audio stream @ %i sr, %i buffersize\nindev = %i:%s\noutdev = %i:%s\n", sr, mBufferSize, inId, GetAudioDeviceName(inId).c_str(), outId, GetAudioDeviceName(outId).c_str());
596 
597  RtAudio::StreamOptions options;
598  options.flags = RTAUDIO_NONINTERLEAVED;
599  // options.streamName = BUNDLE_NAME; // JACK stream name, not used on other streams
600 
601  mBufIndex = 0;
602  mSamplesElapsed = 0;
603  mFadeMult = 0.;
604  mSampleRate = (double) sr;
605 
606  mIPlug->SetBlockSize(APP_SIGNAL_VECTOR_SIZE);
607  mIPlug->SetSampleRate(mSampleRate);
608  mIPlug->OnReset();
609 
610  try
611  {
612  mDAC->openStream(&oParams, &iParams, RTAUDIO_FLOAT64, sr, &mBufferSize, &AudioCallback, NULL, &options /*, &ErrorCallback */);
613  mDAC->startStream();
614 
615  mActiveState = mState;
616  }
617  catch (RtAudioError& e)
618  {
619  e.printMessage();
620  return false;
621  }
622 
623  return true;
624 }

656 // static
657 int IPlugAPPHost::AudioCallback(void* pOutputBuffer, void* pInputBuffer, uint32_t nFrames, double streamTime, RtAudioStreamStatus status, void* pUserData)
658 {
659  if ( status )
660  std::cout << "Stream underflow detected!" << std::endl;
661 
662  IPlugAPPHost* _this = sInstance;
663 
664  double* pInputBufferD = static_cast<double*>(pInputBuffer);
665  double* pOutputBufferD = static_cast<double*>(pOutputBuffer);
666 
667  int inRightOffset = 0;
668 
669 // if(!mState.mAudioInIsMono)
670  inRightOffset = nFrames;
671 
672  if (_this->mVecElapsed > APP_N_VECTOR_WAIT ) // wait APP_N_VECTOR_WAIT * iovs before processing audio, to avoid clicks
673  {
674  for (int i=0; i<nFrames; i++)
675  {
676  _this->mBufIndex %= APP_SIGNAL_VECTOR_SIZE;
677 
678  if (_this->mBufIndex == 0)
679  {
680  double* inputs[2] = {pInputBufferD + i, pInputBufferD + inRightOffset + i};
681  double* outputs[2] = {pOutputBufferD + i, pOutputBufferD + nFrames + i};
682 
683  _this->mIPlug->AppProcess(inputs, outputs, APP_SIGNAL_VECTOR_SIZE);
684 
685  _this->mSamplesElapsed += APP_SIGNAL_VECTOR_SIZE;
686  }
687 
688  // fade in
689  if (_this->mFadeMult < 1.)
690  {
691  _this->mFadeMult += (1. / nFrames);
692  }
693 
694  pOutputBufferD[i] *= _this->mFadeMult;
695  pOutputBufferD[i + nFrames] *= _this->mFadeMult;
696 
697  pOutputBufferD[i] *= APP_MULT;
698  pOutputBufferD[i + nFrames] *= APP_MULT;
699 
700  _this->mBufIndex++;
701  }
702  }
703  else
704  {
705  memset(pOutputBufferD, 0, nFrames * APP_NUM_CHANNELS * sizeof(double));
706  }
707 
708  _this->mVecElapsed++;
709 
710  return 0;
711 }
