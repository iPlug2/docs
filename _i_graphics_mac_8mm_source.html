26 int GetSystemVersion()
27 {
28  static int32_t v;
29  if (!v)
30  {
31  if (NSAppKitVersionNumber >= 1266.0)
32  {
33  if (NSAppKitVersionNumber >= 1404.0)
34  v = 0x10b0;
35  else
36  v = 0x10a0; // 10.10+ Gestalt(gsv) return 0x109x, so we bump this to 0x10a0
37  }
38  else
39  {
40  SInt32 a = 0x1040;
41  Gestalt(gestaltSystemVersion,&a);
42  v=a;
43  }
44  }
45  return v;
46 }

100 bool IGraphicsMac::GetResourcePathFromBundle(const char* fileName, const char* searchExt, WDL_String& fullPath)
101 {
102  CocoaAutoReleasePool pool;
103 
104  const char* ext = fileName+strlen(fileName)-1;
105  while (ext >= fileName && *ext != '.') --ext;
106  ++ext;
107 
108  bool isCorrectType = !strcasecmp(ext, searchExt);
109 
110  NSBundle* pBundle = [NSBundle bundleWithIdentifier:ToNSString(GetBundleID())];
111  NSString* pFile = [[[NSString stringWithCString:fileName encoding:NSUTF8StringEncoding] lastPathComponent] stringByDeletingPathExtension];
112 
113  if (isCorrectType && pBundle && pFile)
114  {
115  NSString* pPath = [pBundle pathForResource:pFile ofType:ToNSString(searchExt)];
116 
117  if (pPath)
118  {
119  if([[NSFileManager defaultManager] fileExistsAtPath : pPath] == YES)
120  {
121  fullPath.Set([pPath cString]);
122  return true;
123  }
124  }
125  }
126 
127  fullPath.Set("");
128  return false;
129 }

169 EResourceLocation IGraphicsMac::OSFindResource(const char* name, const char* type, WDL_String& result)
170 {
171  if(CStringHasContents(name))
172  {
173  // first check this bundle
174  if(GetResourcePathFromBundle(name, type, result))
175  return EResourceLocation::kAbsolutePath;
176 
177  // then check ~/Music/PLUG_NAME, which is a shared folder that can be accessed from app sandbox
178  if(GetResourcePathFromUsersMusicFolder(name, type, result))
179  return EResourceLocation::kAbsolutePath;
180 
181  // finally check name, which might be a full path - if the plug-in is trying to load a resource at runtime (e.g. skin-able UI)
182  NSString* pPath = [NSString stringWithCString:name encoding:NSUTF8StringEncoding];
183 
184  if([[NSFileManager defaultManager] fileExistsAtPath : pPath] == YES)
185  {
186  result.Set([pPath UTF8String]);
187  return EResourceLocation::kAbsolutePath;
188  }
189  }
190  return EResourceLocation::kNotFound;
191 }

358 int IGraphicsMac::ShowMessageBox(const char* str, const char* caption, EMessageBoxType type)
359 {
360  NSInteger ret = 0;
361 
362  if (!str) str= "";
363  if (!caption) caption= "";
364 
365  NSString *msg = (NSString *) CFStringCreateWithCString(NULL,str,kCFStringEncodingUTF8);
366  NSString *cap = (NSString *) CFStringCreateWithCString(NULL,caption,kCFStringEncodingUTF8);
367 
368  msg = msg ? msg : (NSString *) CFStringCreateWithCString(NULL, str, kCFStringEncodingASCII);
369  cap = cap ? cap : (NSString *) CFStringCreateWithCString(NULL, caption, kCFStringEncodingASCII);
370 
371  switch (type)
372  {
373  case kMB_OK:
374  NSRunAlertPanel(msg, @"%@", @"OK", @"", @"", cap);
375  ret = kOK;
376  break;
377  case kMB_OKCANCEL:
378  ret = NSRunAlertPanel(msg, @"%@", @"OK", @"Cancel", @"", cap);
379  ret = ret ? kOK : kCANCEL;
380  break;
381  case kMB_YESNO:
382  ret = NSRunAlertPanel(msg, @"%@", @"Yes", @"No", @"", cap);
383  ret = ret ? kYES : kNO;
384  break;
385  case kMB_RETRYCANCEL:
386  ret = NSRunAlertPanel(msg, @"%@", @"Retry", @"Cancel", @"", cap);
387  ret = ret ? kRETRY : kCANCEL;
388  break;
389  case kMB_YESNOCANCEL:
390  ret = NSRunAlertPanel(msg, @"%@", @"Yes", @"Cancel", @"No", cap);
391  ret = (ret == 1) ? kYES : (ret == -1) ? kNO : kCANCEL;
392  break;
393  }
394 
395  [msg release];
396  [cap release];
397 
398  return static_cast<int>(ret);
399 }
