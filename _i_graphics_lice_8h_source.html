41 inline int LiceBlendMode(const IBlend* pBlend)
42 {
43  if (!pBlend)
44  {
45  return LICE_BLIT_MODE_COPY | LICE_BLIT_USE_ALPHA;
46  }
47  switch (pBlend->mMethod)
48  {
49  case EBlendType::kBlendClobber: return LICE_BLIT_MODE_COPY;
50  case EBlendType::kBlendAdd: return LICE_BLIT_MODE_ADD | LICE_BLIT_USE_ALPHA;
51  case EBlendType::kBlendColorDodge: return LICE_BLIT_MODE_DODGE | LICE_BLIT_USE_ALPHA;
52  case EBlendType::kBlendNone:
53  default:
54  {
55  return LICE_BLIT_MODE_COPY | LICE_BLIT_USE_ALPHA;
56  }
57  }
58 }
