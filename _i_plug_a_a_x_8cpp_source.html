120 AAX_Result IPlugAAX::EffectInit()
121 {
122  TRACE;
123 
124  if (GetHost() == kHostUninit)
125  SetHost("ProTools", 0); // TODO:vendor version correct?
126 
127  AAX_CString bypassID = NULL;
128  this->GetMasterBypassParameter( &bypassID );
129  mBypassParameter = new AAX_CParameter<bool>(bypassID.CString(),
130  AAX_CString("Master Bypass"),
131  false,
132  AAX_CBinaryTaperDelegate<bool>(),
133  AAX_CBinaryDisplayDelegate<bool>("bypass", "on"),
134  true);
135  mBypassParameter->SetNumberOfSteps( 2 );
136  mBypassParameter->SetType( AAX_eParameterType_Discrete );
137  mParameterManager.AddParameter(mBypassParameter);
138 
139  for (int i=0; i<NParams(); i++)
140  {
141  IParam* pParam = GetParam(i);
142  AAX_IParameter* pAAXParam = nullptr;
143 
144  WDL_String* pParamIDStr = new WDL_String("_", 1);
145  pParamIDStr->SetFormatted(MAX_AAX_PARAMID_LEN, "%i", i+kAAXParamIdxOffset);
146  mParamIDs.Add(pParamIDStr);
147 
148  switch (pParam->Type())
149  {
150  case IParam::kTypeDouble:
151  {
152  pAAXParam = new AAX_CParameter<double>(pParamIDStr->Get(),
153  AAX_CString(pParam->GetNameForHost()),
154  pParam->GetDefault(),
155  AAX_CIPlugTaperDelegate<double>(*pParam),
156  AAX_CUnitDisplayDelegateDecorator<double>( AAX_CNumberDisplayDelegate<double>(), AAX_CString(pParam->GetLabelForHost())),
157  pParam->GetCanAutomate());
158 
159  pAAXParam->SetNumberOfSteps(128); // TODO: check this https://developer.digidesign.com/index.php?L1=5&L2=13&L3=56
160  pAAXParam->SetType(AAX_eParameterType_Continuous);
161 
162  break;
163  }
164  case IParam::kTypeInt:
165  {
166  pAAXParam = new AAX_CParameter<int>(pParamIDStr->Get(),
167  AAX_CString(pParam->GetNameForHost()),
168  (int)pParam->GetDefault(),
169  AAX_CLinearTaperDelegate<int,1>((int)pParam->GetMin(), (int)pParam->GetMax()),
170  AAX_CUnitDisplayDelegateDecorator<int>( AAX_CNumberDisplayDelegate<int,0>(), AAX_CString(pParam->GetLabelForHost())),
171  pParam->GetCanAutomate());
172 
173  pAAXParam->SetNumberOfSteps(128);
174  pAAXParam->SetType(AAX_eParameterType_Continuous);
175 
176  break;
177  }
178  case IParam::kTypeEnum:
179  case IParam::kTypeBool:
180  {
181  int nTexts = pParam->NDisplayTexts();
182 
183  std::map<int, AAX_CString> displayTexts;
184 
185  for (int j=0; j<pParam->NDisplayTexts(); j++)
186  {
187  double value;
188  const char* text = pParam->GetDisplayTextAtIdx(j, &value);
189 
190  displayTexts.insert(std::pair<int, AAX_CString>(value, AAX_CString(text)) );
191  }
192 
193  pAAXParam = new AAX_CParameter<int>(pParamIDStr->Get(),
194  AAX_CString(pParam->GetNameForHost()),
195  (int)pParam->GetDefault(),
196  AAX_CLinearTaperDelegate<int,1>((int) pParam->GetMin(), (int) pParam->GetMax()),
197  AAX_CStringDisplayDelegate<int>(displayTexts),
198  pParam->GetCanAutomate());
199 
200  pAAXParam->SetNumberOfSteps(nTexts);
201  pAAXParam->SetType(AAX_eParameterType_Discrete);
202 
203  break;
204  }
205  default:
206  break;
207  }
208 
209  mParameterManager.AddParameter(pAAXParam);
210  }
211 
212  AAX_CSampleRate sr;
213  Controller()->GetSampleRate(&sr);
214  SetSampleRate(sr);
215  OnReset();
216 
217  return AAX_SUCCESS;
218 }

220 AAX_Result IPlugAAX::UpdateParameterNormalizedValue(AAX_CParamID paramID, double iValue, AAX_EUpdateSource iSource )
221 {
222  TRACE;
223 
224  AAX_Result result = AAX_SUCCESS;
225 
226  AAX_IParameter* pAAXParameter = mParameterManager.GetParameterByID(paramID);
227 
228  if (pAAXParameter == nullptr)
229  return AAX_ERROR_INVALID_PARAMETER_ID;
230 
231  // Store the value into the AAX parameter
232  pAAXParameter->UpdateNormalizedValue(iValue);
233 
234  int paramIdx = atoi(paramID) - kAAXParamIdxOffset;
235 
236  if ((paramIdx >= 0) && (paramIdx < NParams()))
237  {
238  ENTER_PARAMS_MUTEX;
239  GetParam(paramIdx)->SetNormalized(iValue);
240  SendParameterValueFromAPI(paramIdx, iValue, true);
241  OnParamChange(paramIdx, kHost);
242  LEAVE_PARAMS_MUTEX;
243  }
244 
245  // Now the control has changed
246  result = mPacketDispatcher.SetDirty(paramID);
247 
248  mNumPlugInChanges++;
249 
250  return result;

253 void IPlugAAX::RenderAudio(AAX_SIPlugRenderInfo* pRenderInfo)
254 {
255  TRACE;
256 
257  // Get bypass parameter value
258  bool bypass;
259  mBypassParameter->GetValueAsBool(&bypass);
260 
261  AAX_EStemFormat inFormat, outFormat;
262  Controller()->GetInputStemFormat(&inFormat);
263  Controller()->GetOutputStemFormat(&outFormat);
264 
265  if (DoesMIDIIn())
266  {
267  AAX_IMIDINode* pMidiIn = pRenderInfo->mInputNode;
268  AAX_CMidiStream* pMidiBuffer = pMidiIn->GetNodeBuffer();
269  AAX_CMidiPacket* pMidiPacket = pMidiBuffer->mBuffer;
270  uint32_t packets_count = pMidiBuffer->mBufferSize;
271 
272  for (int i = 0; i<packets_count; i++, pMidiPacket++)
273  {
274  IMidiMsg msg(pMidiPacket->mTimestamp, pMidiPacket->mData[0], pMidiPacket->mData[1], pMidiPacket->mData[2]);
275  ProcessMidiMsg(msg);
276  mMidiMsgsFromProcessor.Push(msg);
277  }
278  }
279 
280  AAX_IMIDINode* pTransportNode = pRenderInfo->mTransportNode;
281  mTransport = pTransportNode->GetTransport();
282 
283  int32_t numSamples = *(pRenderInfo->mNumSamples);
284  int32_t numInChannels = AAX_STEM_FORMAT_CHANNEL_COUNT(inFormat);
285  int32_t numOutChannels = AAX_STEM_FORMAT_CHANNEL_COUNT(outFormat);
286 
287  SetChannelConnections(ERoute::kInput, 0, numInChannels, true);
288  SetChannelConnections(ERoute::kInput, numInChannels, MaxNChannels(ERoute::kInput) - numInChannels, false);
289  AttachBuffers(ERoute::kInput, 0, MaxNChannels(ERoute::kInput), pRenderInfo->mAudioInputs, numSamples);
290 
291  SetChannelConnections(ERoute::kOutput, 0, numOutChannels, true);
292  SetChannelConnections(ERoute::kOutput, numOutChannels, MaxNChannels(ERoute::kOutput) - numOutChannels, false);
293  AttachBuffers(ERoute::kOutput, 0, MaxNChannels(ERoute::kOutput), pRenderInfo->mAudioOutputs, numSamples);
294 
295  if (bypass)
296  PassThroughBuffers(0.0f, numSamples);
297  else
298  {
299  int32_t num, denom;
300  int64_t ppqPos, samplePos, cStart, cEnd;
301  ITimeInfo timeInfo;
302 
303  mTransport->GetCurrentTempo(&timeInfo.mTempo);
304  mTransport->IsTransportPlaying(&timeInfo.mTransportIsRunning);
305 
306  mTransport->GetCurrentMeter(&num, &denom);
307  timeInfo.mNumerator = (int) num;
308  timeInfo.mDenominator = (int) denom;
309 
310  mTransport->GetCurrentTickPosition(&ppqPos);
311  timeInfo.mPPQPos = (double) ppqPos / 960000.0;
312 
313  mTransport->GetCurrentNativeSampleLocation(&samplePos);
314  timeInfo.mSamplePos = (double) samplePos;
315 
316  mTransport->GetCurrentLoopPosition(&timeInfo.mTransportLoopEnabled, &cStart, &cEnd);
317  timeInfo.mCycleStart = (double) cStart / 960000.0;
318  timeInfo.mCycleEnd = (double) cEnd / 960000.0;
319 
320  SetTimeInfo(timeInfo);
321  //timeInfo.mLastBar ??
322 
323  IMidiMsg msg;
324 
325  while (mMidiMsgsFromEditor.Pop(msg))
326  {
327  ProcessMidiMsg(msg);
328  }
329 
330  ProcessBuffers(0.0f, numSamples);
331  }
332 
333  // Midi Out
334  if (DoesMIDIOut())
335  {
336  AAX_IMIDINode* midiOut = pRenderInfo->mOutputNode;
337 
338  if(midiOut)
339  {
340  //MIDI
341  if (!mMidiOutputQueue.Empty())
342  {
343  while (!mMidiOutputQueue.Empty())
344  {
345  IMidiMsg& msg = mMidiOutputQueue.Peek();
346 
347  AAX_CMidiPacket packet;
348 
349  packet.mIsImmediate = true; // TODO: how does this affect sample accuracy?
350 
351  packet.mTimestamp = (uint32_t) msg.mOffset;
352  packet.mLength = 3;
353 
354  packet.mData[0] = msg.mStatus;
355  packet.mData[1] = msg.mData1;
356  packet.mData[2] = msg.mData2;
357 
358  midiOut->PostMIDIPacket (&packet);
359 
360  mMidiOutputQueue.Remove();
361  }
362  }
363 
364  mMidiOutputQueue.Flush(numSamples);
365 
366  //Output SYSEX from the editor, which has bypassed ProcessSysEx()
367  if(mSysExDataFromEditor.ElementsAvailable())
368  {
369  while (mSysExDataFromEditor.Pop(mSysexBuf))
370  {
371  int numPackets = (int) ceil((float) mSysexBuf.mSize/4.); // each packet can store 4 bytes of data
372  int bytesPos = 0;
373 
374  for (int p = 0; p < numPackets; p++)
375  {
376  AAX_CMidiPacket packet;
377 
378  packet.mTimestamp = (uint32_t) mSysexBuf.mOffset;
379  packet.mIsImmediate = true;
380 
381  int b = 0;
382 
383  while (b < 4 && bytesPos < mSysexBuf.mSize)
384  {
385  packet.mData[b++] = mSysexBuf.mData[bytesPos++];
386  }
387 
388  packet.mLength = (uint32_t) b;
389 
390  midiOut->PostMIDIPacket (&packet);
391  }
392  }
393  }
394  }
395  }
396 }
