145 static int AsciiToVK(int ascii)
146 {
147#ifdef OS_WIN
148 HKL layout = GetKeyboardLayout(0);
149 return VkKeyScanExA((CHAR)ascii, layout);
150#else
151 // Numbers and uppercase alpha chars map directly to VK
152 if ((ascii >= 0x30 && ascii <= 0x39) || (ascii >= 0x41 && ascii <= 0x5A))
153 {
154 return ascii;
155 }
156
157 // Lowercase alpha chars map to VK but need shifting
158 if (ascii >= 0x61 && ascii <= 0x7A)
159 {
160 return ascii - 0x20;
161 }
162
163 return iplug::kVK_NONE;
164#endif
165 }

167 static int VSTKeyCodeToVK(Steinberg::int16 code, char ascii)
168 {
169 // If the keycode provided by the host is 0, we can still calculate the VK from the ascii value
170 // NOTE: VKEY_EQUALS Doesn't seem to map to a Windows VK, so get the VK from the ascii char instead
171 if (code == 0 || code == Steinberg::KEY_EQUALS)
172 {
173 return AsciiToVK(ascii);
174 }
175
176 using namespace Steinberg;
177 using namespace iplug;
178
179 switch (code)
180 {
181 case KEY_BACK: return kVK_BACK;
182 case KEY_TAB: return kVK_TAB;
183 case KEY_CLEAR: return kVK_CLEAR;
184 case KEY_RETURN: return kVK_RETURN;
185 case KEY_PAUSE: return kVK_PAUSE;
186 case KEY_ESCAPE: return kVK_ESCAPE;
187 case KEY_SPACE: return kVK_SPACE;
188 case KEY_NEXT: return kVK_NEXT;
189 case KEY_END: return kVK_END;
190 case KEY_HOME: return kVK_HOME;
191 case KEY_LEFT: return kVK_LEFT;
192 case KEY_UP: return kVK_UP;
193 case KEY_RIGHT: return kVK_RIGHT;
194 case KEY_DOWN: return kVK_DOWN;
195 case KEY_PAGEUP: return kVK_PRIOR;
196 case KEY_PAGEDOWN: return kVK_NEXT;
197 case KEY_SELECT: return kVK_SELECT;
198 case KEY_PRINT: return kVK_PRINT;
199 case KEY_ENTER: return kVK_RETURN;
200 case KEY_SNAPSHOT: return kVK_SNAPSHOT;
201 case KEY_INSERT: return kVK_INSERT;
202 case KEY_DELETE: return kVK_DELETE;
203 case KEY_HELP: return kVK_HELP;
204 case KEY_NUMPAD0: return kVK_NUMPAD0;
205 case KEY_NUMPAD1: return kVK_NUMPAD1;
206 case KEY_NUMPAD2: return kVK_NUMPAD2;
207 case KEY_NUMPAD3: return kVK_NUMPAD3;
208 case KEY_NUMPAD4: return kVK_NUMPAD4;
209 case KEY_NUMPAD5: return kVK_NUMPAD5;
210 case KEY_NUMPAD6: return kVK_NUMPAD6;
211 case KEY_NUMPAD7: return kVK_NUMPAD7;
212 case KEY_NUMPAD8: return kVK_NUMPAD8;
213 case KEY_NUMPAD9: return kVK_NUMPAD9;
214 case KEY_MULTIPLY: return kVK_MULTIPLY;
215 case KEY_ADD: return kVK_ADD;
216 case KEY_SEPARATOR: return kVK_SEPARATOR;
217 case KEY_SUBTRACT: return kVK_SUBTRACT;
218 case KEY_DECIMAL: return kVK_DECIMAL;
219 case KEY_DIVIDE: return kVK_DIVIDE;
220 case KEY_F1: return kVK_F1;
221 case KEY_F2: return kVK_F2;
222 case KEY_F3: return kVK_F3;
223 case KEY_F4: return kVK_F4;
224 case KEY_F5: return kVK_F5;
225 case KEY_F6: return kVK_F6;
226 case KEY_F7: return kVK_F7;
227 case KEY_F8: return kVK_F8;
228 case KEY_F9: return kVK_F9;
229 case KEY_F10: return kVK_F10;
230 case KEY_F11: return kVK_F11;
231 case KEY_F12: return kVK_F12;
232 case KEY_NUMLOCK: return kVK_NUMLOCK;
233 case KEY_SCROLL: return kVK_SCROLL;
234 case KEY_SHIFT: return kVK_SHIFT;
235 case KEY_CONTROL: return kVK_CONTROL;
236 case KEY_ALT: return kVK_MENU;
237 case KEY_EQUALS: return kVK_NONE;
238 }
239
240 if (code >= VKEY_FIRST_ASCII)
241 return (code - VKEY_FIRST_ASCII + kVK_0);
242
243 return kVK_NONE;

246 static iplug::IKeyPress translateKeyMessage(Steinberg::char16 key, Steinberg::int16 keyMsg,
247 Steinberg::int16 modifiers)
248 {
249 WDL_String str;
250
251 if (key == 0)
252 {
253 key = Steinberg::VirtualKeyCodeToChar((Steinberg::uint8)keyMsg);
254 }
255
256 if (key)
257 {
258 Steinberg::String keyStr(STR(" "));
259 keyStr.setChar16(0, key);
260 keyStr.toMultiByte(Steinberg::kCP_Utf8);
261 if (keyStr.length() == 1)
262 {
263 str.Set(keyStr.text8());
264 }
265 }
266
267 iplug::IKeyPress keyPress{
268 str.Get(), VSTKeyCodeToVK(keyMsg, str.Get()[0]), static_cast<bool>(modifiers & Steinberg::kShiftKey),
269 static_cast<bool>(modifiers & Steinberg::kCommandKey), static_cast<bool>(modifiers & Steinberg::kAlternateKey)};
270
271 return keyPress;
272 }
