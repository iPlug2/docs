228 void IVKnobControl::OnResize()
229 {
230  IRECT clickableArea;
231 
232  if(mLabel.GetLength())
233  {
234  IRECT textRect;
235  GetUI()->MeasureText(mLabelText, mLabel.Get(), textRect);
236 
237  mLabelBounds = mRECT.GetFromTop(textRect.H());
238  }
239  else
240  mLabelBounds = IRECT();
241 
242  if(mLabelBounds.H())
243  clickableArea = mRECT.GetReducedFromTop(mLabelBounds.H());
244  else
245  clickableArea = mRECT;
246 
247  if (mDisplayParamValue)
248  {
249  IRECT textRect;
250  WDL_String str;
251  GetParam()->GetDisplayForHost(str);
252 
253  GetUI()->MeasureText(mValueText, str.Get(), textRect);
254 
255  const float valueDisplayWidth = clickableArea.W() * mKnobFrac * 0.5f;
256  switch (mValueText.mVAlign)
257  {
258  case IText::kVAlignMiddle:
259  mHandleBounds = clickableArea;
260  mValueBounds = clickableArea.GetMidVPadded(textRect.H()/2.f).GetMidHPadded(valueDisplayWidth);
261  break;
262  case IText::kVAlignBottom:
263  {
264  mValueBounds = clickableArea.GetFromBottom(textRect.H()).GetMidHPadded(valueDisplayWidth);
265  mHandleBounds = clickableArea.GetReducedFromBottom(textRect.H());
266  break;
267  }
268  case IText::kVAlignTop:
269  mValueBounds = clickableArea.GetFromTop(textRect.H()).GetMidHPadded(valueDisplayWidth);
270  mHandleBounds = clickableArea.GetReducedFromTop(textRect.H());
271  break;
272  default:
273  break;
274  }
275 
276  if(mValueBounds.W() < textRect.W())
277  mValueBounds = mValueBounds.GetMidHPadded(mTargetRECT.W()/2.f);
278  }
279 
280  mHandleBounds = GetAdjustedHandleBounds(mTargetRECT).GetScaledAboutCentre(mKnobFrac);
281 
282  SetDirty(false);
283 }

390 IRECT IBSliderControl::GetHandleBounds(double value) const
391 {
392  if (value < 0.0)
393  value = mValue;
394 
395  IRECT r(mRECT.L, mRECT.T, mRECT.L + mBitmap.W(), mRECT.T + mBitmap.H());
396 
397  if (mDirection == kVertical)
398  {
399  float offs = (1.f - (float) value) * mTrack.H();
400  r.T += offs;
401  r.B += offs;
402  }
403  else
404  {
405  float offs = (float) value * mTrack.W();
406  r.L += offs;
407  r.R += offs;
