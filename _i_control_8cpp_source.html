271 void IControl::SnapToMouse(float x, float y, EDirection direction, IRECT& bounds, float scalar /* TODO: scalar! */)
272 {
273  bounds.Constrain(x, y);

275  float val;

277  //val = 1.f - (y - (mRECT.B - (mRECT.H()*lengthMult)) / 2) / ((mLen*lengthMult));

280  val = 1.f - (y-bounds.T) / bounds.H();

569 void IDirBrowseControlBase::ScanDirectory(const char* path, IPopupMenu& menuToAddTo)
570 {
571 #if !defined OS_IOS
572  WDL_DirScan d;
573  IPopupMenu& parentDirMenu = menuToAddTo;
574 
575  if (!d.First(path))
576  {
577  do
578  {
579  const char* f = d.GetCurrentFN();
580  if (f && f[0] != '.')
581  {
582  if (d.GetCurrentIsDirectory())
583  {
584  WDL_String subdir;
585  d.GetCurrentFullFN(&subdir);
586  IPopupMenu* pNewMenu = new IPopupMenu();
587  parentDirMenu.AddItem(d.GetCurrentFN(), pNewMenu, -2);
588  ScanDirectory(subdir.Get(), *pNewMenu);
589  }
590  else
591  {
592  const char* a = strstr(f, mExtension.Get());
593  if (a && a > f && strlen(a) == strlen(mExtension.Get()))
594  {
595  WDL_String menuEntry {f};
596 
597  if(!mShowFileExtensions)
598  menuEntry.Set(f, (int) (a - f));
599 
600  IPopupMenu::Item* pItem = new IPopupMenu::Item(menuEntry.Get(), IPopupMenu::Item::kNoFlags, mFiles.GetSize());
601  parentDirMenu.AddItem(pItem, -2 /* sort alphabetically */);
602  WDL_String* pFullPath = new WDL_String("");
603  d.GetCurrentFullFN(pFullPath);
604  mFiles.Add(pFullPath);
605  }
606  }
607  }
608  } while (!d.Next());
609 
610  menuToAddTo = parentDirMenu;
611  }
612 
613  if(!mShowEmptySubmenus)
614  parentDirMenu.RemoveEmptySubmenus();
615 
616 #endif
617 }
