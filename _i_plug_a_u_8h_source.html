330 static AudioComponentMethod Lookup(SInt16 selector)
331 {
332 using Method = AudioComponentMethod;
333
334 switch (selector)
335 {
336 case kAudioUnitInitializeSelect: return (Method)IPlugAU::AUMethodInitialize;
337 case kAudioUnitUninitializeSelect: return (Method)IPlugAU::AUMethodUninitialize;
338 case kAudioUnitGetPropertyInfoSelect: return (Method)IPlugAU::AUMethodGetPropertyInfo;
339 case kAudioUnitGetPropertySelect: return (Method)IPlugAU::AUMethodGetProperty;
340 case kAudioUnitSetPropertySelect: return (Method)IPlugAU::AUMethodSetProperty;
341
342 case kAudioUnitAddPropertyListenerSelect: return (Method)IPlugAU::AUMethodAddPropertyListener;
343 case kAudioUnitRemovePropertyListenerSelect: return (Method)IPlugAU::AUMethodRemovePropertyListener;
344 case kAudioUnitRemovePropertyListenerWithUserDataSelect:
345 return (Method)IPlugAU::AUMethodRemovePropertyListenerWithUserData;
346
347 case kAudioUnitAddRenderNotifySelect: return (Method)IPlugAU::AUMethodAddRenderNotify;
348 case kAudioUnitRemoveRenderNotifySelect: return (Method)IPlugAU::AUMethodRemoveRenderNotify;
349 case kAudioUnitGetParameterSelect: return (Method)IPlugAU::AUMethodGetParameter;
350 case kAudioUnitSetParameterSelect: return (Method)IPlugAU::AUMethodSetParameter;
351 case kAudioUnitScheduleParametersSelect: return (Method)IPlugAU::AUMethodScheduleParameters;
352 case kAudioUnitRenderSelect: return (Method)IPlugAU::AUMethodRender;
353 case kAudioUnitResetSelect: return (Method)IPlugAU::AUMethodReset;
354
355 case kMusicDeviceMIDIEventSelect: return MIDIIn ? (Method)IPlugAU::AUMethodMIDIEvent : NULL;
356 case kMusicDeviceSysExSelect: return MIDIIn ? (Method)IPlugAU::AUMethodSysEx : NULL;
357
358 default: break;
359 }
360 return NULL;
361 }
362
363 static OSStatus Open(void* pSelf, AudioUnit compInstance)
364 {
365 AudioComponentPlugInInstance* acpi = (AudioComponentPlugInInstance*)pSelf;
366 assert(acpi);
367
368 (*acpi->mConstruct)(&acpi->mInstanceStorage, compInstance);
369 IPlugAU* plug = (IPlugAU*)&acpi->mInstanceStorage;
370
371 plug->mCI = compInstance;
372 plug->PruneUninitializedPresets();
373
374 return noErr;
375 }
376
377 static OSStatus Close(void* pSelf)
378 {
379 AudioComponentPlugInInstance* acpi = (AudioComponentPlugInInstance*)pSelf;
380 assert(acpi);
381 (*acpi->mDestruct)(&acpi->mInstanceStorage);
382 free(pSelf);
383 return noErr;
384 }
385
386 static AudioComponentPlugInInterface* Factory(const AudioComponentDescription* pInDesc)
387 {
388 void* ptr = malloc(offsetof(AudioComponentPlugInInstance, mInstanceStorage) + sizeof(Plug));
389 AudioComponentPlugInInstance* acpi = reinterpret_cast<AudioComponentPlugInInstance*>(ptr);
390 acpi->mPlugInInterface.Open = Open;
391 acpi->mPlugInInterface.Close = Close;
392 acpi->mPlugInInterface.Lookup = Lookup;
393 acpi->mPlugInInterface.reserved = NULL;
394 acpi->mConstruct = Construct;
395 acpi->mDestruct = Destruct;
396 acpi->mPad[0] = NULL;
397 acpi->mPad[1] = NULL;
398 return (AudioComponentPlugInInterface*)acpi;
399 }
