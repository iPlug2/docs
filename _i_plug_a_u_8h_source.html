234 static AudioComponentMethod Lookup(SInt16 selector)
235 {
236 using Method = AudioComponentMethod;
237
238 switch (selector)
239 {
240 case kAudioUnitInitializeSelect: return (Method)IPlugAU::AUMethodInitialize;
241 case kAudioUnitUninitializeSelect: return (Method)IPlugAU::AUMethodUninitialize;
242 case kAudioUnitGetPropertyInfoSelect: return (Method)IPlugAU::AUMethodGetPropertyInfo;
243 case kAudioUnitGetPropertySelect: return (Method)IPlugAU::AUMethodGetProperty;
244 case kAudioUnitSetPropertySelect: return (Method)IPlugAU::AUMethodSetProperty;
245
246 case kAudioUnitAddPropertyListenerSelect: return (Method)IPlugAU::AUMethodAddPropertyListener;
247 case kAudioUnitRemovePropertyListenerSelect: return (Method)IPlugAU::AUMethodRemovePropertyListener;
248 case kAudioUnitRemovePropertyListenerWithUserDataSelect: return (Method)IPlugAU::AUMethodRemovePropertyListenerWithUserData;
249
250 case kAudioUnitAddRenderNotifySelect: return (Method)IPlugAU::AUMethodAddRenderNotify;
251 case kAudioUnitRemoveRenderNotifySelect: return (Method)IPlugAU::AUMethodRemoveRenderNotify;
252 case kAudioUnitGetParameterSelect: return (Method)IPlugAU::AUMethodGetParameter;
253 case kAudioUnitSetParameterSelect: return (Method)IPlugAU::AUMethodSetParameter;
254 case kAudioUnitScheduleParametersSelect: return (Method)IPlugAU::AUMethodScheduleParameters;
255 case kAudioUnitRenderSelect: return (Method)IPlugAU::AUMethodRender;
256 case kAudioUnitResetSelect: return (Method)IPlugAU::AUMethodReset;
257
258 case kMusicDeviceMIDIEventSelect: return MIDIIn ? (Method)IPlugAU::AUMethodMIDIEvent : NULL;
259 case kMusicDeviceSysExSelect: return MIDIIn ? (Method)IPlugAU::AUMethodSysEx : NULL;
260
261 default: break;
262 }
263 return NULL;
264 }
265
266 static OSStatus Open(void* pSelf, AudioUnit compInstance)
267 {
268 AudioComponentPlugInInstance* acpi = (AudioComponentPlugInInstance*)pSelf;
269 assert(acpi);
270
271 (*acpi->mConstruct)(&acpi->mInstanceStorage, compInstance);
272 IPlugAU* plug = (IPlugAU*)&acpi->mInstanceStorage;
273
274 plug->mCI = compInstance;
275 plug->PruneUninitializedPresets();
276
277 return noErr;
278 }
279
280 static OSStatus Close(void* pSelf)
281 {
282 AudioComponentPlugInInstance* acpi = (AudioComponentPlugInInstance*)pSelf;
283 assert(acpi);
284 (*acpi->mDestruct)(&acpi->mInstanceStorage);
285 free(pSelf);
286 return noErr;
287 }
288
289 static AudioComponentPlugInInterface* Factory(const AudioComponentDescription* pInDesc)
290 {
291 void* ptr = malloc(offsetof(AudioComponentPlugInInstance, mInstanceStorage) + sizeof(Plug));
292 AudioComponentPlugInInstance* acpi = reinterpret_cast<AudioComponentPlugInInstance*>(ptr);
293 acpi->mPlugInInterface.Open = Open;
294 acpi->mPlugInInterface.Close = Close;
295 acpi->mPlugInInterface.Lookup = Lookup;
296 acpi->mPlugInInterface.reserved = NULL;
297 acpi->mConstruct = Construct;
298 acpi->mDestruct = Destruct;
299 acpi->mPad[0] = NULL;
300 acpi->mPad[1] = NULL;
301 return (AudioComponentPlugInInterface*)acpi;
302 }
