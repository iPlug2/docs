24 inline void PreMulCompositeSourceOver(LICE_pixel_chan* out, LICE_pixel_chan* in)
25 {
26  unsigned int alphaCmp = 256 - in[LICE_PIXEL_A];
27 
28  unsigned int A = in[LICE_PIXEL_A] + ((out[LICE_PIXEL_A] * alphaCmp) >> 8);
29  unsigned int R = in[LICE_PIXEL_R] + ((out[LICE_PIXEL_R] * alphaCmp) >> 8);
30  unsigned int G = in[LICE_PIXEL_G] + ((out[LICE_PIXEL_G] * alphaCmp) >> 8);
31  unsigned int B = in[LICE_PIXEL_B] + ((out[LICE_PIXEL_B] * alphaCmp) >> 8);
32 
33  _LICE_MakePixelClamp(out, R, G, B, A);
34 }

36 inline void PreMulCompositeAdd(LICE_pixel_chan* out, LICE_pixel_chan* in)
37 {
38  unsigned int alpha = in[LICE_PIXEL_A];
39 
40  unsigned int A = out[LICE_PIXEL_A] + (255 * in[LICE_PIXEL_A] / alpha);
41  unsigned int R = out[LICE_PIXEL_R] + (255 * in[LICE_PIXEL_R] / alpha);
42  unsigned int G = out[LICE_PIXEL_G] + (255 * in[LICE_PIXEL_G] / alpha);
43  unsigned int B = out[LICE_PIXEL_B] + (255 * in[LICE_PIXEL_B] / alpha);
44 
45  _LICE_MakePixelClamp(out, R, G, B, A);
46 }

48 void PreMulBlit(LICE_IBitmap *dest, LICE_IBitmap *src, int dstx, int dsty, int srcx, int srcy, int srcw, int srch, float alpha, int mode)
49 {
50  srcx = dstx < 0 ? srcx - dstx : srcx;
51  srcy = dsty < 0 ? srcy - dsty : srcy;
52  dstx = std::max(dstx, 0);
53  dsty = std::max(dsty, 0);
54  srcw = std::min(srcw, dest->getWidth() - dstx);
55  srch = std::min(srch, dest->getHeight() - dsty);
56 
57  int inStride = src->getRowSpan() * 4;
58  int outStride = dest->getRowSpan() * 4;
59 
60  LICE_pixel_chan* in = ((LICE_pixel_chan*) src->getBits()) + (srcy * inStride) + (srcx * 4);
61  LICE_pixel_chan* out = ((LICE_pixel_chan*) dest->getBits()) + (dsty * outStride) + (dstx * 4);
62 
63  if ((mode & LICE_BLIT_MODE_MASK) == LICE_BLIT_MODE_ADD)
64  {
65  for (int i = 0; i < srch; i++, in += inStride, out += outStride)
66  {
67  for (int j = 0; j < srcw; j++)
68  PreMulCompositeAdd(out + j * 4, in + j * 4);
69  }
70  }
71  else
72  {
73  for (int i = 0; i < srch; i++, in += inStride, out += outStride)
74  {
75  for (int j = 0; j < srcw; j++)
76  PreMulCompositeSourceOver(out + j * 4, in + j * 4);
77  }
78  }
79 }

424 void IGraphicsLice::FillArc(const IColor& color, float cx, float cy, float r, float aMin, float aMax, const IBlend* pBlend)
425 {
426  NeedsClipping();
427 
428  if (aMax < aMin)
429  std::swap(aMin, aMax);
430 
431  if (aMax >= aMin + 360.f)
432  {
433  FillCircle(color, cx, cy, r, pBlend);
434  return;
435  }
436 
437  float xarray[181];
438  float yarray[181];
439 
440  if (aMax > aMin + 180.f)
441  {
442  if (!OpacityCheck(color, pBlend))
443  {
444  OpacityLayer(&IGraphicsLice::FillArc, pBlend, color, cx, cy, r, aMin, aMax, nullptr);
445  return;
446  }
447 
448  FillArc(color, cx, cy, r, aMin + 178.f, aMax, pBlend);
449  aMax = aMin + 180.f;
450  }
451 
452  aMin = DegToRad(aMin-90.f);
453  aMax = DegToRad(aMax-90.f);
454 
455  int arcpoints = 180.0 * std::min(1., (aMax - aMin) / PI);
456  double arcincrement = (aMax - aMin) / arcpoints;
457  for(int i = 0; i < arcpoints; i++)
458  {
459  xarray[i] = cx + cosf(i * arcincrement + aMin) * r;
460  yarray[i] = cy + sinf(i * arcincrement + aMin) * r;
461  }
462 
463  xarray[arcpoints] = cx;
464  yarray[arcpoints] = cy;
465 
466  FillConvexPolygon(color, xarray, yarray, arcpoints + 1, pBlend);
467 }

476 bool IGraphicsLice::DoDrawMeasureText(const IText& text, const char* str, IRECT& bounds, const IBlend* pBlend, bool measure)
477 {
478  const int ds = GetScreenScale();
479  if (!str || str[0] == '\0')
480  {
481  return true;
482  }
483 
484  LICE_IFont* font = text.mCached;
485 
486  if (!font || text.mCachedScale != ds)
487  {
488  font = CacheFont(text, ds);
489  if (!font) return false;
490  }
491 
492  LICE_pixel color;
493 
494  if (GetTextEntryControl() && GetTextEntryControl()->GetRECT() == bounds)
495  color = LiceColor(text.mTextEntryFGColor, pBlend);
496  else
497  color = LiceColor(text.mFGColor, pBlend);
498 
499  font->SetTextColor(color);
500 
501  UINT fmt = DT_NOCLIP;
502  if (LICE_GETA(color) < 255) fmt |= LICE_DT_USEFGALPHA;
503  if (text.mAlign == IText::kAlignNear)
504  fmt |= DT_LEFT;
505  else if (text.mAlign == IText::kAlignCenter)
506  fmt |= DT_CENTER;
507  else // if (text.mAlign == IText::kAlignFar)
508  fmt |= DT_RIGHT;
509 
510  if (text.mVAlign == IText::kVAlignTop)
511  fmt |= DT_TOP;
512  else if (text.mVAlign == IText::kVAlignBottom)
513  fmt |= DT_BOTTOM;
514  else if (text.mVAlign == IText::kVAlignMiddle)
515  fmt |= DT_VCENTER;
516 
517  if (measure)
518  {
519  fmt |= DT_CALCRECT;

769 void IGraphicsLice::ApplyShadowMask(ILayerPtr& layer, RawBitmapData& mask, const IShadow& shadow)
770 {
771  const APIBitmap* pBitmap = layer->GetAPIBitmap();
772  LICE_IBitmap* pLayerBitmap = pBitmap->GetBitmap();
773 
774  int stride = pLayerBitmap->getRowSpan() * 4;
775  int size = pLayerBitmap->getHeight() * stride;
776 
777  if (mask.GetSize() >= size)
778  {
779  int x = std::round(shadow.mXOffset * GetScreenScale());
780  int y = std::round(shadow.mYOffset * GetScreenScale());
781  int nRows = pBitmap->GetHeight() - std::abs(y);
782  int nCols = pBitmap->GetWidth() - std::abs(x);
783  LICE_pixel_chan* in = mask.Get() + (std::max(-x, 0) * 4) + (std::max(-y, 0) * stride);
784  LICE_pixel_chan* out = ((LICE_pixel_chan*) pLayerBitmap->getBits()) + (std::max(x, 0) * 4) + (std::max(y, 0) * stride);
785 
786  // Pre-multiply color components
787 
788  IColor color = shadow.mPattern.GetStop(0).mColor;
789  color.Clamp();
790  unsigned int ia = (color.A * static_cast<int>(Clip(shadow.mOpacity, 0.f, 1.f) * 255.0));
791  unsigned int ir = (color.R * ia);
792  unsigned int ig = (color.G * ia);
793  unsigned int ib = (color.B * ia);
794 
795  if (!shadow.mDrawForeground)
796  {
797  LICE_Clear(pLayerBitmap, 0);
798 
799  for (int i = 0; i < nRows; i++, in += stride, out += stride)
800  {
801  LICE_pixel_chan* chans = out;
802 
803  for (int j = 0; j < nCols; j++, chans += 4)
804  {
805  unsigned int maskAlpha = in[j * 4 + LICE_PIXEL_A];
806 
807  unsigned int A = (ia * maskAlpha) >> 16;
808  unsigned int R = (ir * maskAlpha) >> 16;
809  unsigned int G = (ig * maskAlpha) >> 16;
810  unsigned int B = (ib * maskAlpha) >> 16;
811 
812  _LICE_MakePixelNoClamp(chans, R, G, B, A);
813  }
814  }
815  }
816  else
817  {
818  for (int i = 0; i < nRows; i++, in += stride, out += stride)
819  {
820  LICE_pixel_chan* chans = out;
821 
822  for (int j = 0; j < nCols; j++, chans += 4)
823  {
824  unsigned int maskAlpha = in[j * 4 + LICE_PIXEL_A];
825  unsigned int alphaCmp = 255 - chans[LICE_PIXEL_A];
826 
827  unsigned int A = chans[LICE_PIXEL_A] + ((alphaCmp * ia * maskAlpha) >> 24);
828  unsigned int R = chans[LICE_PIXEL_R] + ((alphaCmp * ir * maskAlpha) >> 24);
829  unsigned int G = chans[LICE_PIXEL_G] + ((alphaCmp * ig * maskAlpha) >> 24);
830  unsigned int B = chans[LICE_PIXEL_B] + ((alphaCmp * ib * maskAlpha) >> 24);
831 
832  _LICE_MakePixelClamp(chans, R, G, B, A);
833  }
834  }
835  }
836  }
837 }
