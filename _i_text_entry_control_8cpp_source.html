128 bool ITextEntryControl::OnKeyDown(float x, float y, const IKeyPress& key)

136  //TODO: Select All

144  case 'C':

147  return false;
148  }
149  case 'V':
150  {
151  //TODO: Paste
152  return false;
153  }
154 
155  default:
156  break;
157  }
158  }
159 
160  STB_TEXTEDIT_KEYTYPE stbKey = key.Ascii;
161 
162  switch (key.VK)
163  {
164  case kVK_SPACE: stbKey = kVK_SPACE; break;
165  case kVK_TAB: return false;
166  case kVK_DELETE: stbKey = kVK_DELETE; break;
167  case kVK_BACK: stbKey = kVK_BACK; break;
168  case kVK_RETURN: CommitEdit(); break;
169  case kVK_ESCAPE: DismissEdit(); break;
170  default:
171  {
172  if(key.VK >= '0' && key.VK <= '9')
173  break;
174  if(key.VK >= kVK_NUMPAD0 && key.VK <= kVK_NUMPAD9)
175  break;
176  if(key.VK >= 'A' && key.VK <= 'Z')
177  break;
178  else
179  // TODO: need to shift correct bits for VK
180 // stbKey = (key.VK) | VIRTUAL_KEY_BIT;
181  break;
182  }
183  }
184 
185  if (key.C)
186  stbKey |= STB_TEXTEDIT_K_CONTROL;
187  if (key.A)
188  stbKey |= STB_TEXTEDIT_K_ALT;
189  if (key.S)
190  stbKey |= STB_TEXTEDIT_K_SHIFT;
191 
192  return CallSTB([&]() { stb_textedit_key(this, &mEditState, stbKey); }) ? true : false;
193 }
