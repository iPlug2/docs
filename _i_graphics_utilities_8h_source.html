44 static inline void RadialPoints(float angleDegrees, float cx, float cy, float rMin, float rMax, int nPoints, float data[][2])
45 {
46  const float angleRadians = DegToRad(angleDegrees - 90.f);
47  const float sinV = sinf(angleRadians);
48  const float cosV = cosf(angleRadians);
49 
50  for(auto i = 0; i < nPoints; i++)
51  {
52  const float r = rMin+(rMax-rMin) * (float) i / float (nPoints-1);
53  data[i][0] = (cx + r * cosV);
54  data[i][1] = (cy + r * sinV);
55  }
56 }

61 static uint32_t GetAAXModifiersFromIMouseMod(const IMouseMod& mod)
62 {
63  uint32_t aax_mods = 0;
64 
65  if (mod.A) aax_mods |= AAX_eModifiers_Option; // ALT Key on Windows, ALT/Option key on mac
66 
67 #ifdef OS_WIN
68  if (mod.C) aax_mods |= AAX_eModifiers_Command;
69 #else
70  if (mod.C) aax_mods |= AAX_eModifiers_Control;
71  if (mod.R) aax_mods |= AAX_eModifiers_Command;
72 #endif
73  if (mod.S) aax_mods |= AAX_eModifiers_Shift;
74  if (mod.R) aax_mods |= AAX_eModifiers_SecondaryButton;
75 
76  return aax_mods;
77 }
