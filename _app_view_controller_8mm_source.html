38- (void)viewDidLoad
39{
40 [super viewDidLoad];
41
42#if PLUG_HAS_UI
43 NSString* storyBoardName = [NSString stringWithFormat:@"%s-iOS-MainInterface", PLUG_NAME];
44 UIStoryboard* storyboard = [UIStoryboard storyboardWithName:storyBoardName bundle:nil];
45 pluginVC = [storyboard instantiateViewControllerWithIdentifier:@"main"];
46 [self addChildViewController:pluginVC];
47#endif
48
49 AudioComponentDescription desc;
50
51#if PLUG_TYPE == 0
52 #if PLUG_DOES_MIDI_IN
53 desc.componentType = kAudioUnitType_MusicEffect;
54 #else
55 desc.componentType = kAudioUnitType_Effect;
56 #endif
57#elif PLUG_TYPE == 1
58 desc.componentType = kAudioUnitType_MusicDevice;
59#elif PLUG_TYPE == 2
60 desc.componentType = 'aumi';
61#endif
62
63 desc.componentSubType = PLUG_UNIQUE_ID;
64 desc.componentManufacturer = PLUG_MFR_ID;
65 desc.componentFlags = 0;
66 desc.componentFlagsMask = 0;
67
68 [AUAudioUnit registerSubclass:IPLUG_AUAUDIOUNIT.class
69 asComponentDescription:desc
70 name:@"Local AUv3"
71 version:UINT32_MAX];
72
73 player = [[IPlugAUPlayer alloc] initWithComponentType:desc.componentType];
74
75 [player loadAudioUnitWithComponentDescription:desc
76 completion:^{
77 self->pluginVC.audioUnit = (IPLUG_AUAUDIOUNIT*)self->player.currentAudioUnit;
78
79 [self embedPlugInView];
80 }];
81
82 [[NSNotificationCenter defaultCenter] addObserver:self
83 selector:@selector(receiveNotification:)
84 name:@"LaunchBTMidiDialog"
85 object:nil];
86}

88- (void)receiveNotification:(NSNotification*)notification
89{
90 if ([notification.name isEqualToString:@"LaunchBTMidiDialog"])
91 {
92 NSDictionary* dict = notification.userInfo;
93 NSNumber* x = (NSNumber*)dict[@"x"];
94 NSNumber* y = (NSNumber*)dict[@"y"];
95
96 CABTMIDICentralViewController* vc = [[CABTMIDICentralViewController alloc] init];
97 UINavigationController* nc = [[UINavigationController alloc] initWithRootViewController:vc];
98 nc.modalPresentationStyle = UIModalPresentationPopover;
99
100 UIPopoverPresentationController* ppc = nc.popoverPresentationController;
101 ppc.permittedArrowDirections = UIPopoverArrowDirectionAny;
102 ppc.sourceView = self.view;
103 ppc.sourceRect = CGRectMake([x floatValue], [y floatValue], 1., 1.);
104
105 [self presentViewController:nc animated:YES completion:nil];
106 }
107}

109- (void)embedPlugInView
110{

112 UIView* view = pluginVC.view;
113 view.frame = auView.bounds;
114 [auView addSubview:view];
115
116 view.translatesAutoresizingMaskIntoConstraints = NO;
117
118 NSArray* constraints = [NSLayoutConstraint constraintsWithVisualFormat:@"H:|[view]|"
119 options:0
120 metrics:nil
121 views:NSDictionaryOfVariableBindings(view)];
122 [auView addConstraints:constraints];
123
124 constraints = [NSLayoutConstraint constraintsWithVisualFormat:@"V:|[view]|"
125 options:0
126 metrics:nil
127 views:NSDictionaryOfVariableBindings(view)];
128 [auView addConstraints:constraints];
129#endif
130}
