118  void OnMouseDrag(float x, float y, float dX, float dY, const IMouseMod& mod) override
119  {
120  float mouseDownX, mouseDownY;
121  GetUI()->GetMouseDownPoint(mouseDownX, mouseDownY);
122 
123  if(mClickedOnControl > 0)
124  {
125  IControl* pControl = GetUI()->GetControl(mClickedOnControl);
126  IRECT r = pControl->GetRECT();
127 
128  if(mMouseClickedOnResizeHandle)
129  {
130  r.R = SnapToGrid(mMouseDownRECT.R + (x - mouseDownX));
131  r.B = SnapToGrid(mMouseDownRECT.B + (y - mouseDownY));
132 
133  if(r.R < mMouseDownRECT.L +mGridSize) r.R = mMouseDownRECT.L+mGridSize;
134  if(r.B < mMouseDownRECT.T +mGridSize) r.B = mMouseDownRECT.T+mGridSize;
135  }
136  else
137  {
138  r.L = SnapToGrid(mMouseDownRECT.L + (x - mouseDownX));
139  r.T = SnapToGrid(mMouseDownRECT.T + (y - mouseDownY));
140  r.R = r.L + mMouseDownRECT.W();
141  r.B = r.T + mMouseDownRECT.H();
142  }
143 
144  pControl->SetRECT(r);
145  pControl->SetTargetRECT(r);
146 
147  DBGMSG("%i, %i, %i, %i\n", (int) r.L, (int) r.T, (int) r.R, (int) r.B);
148 
149  GetUI()->SetAllControlsDirty();
150  }
