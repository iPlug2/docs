76 template<typename T>
77 void IPlugProcessor<T>::ProcessBlock(T** inputs, T** outputs, int nFrames)
78 {
79  int i, nIn = mChannelData[ERoute::kInput].GetSize(), nOut = mChannelData[ERoute::kOutput].GetSize();
80  int j = 0;
81  for (i = 0; i < nOut; ++i)
82  {
83  if (i < nIn)
84  {
85  memcpy(outputs[i], inputs[i], nFrames * sizeof(T));
86  j++;
87  }
88  }
89  // zero remaining outs
90  for (/* same j */; j < nOut; ++j)
91  {
92  memset(outputs[j], 0, nFrames * sizeof(T));
93  }
94 }

219 int IPlugProcessor<T>::ParseChannelIOStr(const char* IOStr, WDL_PtrList<IOConfig>& channelIOList, int& totalNInChans, int& totalNOutChans, int& totalNInBuses, int& totalNOutBuses)
220 {
221  bool foundAWildcard = false;
222  int IOConfigIndex = 0;
223 
224  DBGMSG("BEGIN IPLUG CHANNEL IO PARSER --------------------------------------------------\n");
225  // lamda function to iterate through the period separated buses and check that none have 0 channel count
226  auto ParseBusToken = [&foundAWildcard, &IOConfigIndex](ERoute busDir, char* pBusStr, char* pBusStrEnd, int& NBuses, int& NChans, IOConfig* pConfig)
227  {
228  while (pBusStr != NULL)
229  {
230  auto NChanOnBus = 0;
231 
232  if(strcmp(pBusStr, "*") == 0)
233  {
234  foundAWildcard = true;
235  NChanOnBus = -MAX_BUS_CHANS; // we put a negative number in here which will be picked up in the api classes in order to deal with NxN or NxM routings
236  }
237  else if (sscanf(pBusStr, "%d", &NChanOnBus) == 1)
238  ; //don't do anything
239  else
240  {
241  DBGMSG("Error: something wrong in the %s part of this io string: %s.\n", RoutingDirStrs[busDir], pBusStr);
242  assert(0);
243  }
244  NChans += NChanOnBus;
245 
246  pBusStr = strtok_r(NULL, ".", &pBusStrEnd);
247 
248  if(NChanOnBus)
249  {
250  pConfig->AddBusInfo(busDir, NChanOnBus, RoutingDirStrs[busDir]);
251  NBuses++;
252  }
253  else if(NBuses > 0)
254  {
255  DBGMSG("Error: with multiple %s buses you can't define one with no channels!\n", RoutingDirStrs[busDir]);
256  assert(NChanOnBus > 0);
257  }
258  }
259  };
260 
261  totalNInChans = 0; totalNOutChans = 0;
262  totalNInBuses = 0; totalNOutBuses = 0;
263 
264  char* pChannelIOStr = strdup(IOStr);
265 
266  char* pIOStrEnd;
267  char* pIOStr = strtok_r(pChannelIOStr, " ", &pIOStrEnd); // a single IO string
268 
269  WDL_PtrList<WDL_String> IOStrlist;
270 
271  // iterate through the space separated IO configs
272  while (pIOStr != NULL)
273  {
274  IOConfig* pConfig = new IOConfig();
275 
276  int NInChans = 0, NOutChans = 0;
277  int NInBuses = 0, NOutBuses = 0;
278 
279  char* pIStr = strtok(pIOStr, "-"); // Input buses part of string
280  char* pOStr = strtok(NULL, "-"); // Output buses part of string
281 
282  WDL_String* thisIOStr = new WDL_String();
283  thisIOStr->SetFormatted(10, "%s-%s", pIStr, pOStr);
284 
285  for (auto str = 0; str < IOStrlist.GetSize(); str++)
286  {
287  if(strcmp(IOStrlist.Get(str)->Get(), thisIOStr->Get()) == 0)
288  {
289  DBGMSG("Error: Duplicate IO string. %s\n", thisIOStr->Get());
290  assert(0);
291  }
292  }
293 
294  IOStrlist.Add(thisIOStr);
295 
296  char* pIBusStrEnd;
297  char* pIBusStr = strtok_r(pIStr, ".", &pIBusStrEnd); // a single input bus
298 
299  ParseBusToken(ERoute::kInput, pIBusStr, pIBusStrEnd, NInBuses, NInChans, pConfig);
300 
301  char* pOBusStrEnd;
302  char* pOBusStr = strtok_r(pOStr, ".", &pOBusStrEnd);
303 
304  ParseBusToken(ERoute::kOutput, pOBusStr, pOBusStrEnd, NOutBuses, NOutChans, pConfig);
305 
306  if(foundAWildcard == true && IOConfigIndex > 0)
307  {
308  DBGMSG("Error: You can only have a single IO config when using wild cards.\n");
309  assert(0);
310  }
311 
312  DBGMSG("Channel I/O #%i - %s\n", IOConfigIndex + 1, thisIOStr->Get());
313  DBGMSG(" - input bus count: %i, output bus count %i\n", NInBuses, NOutBuses);
314  for (auto i = 0; i < NInBuses; i++)
315  DBGMSG(" - channel count on input bus %i: %i\n", i + 1, pConfig->NChansOnBusSAFE(ERoute::kInput, i));
316  for (auto i = 0; i < NOutBuses; i++)
317  DBGMSG(" - channel count on output bus %i: %i\n", i + 1, pConfig->NChansOnBusSAFE(ERoute::kOutput, i));
318  DBGMSG(" - input channel count across all buses: %i, output channel count across all buses %i\n\n", NInChans, NOutChans);
319 
320  totalNInChans = std::max(totalNInChans, NInChans);
321  totalNOutChans = std::max(totalNOutChans, NOutChans);
322  totalNInBuses = std::max(totalNInBuses, NInBuses);
323  totalNOutBuses = std::max(totalNOutBuses, NOutBuses);
324 
325  channelIOList.Add(pConfig);
326 
327  IOConfigIndex++;
328 
329  pIOStr = strtok_r(NULL, " ", &pIOStrEnd); // go to next io string
330  }
331 
332  free(pChannelIOStr);
333  IOStrlist.Empty(true);
334  DBGMSG("%i I/O configs detected\n", IOConfigIndex);
335  DBGMSG("Total # in chans: %i, Total # out chans: %i \n\n", totalNInChans, totalNOutChans);
336  DBGMSG("END IPLUG CHANNEL IO PARSER --------------------------------------------------\n");
337 
338  return IOConfigIndex;
339 }
