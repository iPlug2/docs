19 static uint64_t GetAPIBusTypeForChannelIOConfig(int configIdx, ERoute dir, int busIdx, IOConfig* pConfig)
20 {
21  assert(pConfig != nullptr);
22  assert(busIdx >= 0 && busIdx < pConfig->NBuses(dir));
23 
24  int numChans = pConfig->GetBusInfo(dir, busIdx)->mNChans;
25 
26  switch (numChans)
27  {
28  case 0: return SpeakerArr::kEmpty;
29  case 1: return SpeakerArr::kMono;
30  case 2: return SpeakerArr::kStereo;
31  case 3: return SpeakerArr::k30Cine; // CHECK - not the same as protools
32  case 4: return SpeakerArr::kAmbi1stOrderACN;
33  case 5: return SpeakerArr::k50;
34  case 6: return SpeakerArr::k51;
35  case 7: return SpeakerArr::k70Cine;
36  case 8: return SpeakerArr::k71CineSideFill; // CHECK - not the same as protools
37  case 9: return SpeakerArr::kAmbi2cdOrderACN;
38  case 10:return SpeakerArr::k71_2; // aka k91Atmos
39  case 16:return SpeakerArr::kAmbi3rdOrderACN;
40  default:
41  DBGMSG("do not yet know what to do here\n");
42  assert(0);
43  return SpeakerArr::kEmpty;
44  }
45 }

208 tresult PLUGIN_API IPlugVST3Processor::process(ProcessData& data)
209 {
210  TRACE;
211 
212  if(data.processContext)
213  memcpy(&mProcessContext, data.processContext, sizeof(ProcessContext));
214 
215  PrepareProcessContext();
216 
217  //process parameters
218  IParameterChanges* paramChanges = data.inputParameterChanges;
219  if (paramChanges)
220  {
221  int32 numParamsChanged = paramChanges->getParameterCount();
222 
223  //it is possible to get a finer resolution of control here by retrieving more values (points) from the queue
224  //for now we just grab the last one
225 
226  for (int32 i = 0; i < numParamsChanged; i++)
227  {
228  IParamValueQueue* paramQueue = paramChanges->getParameterData(i);
229  if (paramQueue)
230  {
231  int32 numPoints = paramQueue->getPointCount();
232  int32 offsetSamples;
233  double value;
234 
235  if (paramQueue->getPoint(numPoints - 1, offsetSamples, value) == kResultTrue)
236  {
237  int idx = paramQueue->getParameterId();
238 
239  switch (idx)
240  {
241  case kBypassParam:
242  {
243  const bool bypassed = (value > 0.5);
244 
245  if (bypassed != GetBypassed())
246  SetBypassed(bypassed);
247 
248  break;
249  }
250  case kPresetParam:
251  //RestorePreset((int)round(FromNormalizedParam(value, 0, NPresets(), 1.))); // TODO
252  break;
253  //TODO: pitch bend, modwheel etc
254  default:
255  {
256  if (idx >= 0 && idx < NParams())
257  {
258  ENTER_PARAMS_MUTEX;
259  GetParam(idx)->SetNormalized((double)value);
260  OnParamChange(idx, kHost, offsetSamples);
261  LEAVE_PARAMS_MUTEX;
262  }
263  }
264  break;
265  }
266 
267  }
268  }
269  }
270  }
271 
272  if(DoesMIDIIn())
273  {
274  IMidiMsg msg;
275 
276  //process events.. only midi note on and note off?
277  IEventList* eventList = data.inputEvents;
278  if (eventList)
279  {
280  int32 numEvent = eventList->getEventCount();
281  for (int32 i=0; i<numEvent; i++)
282  {
283  Event event;
284  if (eventList->getEvent(i, event) == kResultOk)
285  {
286  switch (event.type)
287  {
288  case Event::kNoteOnEvent:
289  {
290  msg.MakeNoteOnMsg(event.noteOn.pitch, event.noteOn.velocity * 127, event.sampleOffset, event.noteOn.channel);
291  ProcessMidiMsg(msg);
292  mMidiMsgsFromProcessor.Push(msg);
293  break;
294  }
295 
296  case Event::kNoteOffEvent:
297  {
298  msg.MakeNoteOffMsg(event.noteOff.pitch, event.sampleOffset, event.noteOff.channel);
299  ProcessMidiMsg(msg);
300  mMidiMsgsFromProcessor.Push(msg);
301  break;
302  }
303  case Event::kPolyPressureEvent:
304  {
305  msg.MakePolyATMsg(event.polyPressure.pitch, event.polyPressure.pressure * 127., event.sampleOffset, event.polyPressure.channel);
306  ProcessMidiMsg(msg);
307  mMidiMsgsFromProcessor.Push(msg);
308  break;
309  }
310  case Event::kDataEvent:
311  {
312  ISysEx syx = ISysEx(event.sampleOffset, event.data.bytes, event.data.size);
313  ProcessSysEx(syx);
314  //mSysexMsgsFromProcessor.Push
315  break;
316  }
317  }
318  }
319  }
320  }
321 
322  while (mMidiMsgsFromEditor.Pop(msg))
323  {
324  ProcessMidiMsg(msg);
325  }
326  }
327 
328 #pragma mark process single precision
329 
330  if (processSetup.symbolicSampleSize == kSample32)
331  {
332  if (data.numInputs)
333  {
334  if (HasSidechainInput())
335  {
336  if (getAudioInput(1)->isActive()) // Sidechain is active
337  {
338  mSidechainActive = true;
339  SetChannelConnections(ERoute::kInput, 0, MaxNChannels(ERoute::kInput), true);
340  }
341  else
342  {
343  if (mSidechainActive)
344  {
345  ZeroScratchBuffers();
346  mSidechainActive = false;
347  }
348 
349  SetChannelConnections(ERoute::kInput, 0, MaxNChannels(ERoute::kInput), true);
350  SetChannelConnections(ERoute::kInput, data.inputs[0].numChannels, MaxNChannels(ERoute::kInput) - NSidechainChannels(), false);
351  }
352 
353  AttachBuffers(ERoute::kInput, 0, MaxNChannels(ERoute::kInput) - NSidechainChannels(), data.inputs[0].channelBuffers32, data.numSamples);
354  AttachBuffers(ERoute::kInput, NSidechainChannels(), MaxNChannels(ERoute::kInput) - NSidechainChannels(), data.inputs[1].channelBuffers32, data.numSamples);
355  }
356  else
357  {
358  SetChannelConnections(ERoute::kInput, 0, data.inputs[0].numChannels, true);
359  SetChannelConnections(ERoute::kInput, data.inputs[0].numChannels, MaxNChannels(ERoute::kInput) - data.inputs[0].numChannels, false);
360  AttachBuffers(ERoute::kInput, 0, MaxNChannels(ERoute::kInput), data.inputs[0].channelBuffers32, data.numSamples);
361  }
362  }
363 
364  for (int outBus = 0, chanOffset = 0; outBus < data.numOutputs; outBus++)
365  {
366  int busChannels = data.outputs[outBus].numChannels;
367  SetChannelConnections(ERoute::kOutput, chanOffset, busChannels, (bool) getAudioOutput(outBus)->isActive());
368  SetChannelConnections(ERoute::kOutput, chanOffset + busChannels, MaxNChannels(ERoute::kOutput) - (chanOffset + busChannels), false);
369  AttachBuffers(ERoute::kOutput, chanOffset, busChannels, data.outputs[outBus].channelBuffers32, data.numSamples);
370  chanOffset += busChannels;
371  }
372 
373  if (GetBypassed())
374  PassThroughBuffers(0.0f, data.numSamples);
375  else
376  ProcessBuffers(0.0f, data.numSamples); // process buffers single precision
377  }
378 
379 #pragma mark process double precision
380 
381  else if (processSetup.symbolicSampleSize == kSample64)
382  {
383  if (data.numInputs)
384  {
385  if (HasSidechainInput())
386  {
387  if (getAudioInput(1)->isActive()) // Sidechain is active
388  {
389  mSidechainActive = true;
390  SetChannelConnections(ERoute::kInput, 0, MaxNChannels(ERoute::kInput), true);
391  }
392  else
393  {
394  if (mSidechainActive)
395  {
396  ZeroScratchBuffers();
397  mSidechainActive = false;
398  }
399 
400  SetChannelConnections(ERoute::kInput, 0, MaxNChannels(ERoute::kInput), true);
401  SetChannelConnections(ERoute::kInput, data.inputs[0].numChannels, MaxNChannels(ERoute::kInput) - NSidechainChannels(), false);
402  }
403 
404  AttachBuffers(ERoute::kInput, 0, MaxNChannels(ERoute::kInput) - NSidechainChannels(), data.inputs[0].channelBuffers64, data.numSamples);
405  AttachBuffers(ERoute::kInput, NSidechainChannels(), MaxNChannels(ERoute::kInput) - NSidechainChannels(), data.inputs[1].channelBuffers64, data.numSamples);
406  }
407  else
408  {
409  SetChannelConnections(ERoute::kInput, 0, data.inputs[0].numChannels, true);
410  SetChannelConnections(ERoute::kInput, data.inputs[0].numChannels, MaxNChannels(ERoute::kInput) - data.inputs[0].numChannels, false);
411  AttachBuffers(ERoute::kInput, 0, MaxNChannels(ERoute::kInput), data.inputs[0].channelBuffers64, data.numSamples);
412  }
413  }
414 
415  for (int outBus = 0, chanOffset = 0; outBus < data.numOutputs; outBus++)
416  {
417  int busChannels = data.outputs[outBus].numChannels;
418  SetChannelConnections(ERoute::kOutput, chanOffset, busChannels, (bool) getAudioOutput(outBus)->isActive());
419  SetChannelConnections(ERoute::kOutput, chanOffset + busChannels, MaxNChannels(ERoute::kOutput) - (chanOffset + busChannels), false);
420  AttachBuffers(ERoute::kOutput, chanOffset, busChannels, data.outputs[outBus].channelBuffers64, data.numSamples);
421  chanOffset += busChannels;
422  }
423 
424  if (GetBypassed())
425  PassThroughBuffers(0.0, data.numSamples);
426  else
427  ProcessBuffers(0.0, data.numSamples); // process buffers double precision
428  }
429 
430  if (DoesMIDIOut())
431  {
432  IEventList* outputEvents = data.outputEvents;
433 
434  //MIDI
435  if (!mMidiOutputQueue.Empty() && outputEvents)
436  {
437  Event toAdd = {0};
438  IMidiMsg msg;
439 
440  while (!mMidiOutputQueue.Empty())
441  {
442  IMidiMsg& msg = mMidiOutputQueue.Peek();
443 
444  if (msg.StatusMsg() == IMidiMsg::kNoteOn)
445  {
446  toAdd.type = Event::kNoteOnEvent;
447  toAdd.noteOn.channel = msg.Channel();
448  toAdd.noteOn.pitch = msg.NoteNumber();
449  toAdd.noteOn.tuning = 0.;
450  toAdd.noteOn.velocity = (float) msg.Velocity() * (1.f / 127.f);
451  toAdd.noteOn.length = -1;
452  toAdd.noteOn.noteId = -1; // TODO ?
453  toAdd.sampleOffset = msg.mOffset;
454  outputEvents->addEvent(toAdd);
455  }
456  else if (msg.StatusMsg() == IMidiMsg::kNoteOff)
457  {
458  toAdd.type = Event::kNoteOffEvent;
459  toAdd.noteOff.channel = msg.Channel();
460  toAdd.noteOff.pitch = msg.NoteNumber();
461  toAdd.noteOff.velocity = (float) msg.Velocity() * (1.f / 127.f);
462  toAdd.noteOff.noteId = -1; // TODO ?
463  toAdd.sampleOffset = msg.mOffset;
464  outputEvents->addEvent(toAdd);
465  }
466  else if (msg.StatusMsg() == IMidiMsg::kPolyAftertouch)
467  {
468  toAdd.type = Event::kPolyPressureEvent;
469  toAdd.polyPressure.channel = msg.Channel();
470  toAdd.polyPressure.pitch = msg.NoteNumber();
471  toAdd.polyPressure.pressure = (float) msg.PolyAfterTouch() * (1.f / 127.f);
472  toAdd.polyPressure.noteId = -1; // TODO ?
473  toAdd.sampleOffset = msg.mOffset;
474  outputEvents->addEvent(toAdd);
475  }
476 
477  mMidiOutputQueue.Remove();
478  }
479  }
480 
481  mMidiOutputQueue.Flush(data.numSamples);
482 
483  //Output SYSEX from the editor, which has bypassed the processors' ProcessSysEx()
484  if(mSysExDataFromEditor.ElementsAvailable())
485  {
486  Event toAdd = {0};
487 
488  while (mSysExDataFromEditor.Pop(mSysexBuf))
489  {
490  toAdd.type = Event::kDataEvent;
491  toAdd.sampleOffset = mSysexBuf.mOffset;
492  toAdd.data.type = DataEvent::kMidiSysEx;
493  toAdd.data.size = mSysexBuf.mSize;
494  toAdd.data.bytes = (uint8*) mSysexBuf.mData; // TODO! this is a problem if more than one message in this block!
495  outputEvents->addEvent(toAdd);
496  }
497  }
498  }
499 
500  return kResultOk;
501 }

521 tresult PLUGIN_API IPlugVST3Processor::setState(IBStream* state)
522 {
523  TRACE;
524 
525  IByteChunk chunk;
526 
527  const int bytesPerBlock = 128;
528  char buffer[bytesPerBlock];
529 
530  while(true)
531  {
532  Steinberg::int32 bytesRead = 0;
533  auto status = state->read (buffer, (Steinberg::int32) bytesPerBlock, &bytesRead);
534 
535  if (bytesRead <= 0 || (status != kResultTrue && GetHost() != kHostWaveLab))
536  break;
537 
538  chunk.PutBytes(buffer, bytesRead);
539  }
540  int pos = UnserializeState(chunk,0);
541 
542  int32 savedBypass = 0;
543 
544  state->seek(pos,IBStream::IStreamSeekMode::kIBSeekSet);
545  if (state->read (&savedBypass, sizeof (Steinberg::int32)) != kResultOk) {
546  return kResultFalse;
547  }
548 
549  SetBypassed((bool) savedBypass);
550 
551  OnRestoreState();
552  return kResultOk;
553 }

555 tresult PLUGIN_API IPlugVST3Processor::getState(IBStream* state)
556 {
557  TRACE;
558  IByteChunk chunk;
559  IByteChunk::InitChunkWithIPlugVer(chunk);
560  int bypassState = GetBypassed();
561  chunk.Put(&bypassState);
562 
563  if (SerializeState(chunk))
564  {
565  int chunkSize = chunk.Size();
566  void* data = (void*) &chunkSize;
567  state->write(data, (int32) sizeof(int));
568  state->write(chunk.GetData(), chunkSize);
569  }
570  else
571  return kResultFalse;
572 
573  return kResultTrue;
574 }
