98 void OnMouseDown(float x, float y, const IMouseMod& mod) override
99 {
100 int prevKey = mLastTouchedKey;
101 mLastTouchedKey = GetKeyAtPoint(x, y);
102
103 SetKeyIsPressed(mLastTouchedKey, true);
104
105 mMouseOverKey = mLastTouchedKey;
106
107 if (mLastTouchedKey != prevKey)
108 {
109 mLastVelocity = GetVelocity(y);
110
111 TriggerMidiMsgFromKeyPress(mLastTouchedKey, (int)(mLastVelocity * 127.f));
112 }
113
114 SetDirty(true);
115 }

189 void OnResize() override
190 {
191 float r = mRECT.W() / mTargetRECT.W();
192 float dx = mRECT.L - mTargetRECT.L;
193 mWKWidth *= r;
194 for (int i = 0; i < NKeys(); ++i)
195 {
196 float* pKeyL = GetKeyXPos(i);
197 float d = *pKeyL - mRECT.L;
198 *pKeyL = mRECT.L + d * r + dx;
199 }
200
201 mTargetRECT = mRECT;
202 RecreateKeyBounds(true);
203 SetDirty(false);
204 }

206 void OnMidi(const IMidiMsg& msg) override
207 {
208 switch (msg.StatusMsg())
209 {
210 case IMidiMsg::kNoteOn: SetNoteFromMidi(msg.NoteNumber(), (msg.Velocity() != 0)); break;
211 case IMidiMsg::kNoteOff: SetNoteFromMidi(msg.NoteNumber(), false); break;
212 case IMidiMsg::kControlChange:
213 if (msg.ControlChangeIdx() == IMidiMsg::kAllNotesOff)
214 ClearNotesFromMidi();
215 break;
216 default: break;
217 }
218
219 SetDirty(false);

232 void Draw(IGraphics& g) override
233 {
234 IColor shadowColor = IColor(60, 0, 0, 0);
235
236 float BKBottom = mRECT.T + mRECT.H() * mBKHeightRatio;
237 float BKWidth = GetBKWidth();
238
239 // first draw white keys
240 for (int i = 0; i < NKeys(); ++i)
241 {
242 if (!IsBlackKey(i))
243 {
244 float kL = *GetKeyXPos(i);
245 IRECT keyBounds = IRECT(kL, mRECT.T, kL + mWKWidth, mRECT.B);
246
247 DrawKey(g, keyBounds, i == mHighlight ? mHK_COLOR : mWK_COLOR);
248
249 if (GetKeyIsPressed(i))
250 {
251 // draw played white key
252 DrawKey(g, keyBounds, mPK_COLOR);
253
254 if (mDrawShadows)
255 {
256 IRECT shadowBounds = keyBounds;
257 shadowBounds.R = shadowBounds.L + 0.35f * shadowBounds.W();
258
259 if (!mRoundedKeys)
260 g.FillRect(shadowColor, shadowBounds, &mBlend);
261 else
262 {
263 g.FillRoundRect(shadowColor, shadowBounds, 0., 0., mRoundness, mRoundness, &mBlend); // this one looks strange with rounded corners
264 }
265 }
266 }
267 if (mDrawFrame && i != 0)
268 { // only draw the left border if it doesn't overlay mRECT left border
269 g.DrawLine(mFR_COLOR, kL, mRECT.T, kL, mRECT.B, &mBlend, mFrameThickness);
270 if (i == NKeys() - 2 && IsBlackKey(NKeys() - 1))
271 g.DrawLine(mFR_COLOR, kL + mWKWidth, mRECT.T, kL + mWKWidth, mRECT.B, &mBlend, mFrameThickness);
272 }
273 }
274 }
275
276 // then blacks
277 for (int i = 0; i < NKeys(); ++i)
278 {
279 if (IsBlackKey(i))
280 {
281 float kL = *GetKeyXPos(i);
282 IRECT keyBounds = IRECT(kL, mRECT.T, kL + BKWidth, BKBottom);
283 // first draw underlying shadows
284 if (mDrawShadows && !GetKeyIsPressed(i) && i < NKeys() - 1)
285 {
286 IRECT shadowBounds = keyBounds;
287 float w = shadowBounds.W();
288 shadowBounds.L += 0.6f * w;
289 if (GetKeyIsPressed(i + 1))
290 {
291 // if white to the right is pressed, shadow is longer
292 w *= 1.3f;
293 shadowBounds.B = shadowBounds.T + 1.05f * shadowBounds.H();
294 }
295 shadowBounds.R = shadowBounds.L + w;
296 DrawKey(g, shadowBounds, shadowColor);
297 }
298 DrawKey(g, keyBounds, (i == mHighlight ? mHK_COLOR : mBK_COLOR.WithContrast(IsDisabled() ? GRAYED_ALPHA : 0.f)));
299
300 if (GetKeyIsPressed(i))
301 {
302 // draw pressed black key
303 IColor cBP = mPK_COLOR;
304 cBP.A = (int)mBKAlpha;
305 g.FillRect(cBP, keyBounds, &mBlend);
306 }
307
308 if (!mRoundedKeys)
309 {
310 // draw l, r and bottom if they don't overlay the mRECT borders
311 if (mBKHeightRatio != 1.0)
312 g.DrawLine(mFR_COLOR, kL, BKBottom, kL + BKWidth, BKBottom, &mBlend);
313 if (i > 0)
314 g.DrawLine(mFR_COLOR, kL, mRECT.T, kL, BKBottom, &mBlend);
315 if (i != NKeys() - 1)
316 g.DrawLine(mFR_COLOR, kL + BKWidth, mRECT.T, kL + BKWidth, BKBottom, &mBlend);
317 }
318 }
319 }
320
321 if (mDrawFrame)
322 g.DrawRect(mFR_COLOR, mRECT, &mBlend, mFrameThickness);
323
324 if (mShowNoteAndVel)
325 {
326 if (mMouseOverKey > -1)
327 {
328 IRECT r = IRECT(*GetKeyXPos(mMouseOverKey), mRECT.T, 0, 0);
329 r.B = r.T + 1.2f * mText.mSize;
330 r.R = r.L + 35.0f;
331 WDL_String t;
332 GetNoteNameStr(mMinNote + mMouseOverKey, false, t);
333 if (mLastTouchedKey > -1)
334 {
335 t.AppendFormatted(16, ", vel: %3.2f", mLastVelocity * 127.f);
336 r.R += 60.0;
337 }
338 float e = r.R - mRECT.R;
339 if (e > 0.0)
340 {
341 r.L -= e;
342 r.R -= e;
343 }
344 g.FillRect(mWK_COLOR, r, &mBlend);
345 g.DrawRect(mFR_COLOR, r, &mBlend);
346 g.DrawText(mText, t.Get(), r, &mBlend);
347 }
348 }
349
350#ifdef _DEBUG
351 // g.DrawRect(COLOR_GREEN, mTargetRECT);
352 // g.DrawRect(COLOR_BLUE, mRECT);
353 WDL_String ti;
354 ti.SetFormatted(32, "key: %d, vel: %3.2f", mLastTouchedKey, mLastVelocity * 127.f);
355 // ti.SetFormatted(16, "mBAlpha: %d", mBAlpha);
356 IText txt(20, COLOR_RED);
357 IRECT tr(mRECT.L + 20, mRECT.B - 20, mRECT.L + 160, mRECT.B);
358 g.DrawText(txt, ti.Get(), tr, &mBlend);
359#endif
360 }

410 void SetBlackToWhiteRatios(float widthRatio, float heightRatio = 0.6)
411 {
412 widthRatio = Clip(widthRatio, 0.1f, 1.f);
413 heightRatio = Clip(heightRatio, 0.1f, 1.f);
414
415 float halfW = 0.5f * mWKWidth * mBKWidthRatio;
416 float r = widthRatio / mBKWidthRatio;
417 mBKWidthRatio = widthRatio;
418 mBKHeightRatio = heightRatio;
419
420 for (int i = 0; i < NKeys(); ++i)
421 {
422 if (IsBlackKey(i))
423 {
424 float* pKeyL = GetKeyXPos(i);
425 float mid = *pKeyL + halfW;
426 *pKeyL = mid - halfW * r;
427 if (*pKeyL < mRECT.L)
428 *pKeyL = mRECT.L;
429 }
430 }
431
432 SetDirty(false);
433 }

473 void SetColors(const IColor BKColor, const IColor& WKColor, const IColor& PKColor = DEFAULT_PK_COLOR, const IColor& FRColor = DEFAULT_FR_COLOR)
474 {
475 mBK_COLOR = BKColor;
476 mWK_COLOR = WKColor;
477 mPK_COLOR = PKColor;
478 mFR_COLOR = FRColor;
479
480 mBKAlpha = (float)PKColor.A;
481
482 if (mBKAlpha < 240.f)
483 {
484 const float lumWK = WKColor.GetLuminosity() * WKColor.A / 255.f;
485 const float adjustment = PKColor.A / 255.f;
486 const float lumPK = PKColor.GetLuminosity() * adjustment;
487 const float lumRes = (1.f - adjustment) * lumWK + lumPK;
488 const float lumDW = lumRes - lumWK;
489 const float lumBK = BKColor.GetLuminosity() * BKColor.A / 255.f;
490
491 if ((lumDW < 0 && lumBK < lumWK) || (lumDW > 0 && lumBK > lumWK))
492 {
493 float dbWB = lumWK - lumBK; // not used in the conditions ^^ for readability
494 mBKAlpha += (255.f - mBKAlpha) * (1.f - dbWB * dbWB / 255.f / 255.f) + 0.5f;
495 }
496 else
497 mBKAlpha += lumDW + 0.5f;
498
499 mBKAlpha = Clip(mBKAlpha, 15.f, 255.f);
500 }
501
502 SetDirty(false);
503 }

516private:
517 void RecreateKeyBounds(bool keepWidth)
518 {
519 if (keepWidth)
520 mWKWidth = 0.f;
521
522 // create size-independent data.
523 mIsBlackKeyList.Resize(NKeys());
524 mKeyXPos.Resize(NKeys());
525
526 float numWhites = 0.f;
527 for (int n = mMinNote, i = 0; n <= mMaxNote; ++n, i++)
528 {
529 if (n % 12 == 1 || n % 12 == 3 || n % 12 == 6 || n % 12 == 8 || n % 12 == 10)
530 {
531 mIsBlackKeyList.Get()[i] = true;
532 }
533 else
534 {
535 mIsBlackKeyList.Get()[i] = false;
536 numWhites += 1.f;
537 }
538 }
539
540 // black key middle isn't aligned exactly between whites
541 float WKPadStart = 0.f; // 1st note may be black
542 float WKPadEnd = 0.f; // last note may be black
543
544 auto GetShiftForPitchClass = [this](int pitch) {
545 // usually black key width + distance to the closest black key = white key width,
546 // and often b width is ~0.6 * w width
547 if (pitch == 0)
548 return 0.f;
549 else if (pitch % 12 == 1)
550 return 7.f / 12.f;
551 else if (pitch % 12 == 3)
552 return 5.f / 12.f;
553 else if (pitch % 12 == 6)
554 return 2.f / 3.f;
555 else if (pitch % 12 == 8)
556 return 0.5f;
557 else if (pitch % 12 == 10)
558 return 1.f / 3.f;
559 else
560 return 0.f;
561 };
562
563 WKPadStart = GetShiftForPitchClass(mMinNote);
564
565 if (mMinNote != mMaxNote && IsBlackKey(mIsBlackKeyList.GetSize() - 1))
566 WKPadEnd = 1.f - GetShiftForPitchClass(mMaxNote);
567
568 // build rects
569 if (mWKWidth == 0.f)
570 mWKWidth = 0.2f * mRECT.H(); // first call from the constructor
571
572 if (keepWidth)
573 {
574 mWKWidth = mRECT.W();
575 if (numWhites)
576 mWKWidth /= (numWhites + mBKWidthRatio * (WKPadStart + WKPadEnd));
577 }
578
579 float BKWidth = mWKWidth;
580
581 if (numWhites)
582 BKWidth *= mBKWidthRatio;
583
584 float prevWKLeft = mRECT.L;
585
586 for (int k = 0; k < mIsBlackKeyList.GetSize(); ++k)
587 {
588 if (IsBlackKey(k))
589 {
590 float l = prevWKLeft;
591 if (k != 0)
592 {
593 l -= GetShiftForPitchClass(mMinNote + k) * BKWidth;
594 }
595 else
596 prevWKLeft += WKPadStart * BKWidth;
597 mKeyXPos.Get()[k] = l;
598 }
599 else
600 {
601 mKeyXPos.Get()[k] = prevWKLeft;
602 prevWKLeft += mWKWidth;
603 }
604 }
605
606 mTargetRECT = mRECT;
607 SetDirty(false);
608 }

610 int GetKeyAtPoint(float x, float y)
611 {
612 IRECT clipRect = mRECT.GetPadded(-2);
613 clipRect.Constrain(x, y);
614
615 float BKBottom = mRECT.T + mRECT.H() * mBKHeightRatio;
616 float BKWidth = GetBKWidth();
617
618 // black keys are on top
619 int k = -1;
620 for (int i = 0; i < NKeys(); ++i)
621 {
622 if (IsBlackKey(i))
623 {
624 float kL = *GetKeyXPos(i);
625 IRECT keyBounds = IRECT(kL, mRECT.T, kL + BKWidth, BKBottom);
626 if (keyBounds.Contains(x, y))
627 {
628 k = i;
629 break;
630 }
631 }
632 }
633
634 if (k == -1)
635 {
636 for (int i = 0; i < NKeys(); ++i)
637 {
638 if (!IsBlackKey(i))
639 {
640 float kL = *GetKeyXPos(i);
641 IRECT keyBounds = IRECT(kL, mRECT.T, kL + mWKWidth, mRECT.B);
642 if (keyBounds.Contains(x, y))
643 {
644 k = i;
645 break;
646 }
647 }
648 }
649 }
650
651 return k;
652 }

654 float GetVelocity(float yPos)
655 {
656 float velocity = 0.;
657
658 if (mLastTouchedKey > -1)
659 {
660 float h = mRECT.H();
661
662 if (IsBlackKey(mLastTouchedKey))
663 h *= mBKHeightRatio;
664
665 float fracPos = (yPos - mRECT.T) / (0.95f * h); // 0.95 is to get max velocity around the bottom
666
667 velocity = Clip(fracPos, 1.f / 127.f, 1.f);
668 }
669
670 return velocity;
671 }

673 void GetNoteNameStr(int midiNoteNum, bool addOctave, WDL_String& str)
674 {
675 int oct = midiNoteNum / 12;
676 midiNoteNum -= 12 * oct;
677 const char* notes[12] = {"C", "C#", "D", "D#", "E", "F", "F#", "G", "G#", "A", "A#", "B"};
678 const char* n = notes[midiNoteNum];
679 str.Set(n);
680 if (addOctave)
681 str.AppendFormatted(2, "%d", --oct);
682 }

700 void TriggerMidiMsgFromKeyPress(int key, int velocity)
701 {
702 IMidiMsg msg;
703
704 const int nn = GetMidiNoteNumberForKey(key);
705
706 if (velocity > 0)
707 msg.MakeNoteOnMsg(nn, velocity, 0);
708 else
709 msg.MakeNoteOffMsg(nn, 0);
710
711 GetDelegate()->SendMidiMsgFromUI(msg);
712 }

754 IWheelControl(const IRECT& bounds, IMidiMsg::EControlChangeMsg cc = IMidiMsg::EControlChangeMsg::kNoCC, int initBendRange = 12)
755 : ISliderControlBase(bounds, kNoParameter, EDirection::Vertical, DEFAULT_GEARING, 40.f)
756 , mPitchBendRange(initBendRange)
757 , mCC(cc)
758 {
759 mMenu.AddItem("1 semitone");
760 mMenu.AddItem("2 semitones");
761 mMenu.AddItem("Fifth");
762 mMenu.AddItem("Octave");
763
764 SetValue(cc == IMidiMsg::EControlChangeMsg::kNoCC ? 0.5 : 0.);
765 SetWantsMidi(true);
766 SetActionFunction([cc](IControl* pControl) {
767 IMidiMsg msg;
768 if (cc == IMidiMsg::EControlChangeMsg::kNoCC) // pitchbend
769 msg.MakePitchWheelMsg((pControl->GetValue() * 2.) - 1.);

777 void Draw(IGraphics& g) override
778 {
779 IRECT handleBounds = mRECT.GetPadded(-10.f);
780 const float stepSize = handleBounds.H() / (float)kNumRungs;
781 g.FillRoundRect(DEFAULT_SHCOLOR, mRECT.GetPadded(-5.f));
782
783 if (!g.CheckLayer(mLayer))
784 {
785 const IRECT layerRect = handleBounds.GetMidVPadded(handleBounds.H() + stepSize);
786
787 if (layerRect.W() > 0 && layerRect.H() > 0)
788 {
789 g.StartLayer(this, layerRect);
790 g.DrawGrid(COLOR_BLACK.WithOpacity(0.5f), layerRect, 0.f, stepSize, nullptr, 2.f);
791 mLayer = g.EndLayer();
792 }
793 }
794
795 // NanoVG only has 2 stop gradients
796 IRECT r = handleBounds.FracRectVertical(0.5, true);
797 g.PathRect(r);
798 g.PathFill(IPattern::CreateLinearGradient(r, EDirection::Vertical, {{COLOR_BLACK, 0.f}, {COLOR_MID_GRAY, 1.f}}));
799 r = handleBounds.FracRectVertical(0.51f, false); // slight overlap
800 g.PathRect(r);
801 g.PathFill(IPattern::CreateLinearGradient(r, EDirection::Vertical, {{COLOR_MID_GRAY, 0.f}, {COLOR_BLACK, 1.f}}));
802
803 const float value = static_cast<float>(GetValue());
804 const float y = (handleBounds.H() - (stepSize)) * value;
805 const float triangleRamp = std::fabs(value - 0.5f) * 2.f;
806
807 g.DrawBitmap(mLayer->GetBitmap(), handleBounds, 0, (int)y);
808
809 const IRECT cutoutBounds = handleBounds.GetFromBottom(stepSize).GetTranslated(0, -y);
810 g.PathRect(cutoutBounds);
811 g.PathFill(IPattern::CreateLinearGradient(cutoutBounds, EDirection::Vertical,
812 {// TODO: this can be improved
813 {COLOR_BLACK.WithContrast(iplug::Lerp(0.f, 0.5f, triangleRamp)), 0.f},
814 {COLOR_BLACK.WithContrast(iplug::Lerp(0.5f, 0.f, triangleRamp)), 1.f}}));
815
816 g.DrawVerticalLine(COLOR_BLACK, cutoutBounds, 0.f);
817 g.DrawVerticalLine(COLOR_BLACK, cutoutBounds, 1.f);
818 g.DrawRect(COLOR_BLACK, handleBounds);
819 }

822 {

890 pCaller->OnEndAnimation();

894 kSpringAnimationTime);
