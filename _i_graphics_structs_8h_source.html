339 void GetHSLA(float& h, float& s, float& l, float& a) const
340 {
341 const float fR = R / 255.f;
342 const float fG = G / 255.f;
343 const float fB = B / 255.f;
344 a = A / 255.f;
345
346 const float fMin = std::min(fR, std::min(fG, fB));
347 const float fMax = std::max(fR, std::max(fG, fB));
348 const float fDiff = fMax - fMin;
349 const float fSum = fMax + fMin;
350
351 l = 50.f * fSum;
352
353 if (fMin == fMax)
354 {
355 s = 0.f;
356 h = 0.f;
357 l /= 100.f;
358 return;
359 }
360 else if (l < 50.f)
361 {
362 s = 100.f * fDiff / fSum;
363 }
364 else
365 {
366 s = 100.f * fDiff / (2.f - fDiff);
367 }
368
369 if (fMax == fR)
370 {
371 h = 60.f * (fG - fB) / fDiff;
372 }
373 if (fMax == fG)
374 {
375 h = 60.f * (fB - fR) / fDiff + 120.f;
376 }
377 if (fMax == fB)
378 {
379 h = 60.f * (fR - fG) / fDiff + 240.f;
380 }
381
382 if (h < 0.f)
383 {
384 h = h + 360.f;
385 }
386
387 h /= 360.f;
388 s /= 100.f;
389 l /= 100.f;
390 }

490 static IColor FromHSLA(float h, float s, float l, float a = 1.f)
491 {
492 auto hue = [](float h, float m1, float m2) {
493 if (h < 0)
494 h += 1;
495 if (h > 1)
496 h -= 1;
497 if (h < 1.0f / 6.0f)
498 return m1 + (m2 - m1) * h * 6.0f;
499 else if (h < 3.0f / 6.0f)
500 return m2;
501 else if (h < 4.0f / 6.0f)
502 return m1 + (m2 - m1) * (2.0f / 3.0f - h) * 6.0f;
503 return m1;
504 };
505
506 IColor col;
507 h = std::fmodf(h, 1.0f);
508 if (h < 0.0f)
509 h += 1.0f;
510 s = Clip(s, 0.0f, 1.0f);
511 l = Clip(l, 0.0f, 1.0f);
512 float m2 = l <= 0.5f ? (l * (1 + s)) : (l + s - l * s);
513 float m1 = 2 * l - m2;
514 col.R = static_cast<int>(Clip(hue(h + 1.0f / 3.0f, m1, m2), 0.0f, 1.0f) * 255.f);
515 col.G = static_cast<int>(Clip(hue(h, m1, m2), 0.0f, 1.0f) * 255.f);
516 col.B = static_cast<int>(Clip(hue(h - 1.0f / 3.0f, m1, m2), 0.0f, 1.0f) * 255.f);
517 col.A = static_cast<int>(a * 255.f);
518 return col;
519 }

1958 {
1959 for (int i = 0; i < Size(); i++)
1960 {
1961 for (int j = i + 1; j < Size(); j++)
1962 {
1963 if (Get(i).Contains(Get(j)))
1964 {
1965 mRects.Delete(j);
1966 j--;
1967 }
1968 else if (Get(j).Contains(Get(i)))
1969 {
1970 mRects.Delete(i);
1971 i--;

1975 {

1978 if (Get(i).Mergeable(intersection))
1979 Set(i, Shrink(Get(i), intersection));
1980 else if (Get(j).Mergeable(intersection))
1981 Set(j, Shrink(Get(j), intersection));
1982 else if (Get(i).Area() < Get(j).Area())
1983 Set(i, Split(Get(i), intersection));
1984 else

1991 {
1992 for (int j = i + 1; j < Size(); j++)
1993 {
1994 if (Get(i).Mergeable(Get(j)))
1995 {
