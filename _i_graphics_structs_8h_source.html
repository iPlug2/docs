340 void GetHSLA(float& h, float& s, float& l, float& a) const
341 {
342 const float fR = R / 255.f;
343 const float fG = G / 255.f;
344 const float fB = B / 255.f;
345 a = A / 255.f;
346
347 const float fMin = std::min(fR, std::min(fG, fB));
348 const float fMax = std::max(fR, std::max(fG, fB));
349 const float fDiff = fMax - fMin;
350 const float fSum = fMax + fMin;
351
352 l = 50.f * fSum;
353
354 if (fMin == fMax)
355 {
356 s = 0.f;
357 h = 0.f;
358 l /= 100.f;
359 return;
360 }
361 else if (l < 50.f)
362 {
363 s = 100.f * fDiff / fSum;
364 }
365 else
366 {
367 s = 100.f * fDiff / (2.f - fDiff);
368 }
369
370 if (fMax == fR)
371 {
372 h = 60.f * (fG - fB) / fDiff;
373 }
374 if (fMax == fG)
375 {
376 h = 60.f * (fB - fR) / fDiff + 120.f;
377 }
378 if (fMax == fB)
379 {
380 h = 60.f * (fR - fG) / fDiff + 240.f;
381 }
382
383 if (h < 0.f)
384 {
385 h = h + 360.f;
386 }
387
388 h /= 360.f;
389 s /= 100.f;
390 l /= 100.f;
391 }

492 static IColor FromHSLA(float h, float s, float l, float a = 1.f)
493 {
494 auto hue = [](float h, float m1, float m2) {
495 if (h < 0)
496 h += 1;
497 if (h > 1)
498 h -= 1;
499 if (h < 1.0f / 6.0f)
500 return m1 + (m2 - m1) * h * 6.0f;
501 else if (h < 3.0f / 6.0f)
502 return m2;
503 else if (h < 4.0f / 6.0f)
504 return m1 + (m2 - m1) * (2.0f / 3.0f - h) * 6.0f;
505 return m1;
506 };
507
508 IColor col;
509 h = std::fmodf(h, 1.0f);
510 if (h < 0.0f)
511 h += 1.0f;
512 s = Clip(s, 0.0f, 1.0f);
513 l = Clip(l, 0.0f, 1.0f);
514 float m2 = l <= 0.5f ? (l * (1 + s)) : (l + s - l * s);
515 float m1 = 2 * l - m2;
516 col.R = static_cast<int>(Clip(hue(h + 1.0f / 3.0f, m1, m2), 0.0f, 1.0f) * 255.f);
517 col.G = static_cast<int>(Clip(hue(h, m1, m2), 0.0f, 1.0f) * 255.f);
518 col.B = static_cast<int>(Clip(hue(h - 1.0f / 3.0f, m1, m2), 0.0f, 1.0f) * 255.f);
519 col.A = static_cast<int>(a * 255.f);
520 return col;
521 }
