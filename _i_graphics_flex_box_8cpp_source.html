41YGNodeRef IFlexBox::AddItem(float width, float height, YGAlign alignSelf, float grow, float shrink, float margin)
42{
43 int index = mNodeCounter;
44 YGNodeRef child = YGNodeNew();
45
46 if (width == YGUndefined)
47 YGNodeStyleSetWidthAuto(child);
48 else if (width < 0.f)
49 YGNodeStyleSetWidthPercent(child, width * -1.f);
50 else
51 YGNodeStyleSetWidth(child, width);
52
53 if (height == YGUndefined)
54 YGNodeStyleSetHeightAuto(child);
55 else if (height < 0.f)
56 YGNodeStyleSetHeightPercent(child, height * -1.f);
57 else
58 YGNodeStyleSetHeight(child, height);
59
60 YGNodeStyleSetAlignSelf(child, alignSelf);
61 YGNodeStyleSetMargin(child, YGEdgeAll, margin);
62 YGNodeStyleSetFlexGrow(child, grow);
63 YGNodeStyleSetFlexShrink(child, shrink);
64 YGNodeInsertChild(mRootNodeRef, child, index);
65
66 mNodeCounter++;
67
68 return child;
69}
