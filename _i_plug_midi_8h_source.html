538class IMidiQueue
539{
540public:
541 IMidiQueue(int size = DEFAULT_BLOCK_SIZE)
542 : mBuf(NULL)
543 , mSize(0)
544 , mGrow(Granulize(size))
545 , mFront(0)
546 , mBack(0)
547 {
548 Expand();
549 }
550
551 ~IMidiQueue() { free(mBuf); }
552
553 // Adds a MIDI message at the back of the queue. If the queue is full,
554 // it will automatically expand itself.
555 void Add(const IMidiMsg& msg)
556 {
557 if (mBack >= mSize)
558 {
559 if (mFront > 0)
560 Compact();
561 else if (!Expand())
562 return;
563 }
564
565#ifndef DONT_SORT_IMIDIQUEUE
566 // Insert the MIDI message at the right offset.
567 if (mBack > mFront && msg.mOffset < mBuf[mBack - 1].mOffset)
568 {
569 int i = mBack - 2;
570 while (i >= mFront && msg.mOffset < mBuf[i].mOffset)
571 --i;
572 i++;
573 memmove(&mBuf[i + 1], &mBuf[i], (mBack - i) * sizeof(IMidiMsg));
574 mBuf[i] = msg;
575 }
576 else
577#endif
578 mBuf[mBack] = msg;
579 ++mBack;
580 }
581
582 // Removes a MIDI message from the front of the queue (but does *not*
583 // free up its space until Compact() is called).
584 inline void Remove() { ++mFront; }
585
586 // Returns true if the queue is empty.
587 inline bool Empty() const { return mFront == mBack; }
588
589 // Returns the number of MIDI messages in the queue.
590 inline int ToDo() const { return mBack - mFront; }
591
592 // Returns the number of MIDI messages for which memory has already been
593 // allocated.
594 inline int GetSize() const { return mSize; }
595
596 // Returns the "next" MIDI message (all the way in the front of the
597 // queue), but does *not* remove it from the queue.
598 inline IMidiMsg& Peek() const { return mBuf[mFront]; }
599
600 // Moves back MIDI messages all the way to the front of the queue, thus
601 // freeing up space at the back, and updates the sample offset of the
602 // remaining MIDI messages by substracting nFrames.
603 inline void Flush(int nFrames)
604 {
605 // Move everything all the way to the front.
606 if (mFront > 0)
607 Compact();
608
609 // Update the sample offset.
610 for (int i = 0; i < mBack; ++i)
611 mBuf[i].mOffset -= nFrames;
612 }
613
614 // Clears the queue.
615 inline void Clear() { mFront = mBack = 0; }
616
617 // Resizes (grows or shrinks) the queue, returns the new size.
618 int Resize(int size)
619 {
620 if (mFront > 0)
621 Compact();
622 mGrow = size = Granulize(size);
623 // Don't shrink below the number of currently queued MIDI messages.
624 if (size < mBack)
625 size = Granulize(mBack);
626 if (size == mSize)
627 return mSize;
628
629 void* buf = realloc(mBuf, size * sizeof(IMidiMsg));
630 if (!buf)
631 return mSize;
632
633 mBuf = (IMidiMsg*)buf;
634 mSize = size;
635 return size;
636 }
637
638protected:
639 // Automatically expands the queue.
640 bool Expand()
641 {
642 if (!mGrow)
643 return false;
644 int size = (mSize / mGrow + 1) * mGrow;
645
646 void* buf = realloc(mBuf, size * sizeof(IMidiMsg));
647 if (!buf)
648 return false;
649
650 mBuf = (IMidiMsg*)buf;
651 mSize = size;
652 return true;
653 }
654
655 // Moves everything all the way to the front.
656 inline void Compact()
657 {
658 mBack -= mFront;
659 if (mBack > 0)
660 memmove(&mBuf[0], &mBuf[mFront], mBack * sizeof(IMidiMsg));
661 mFront = 0;
662 }
663
664 // Rounds the MIDI queue size up to the next 4 kB memory page size.
665 inline int Granulize(int size) const
666 {
667 int bytes = size * sizeof(IMidiMsg);
668 int rest = bytes % 4096;
669 if (rest)
670 size = (bytes - rest + 4096) / sizeof(IMidiMsg);
671 return size;
672 }
673
674 IMidiMsg* mBuf;
675
676 int mSize, mGrow;
677 int mFront, mBack;
678};
