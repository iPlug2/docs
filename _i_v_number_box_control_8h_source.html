26class IVNumberBoxControl : public IControl, public IVectorBase
27{
28public:
29 IVNumberBoxControl(const IRECT& bounds,
30 int paramIdx = kNoParameter,
31 IActionFunction actionFunc = nullptr,
32 const char* label = "",
33 const IVStyle& style = DEFAULT_STYLE,
34 double defaultValue = 50.f,
35 double minValue = 1.f,
36 double maxValue = 100.f,
37 const char* fmtStr = "%0.0f")
38 : IControl(bounds, paramIdx, actionFunc)
39 , IVectorBase(style.WithDrawShadows(false).WithValueText(style.valueText.WithVAlign(EVAlign::Middle)))
40 , mFmtStr(fmtStr)
41 , mMinValue(minValue)
42 , mMaxValue(maxValue)
43 , mRealValue(defaultValue)
44 {
45 assert(defaultValue >= minValue && defaultValue <= maxValue);
46
47 AttachIControl(this, label);
48 }
49
50 void OnInit() override
51 {
52 if (GetParam())
53 {
54 mMinValue = GetParam()->GetMin();
55 mMaxValue = GetParam()->GetMax();
56 mRealValue = GetParam()->GetDefault();
57 }
58 }
59
60 void Draw(IGraphics& g) override
61 {
62 DrawLabel(g);
63
64 if (mMouseIsOver)
65 g.FillRect(GetColor(kHL), mTextReadout->GetRECT());
66 }
67
68 void OnResize() override
69 {
70 MakeRects(mRECT, false);
71
72 if (mIncButton && mDecButton)
73 {
74 IRECT sections = mWidgetBounds;
75 mTextReadout->SetTargetAndDrawRECTs(sections.ReduceFromLeft(sections.W() * 0.75f));
76 sections.Pad(-1.f, 1.f, 0.f, 1.f);
77 mIncButton->SetTargetAndDrawRECTs(sections.FracRectVertical(0.5f, true));
78 mDecButton->SetTargetAndDrawRECTs(sections.FracRectVertical(0.5f, false));
79 SetTargetRECT(mTextReadout->GetRECT());
80 }
81 }
82
83 void OnAttached() override
84 {
85 IRECT sections = mWidgetBounds;
86 GetUI()->AttachControl(mTextReadout = new IVLabelControl(
87 sections.ReduceFromLeft(sections.W() * 0.75f), "0", mStyle.WithDrawFrame(true)));
88
89 mTextReadout->SetStrFmt(32, mFmtStr.Get(), mRealValue);
90
91 sections.Pad(-1.f, 1.f, 0.f, 1.f);
92 GetUI()
93 ->AttachControl(mIncButton =
94 new IVButtonControl(sections.FracRectVertical(0.5f, true), SplashClickActionFunc, "+", mStyle))
95 ->SetAnimationEndActionFunction(mIncrementFunc);
96 GetUI()
97 ->AttachControl(mDecButton =
98 new IVButtonControl(sections.FracRectVertical(0.5f, false), SplashClickActionFunc, "-", mStyle))
99 ->SetAnimationEndActionFunction(mDecrementFunc);
100 }
101
102 void OnMouseDown(float x, float y, const IMouseMod& mod) override
103 {
104 if (mHideCursorOnDrag)
105 GetUI()->HideMouseCursor(true, true);
106
107 if (GetParam())
108 GetDelegate()->BeginInformHostOfParamChangeFromUI(GetParamIdx());
109 }
110
111 void OnMouseUp(float x, float y, const IMouseMod& mod) override
112 {
113 if (mHideCursorOnDrag)
114 GetUI()->HideMouseCursor(false);
115
116 if (GetParam())
117 GetDelegate()->EndInformHostOfParamChangeFromUI(GetParamIdx());
118 }
119
120 void OnMouseDrag(float x, float y, float dX, float dY, const IMouseMod& mod) override
121 {
122 double gearing = IsFineControl(mod, true) ? mSmallIncrement : mLargeIncrement;
123 mRealValue -= (double(dY) * gearing);
124 OnValueChanged();
125 }
126
127 void OnMouseDblClick(float x, float y, const IMouseMod& mod) override
128 {
129 if (mTextReadout->GetRECT().Contains(x, y))
130 GetUI()->CreateTextEntry(*this, mText, mTextReadout->GetRECT(), mTextReadout->GetStr());
131 }
132
133 void OnTextEntryCompletion(const char* str, int valIdx) override
134 {
135 mRealValue = atof(str);
136 OnValueChanged();
137 }
138
139 void OnMouseWheel(float x, float y, const IMouseMod& mod, float d) override
140 {
141 double gearing = IsFineControl(mod, true) ? mSmallIncrement : mLargeIncrement;
142 double inc = (d > 0.f ? 1. : -1.) * gearing;
143 mRealValue += inc;
144 OnValueChanged();
145 }
146
147 void SetValueFromDelegate(double value, int valIdx = 0) override
148 {
149 if (GetParam())
150 {
151 mRealValue = GetParam()->FromNormalized(value);
152 OnValueChanged(true);
153 }
154
155 IControl::SetValueFromDelegate(value, valIdx);
156 }
157
158 void SetValueFromUserInput(double value, int valIdx = 0) override
159 {
160 if (GetParam())
161 {
162 mRealValue = GetParam()->FromNormalized(value);
163 OnValueChanged(true);
164 }
165
166 IControl::SetValueFromUserInput(value, valIdx);
167 }
168
169 void SetStyle(const IVStyle& style) override
170 {
171 IVectorBase::SetStyle(style);
172 mTextReadout->SetStyle(style);
173 mIncButton->SetStyle(style);
174 mDecButton->SetStyle(style);
175 }
176
177 bool IsFineControl(const IMouseMod& mod, bool wheel) const
178 {
179#ifdef PROTOOLS
180 #ifdef OS_WIN
181 return mod.C;
182 #else
183 return wheel ? mod.C : mod.R;
184 #endif
185#else
186 return (mod.C || mod.S);
187#endif
188 }
189
190 void OnValueChanged(bool preventAction = false)
191 {
192 mRealValue = Clip(mRealValue, mMinValue, mMaxValue);
193
194 mTextReadout->SetStrFmt(32, mFmtStr.Get(), mRealValue);
195
196 if (!preventAction && GetParam())
197 SetValue(GetParam()->ToNormalized(mRealValue));
198
199 SetDirty(!preventAction);
200 }
201
202 double GetRealValue() const { return mRealValue; }
203
204protected:
205 IActionFunction mIncrementFunc = [this](IControl* pCaller) {
206 mRealValue += mLargeIncrement;
207 OnValueChanged();
208 };
209 IActionFunction mDecrementFunc = [this](IControl* pCaller) {
210 mRealValue -= mLargeIncrement;
211 OnValueChanged();
212 };
213 IVLabelControl* mTextReadout = nullptr;
214 IVButtonControl* mIncButton = nullptr;
215 IVButtonControl* mDecButton = nullptr;
216 WDL_String mFmtStr;
217 double mLargeIncrement = 1.f;
218 double mSmallIncrement = 0.1f;
219 double mMinValue;
220 double mMaxValue;
221 double mRealValue = 0.f;
222 bool mHideCursorOnDrag = true;
223};
