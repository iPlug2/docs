375 tresult PLUGIN_API IPlugVST3Controller::notify(IMessage* message)
376 {
377  if (!message)
378  return kInvalidArgument;
379 
380  if (!strcmp (message->getMessageID(), "SCVFD"))
381  {
382  Steinberg::int64 controlTag = kNoTag;
383  double normalizedValue = 0.;
384 
385  if(message->getAttributes()->getInt("CT", controlTag) == kResultFalse)
386  return kResultFalse;
387 
388  if(message->getAttributes()->getFloat("NV", normalizedValue) == kResultFalse)
389  return kResultFalse;
390 
391  SendControlValueFromDelegate((int) controlTag, normalizedValue);
392 
393  }
394  else if (!strcmp (message->getMessageID(), "SCMFD"))
395  {
396  const void* data;
397  Steinberg::int64 controlTag = kNoTag;
398  Steinberg::int64 messageTag = kNoTag;
399 
400  if(message->getAttributes()->getInt("CT", controlTag) == kResultFalse)
401  return kResultFalse;
402 
403  if(message->getAttributes()->getInt("MT", messageTag) == kResultFalse)
404  return kResultFalse;
405 
406  Steinberg::uint32 size;
407 
408  if (message->getAttributes()->getBinary("D", data, size) == kResultOk)
409  {
410  SendControlMsgFromDelegate((int) controlTag, (int) messageTag, size, data);
411  return kResultOk;
412  }
413  }
414  else if (!strcmp (message->getMessageID(), "SMMFD"))
415  {
416  const void* data = nullptr;
417  uint32 size;
418 
419  if (message->getAttributes()->getBinary("D", data, size) == kResultOk)
420  {
421  if (size == sizeof(IMidiMsg))
422  {
423  IMidiMsg msg;
424  memcpy(&msg, data, size);
425  SendMidiMsgFromDelegate(msg);
426  }
427  }
428  }
429  else if (!strcmp (message->getMessageID(), "SSMFD"))
430  {
431  const void* data = nullptr;
432  uint32 size;
433  int64 offset;
434 
435  if (message->getAttributes()->getBinary("D", data, size) == kResultOk)
436  {
437  if (message->getAttributes()->getInt("O", offset) == kResultOk)
438  {
439  ISysEx msg {static_cast<int>(offset), static_cast<const uint8_t*>(data), static_cast<int>(size)};
440  SendSysexMsgFromDelegate(msg);
441  }
442  }
443  }
444 
445  return ComponentBase::notify(message);
446 }
