48bool IPlugAPP::EditorResize(int viewWidth, int viewHeight)
49{
50 bool parentResized = false;
51
52 if (viewWidth != GetEditorWidth() || viewHeight != GetEditorHeight())
53 {
54 #ifdef OS_MAC
55 RECT rcClient, rcWindow;
56 POINT ptDiff;
57
58 GetClientRect(gHWND, &rcClient);
59 GetWindowRect(gHWND, &rcWindow);
60
61 ptDiff.x = (rcWindow.right - rcWindow.left) - rcClient.right;
62 ptDiff.y = (rcWindow.bottom - rcWindow.top) - rcClient.bottom;
63
64 SetWindowPos(gHWND, 0, rcWindow.left, rcWindow.bottom - viewHeight - ptDiff.y, viewWidth + ptDiff.x, viewHeight + ptDiff.y, 0);
65 parentResized = true;
66 #endif
67
68 SetEditorSize(viewWidth, viewHeight);
69 }
70
71 return parentResized;
72}

122void IPlugAPP::AppProcess(double** inputs, double** outputs, int nFrames)
123{
124 SetChannelConnections(ERoute::kInput, 0, MaxNChannels(ERoute::kInput), !IsInstrument()); //TODO: go elsewhere - enable inputs
125 SetChannelConnections(ERoute::kOutput, 0, MaxNChannels(ERoute::kOutput), true); //TODO: go elsewhere
126 AttachBuffers(ERoute::kInput, 0, NChannelsConnected(ERoute::kInput), inputs, GetBlockSize());
127 AttachBuffers(ERoute::kOutput, 0, NChannelsConnected(ERoute::kOutput), outputs, GetBlockSize());
128
129 if(mMidiMsgsFromCallback.ElementsAvailable())
130 {
131 IMidiMsg msg;
132
133 while (mMidiMsgsFromCallback.Pop(msg))
134 {
135 ProcessMidiMsg(msg);
136 mMidiMsgsFromProcessor.Push(msg); // queue incoming MIDI for UI
137 }
138 }
139
140 if(mSysExMsgsFromCallback.ElementsAvailable())
141 {
142 SysExData data;
143
144 while (mSysExMsgsFromCallback.Pop(data))
145 {
146 ISysEx msg { data.mOffset, data.mData, data.mSize };
147 ProcessSysEx(msg);
148 mSysExDataFromProcessor.Push(data); // queue incoming Sysex for UI
149 }
150 }
151
152 if(mMidiMsgsFromEditor.ElementsAvailable())
153 {
154 IMidiMsg msg;
155
156 while (mMidiMsgsFromEditor.Pop(msg))
157 {
158 ProcessMidiMsg(msg);
159 }
160 }
161
162 //Do not handle Sysex messages here - SendSysexMsgFromUI overridden
163
164 ENTER_PARAMS_MUTEX
165 ProcessBuffers(0.0, GetBlockSize());
166 LEAVE_PARAMS_MUTEX
167}
