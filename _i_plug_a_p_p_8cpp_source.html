103  AttachBuffers(ERoute::kInput, 0, NChannelsConnected(ERoute::kInput), inputs, GetBlockSize());
104  AttachBuffers(ERoute::kOutput, 0, NChannelsConnected(ERoute::kOutput), outputs, GetBlockSize());

106  if(mMidiMsgsFromCallback.ElementsAvailable())

128

134  {
135  ProcessMidiMsg(msg);
136  }
137  }

139  //Do not handle Sysex messages here - SendSysexMsgFromUI overridden
