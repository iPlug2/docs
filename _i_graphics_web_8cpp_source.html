209 EM_BOOL key_callback(int eventType, const EmscriptenKeyboardEvent* pEvent, void* pUserData)
210 {
211  IGraphicsWeb* pGraphicsWeb = (IGraphicsWeb*) pUserData;
212 
213  switch (eventType)
214  {
215  case EMSCRIPTEN_EVENT_KEYDOWN:
216  {
217  IKeyPress keyPress {*pEvent->key, domVKToWinVK(pEvent->keyCode), static_cast<bool>(pEvent->shiftKey),
218  static_cast<bool>(pEvent->ctrlKey),
219  static_cast<bool>(pEvent->altKey)};
220 
221  pGraphicsWeb->OnKeyDown(pGraphicsWeb->mPrevX, pGraphicsWeb->mPrevY, keyPress);
222  break;
223  }
224  default:
225  break;
226  }
227 
228  return 0;
229 }

231 EM_BOOL outside_mouse_callback(int eventType, const EmscriptenMouseEvent* pEvent, void* pUserData)
232 {
233  IGraphicsWeb* pGraphics = (IGraphicsWeb*) pUserData;
234 
235  IMouseMod modifiers(0, 0, pEvent->shiftKey, pEvent->ctrlKey, pEvent->altKey);
236 
237  double x = pEvent->targetX;
238  double y = pEvent->targetY;
239 
240  val rect = GetCanvas().call<val>("getBoundingClientRect");
241  x -= rect["left"].as<double>();
242  y -= rect["top"].as<double>();
243 
244  x /= pGraphics->GetDrawScale();
245  y /= pGraphics->GetDrawScale();
246 
247  switch (eventType)
248  {
249  case EMSCRIPTEN_EVENT_MOUSEUP: pGraphics->OnMouseUp(x, y, modifiers);
250  pGraphics->OnMouseUp(x, y, modifiers); break;
251  emscripten_set_mousemove_callback("#window", pGraphics, 1, nullptr);
252  emscripten_set_mouseup_callback("#window", pGraphics, 1, nullptr);
253  break;
254  case EMSCRIPTEN_EVENT_MOUSEMOVE:
255  if(pEvent->buttons != 0)
256  pGraphics->OnMouseDrag(x, y, pEvent->movementX, pEvent->movementY, modifiers);
257  break;
258  default:
259  break;
260  }
261 
262  pGraphics->mPrevX = x;
263  pGraphics->mPrevY = y;
264 
265  return true;
266 }

268 EM_BOOL mouse_callback(int eventType, const EmscriptenMouseEvent* pEvent, void* pUserData)
269 {
270  IGraphicsWeb* pGraphics = (IGraphicsWeb*) pUserData;
271 
272  IMouseMod modifiers(pEvent->buttons == 1, pEvent->buttons == 2, pEvent->shiftKey, pEvent->ctrlKey, pEvent->altKey);
273 
274  double x = pEvent->targetX;
275  double y = pEvent->targetY;
276 
277  x /= pGraphics->GetDrawScale();
278  y /= pGraphics->GetDrawScale();
279 
280  switch (eventType)
281  {
282  case EMSCRIPTEN_EVENT_CLICK: break;
283  case EMSCRIPTEN_EVENT_MOUSEDOWN: pGraphics->OnMouseDown(x, y, modifiers); break;
284  case EMSCRIPTEN_EVENT_MOUSEUP: pGraphics->OnMouseUp(x, y, modifiers); break;
285  case EMSCRIPTEN_EVENT_DBLCLICK: pGraphics->OnMouseDblClick(x, y, modifiers);break;
286  case EMSCRIPTEN_EVENT_MOUSEMOVE:
287  if(pEvent->buttons == 0)
288  pGraphics->OnMouseOver(x, y, modifiers);
289  else
290  pGraphics->OnMouseDrag(x, y, pEvent->movementX, pEvent->movementY, modifiers);
291  break;
292  case EMSCRIPTEN_EVENT_MOUSEENTER:
293  pGraphics->OnSetCursor();
294  pGraphics->OnMouseOver(x, y, modifiers);
295  emscripten_set_mousemove_callback("#window", pGraphics, 1, nullptr);
296  break;
297  case EMSCRIPTEN_EVENT_MOUSELEAVE:
298  if(pEvent->buttons != 0)
299  {
300  emscripten_set_mousemove_callback("#window", pGraphics, 1, outside_mouse_callback);
301  emscripten_set_mouseup_callback("#window", pGraphics, 1, outside_mouse_callback);
302  }
303  pGraphics->OnMouseOut(); break;
304  default:
305  break;
306  }
307 
308  pGraphics->mPrevX = x;
309  pGraphics->mPrevY = y;
310 
311  return true;
312 }
