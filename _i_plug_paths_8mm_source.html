144bool GetResourcePathFromBundle(const char* fileName, const char* searchExt, WDL_String& fullPath, const char* bundleID)
145{
146 @autoreleasepool
147 {
148 const char* ext = fileName + strlen(fileName) - 1;
149 while (ext >= fileName && *ext != '.')
150 --ext;
151 ++ext;
152
153 bool isCorrectType = !strcasecmp(ext, searchExt);
154
155 bool isAppExtension = IsOOPAuv3AppExtension();
156
157 NSBundle* pBundle = [NSBundle bundleWithIdentifier:[NSString stringWithCString:bundleID
158 encoding:NSUTF8StringEncoding]];
159
160 NSString* pFile = [[NSString stringWithCString:fileName
161 encoding:NSUTF8StringEncoding] stringByDeletingPathExtension];
162 NSString* pExt = [NSString stringWithCString:searchExt encoding:NSUTF8StringEncoding];
163
164 if (isCorrectType && pBundle && pFile)
165 {
166 NSString* pBasePath;
167
168 if (isAppExtension)
169 pBasePath = [[[[pBundle bundlePath] stringByDeletingLastPathComponent] stringByDeletingLastPathComponent]
170 stringByAppendingString:@"/Resources/"];
171 else
172 pBasePath = [[pBundle resourcePath] stringByAppendingString:@"/"];
173
174 NSString* pPath =
175 [[[pBasePath stringByAppendingString:pFile] stringByAppendingString:@"."] stringByAppendingString:pExt];
176
177 if (pPath && [[NSFileManager defaultManager] fileExistsAtPath:pPath] == YES)
178 {
179 fullPath.Set([pPath cStringUsingEncoding:NSUTF8StringEncoding]);
180 return true;
181 }
182 }
183
184 fullPath.Set("");
185 return false;
186 }
187}

189bool GetResourcePathFromSharedLocation(const char* fileName, const char* searchExt, WDL_String& fullPath,
190 const char* subfolder)
191{
192 @autoreleasepool
193 {
194 const char* ext = fileName + strlen(fileName) - 1;
195 while (ext >= fileName && *ext != '.')
196 --ext;
197 ++ext;
198
199 bool isCorrectType = !strcasecmp(ext, searchExt);
200
201 NSString* pExt = [NSString stringWithCString:searchExt encoding:NSUTF8StringEncoding];
202 NSString* pFile = [[[NSString stringWithCString:fileName encoding:NSUTF8StringEncoding] lastPathComponent]
203 stringByDeletingPathExtension];
204
205 if (isCorrectType && pFile)
206 {
207 WDL_String musicPath;
208 SharedMusicPath(musicPath);
209
210 if (subfolder)
211 {
212 NSString* pPluginName = [NSString stringWithCString:subfolder encoding:NSUTF8StringEncoding];
213 NSString* pMusicLocation = [NSString stringWithCString:musicPath.Get() encoding:NSUTF8StringEncoding];
214 NSString* pPath =
215 [[[[pMusicLocation stringByAppendingPathComponent:pPluginName] stringByAppendingPathComponent:@"Resources"]
216 stringByAppendingPathComponent:pFile] stringByAppendingPathExtension:pExt];
217
218 if ([[NSFileManager defaultManager] fileExistsAtPath:pPath] == YES)
219 {
220 fullPath.Set([pPath cString]);
221 return true;
222 }
223 }
224 }
225
226 fullPath.Set("");
227 return false;
228 }
229}

231EResourceLocation LocateResource(const char* name, const char* type, WDL_String& result, const char* bundleID, void*,
232 const char* sharedResourcesSubPath)
233{
234 if (CStringHasContents(name))
235 {
236 #ifndef AUv3_API
237 // first check this bundle
238 if (GetResourcePathFromBundle(name, type, result, bundleID))
239 return EResourceLocation::kAbsolutePath;
240 #endif
241
242 // then check ~/Music/sharedResourcesSubPath, which is a shared folder that can be accessed from app sandbox
243 if (GetResourcePathFromSharedLocation(name, type, result, sharedResourcesSubPath))
244 return EResourceLocation::kAbsolutePath;
245
246 // finally check name, which might be a full path - if the plug-in is trying to load a resource at runtime (e.g.
247 // skin-able UI)
248 NSString* pPath = [NSString stringWithCString:name encoding:NSUTF8StringEncoding];
249
250 if ([[NSFileManager defaultManager] fileExistsAtPath:pPath] == YES)
251 {
252 result.Set([pPath UTF8String]);
253 return EResourceLocation::kAbsolutePath;
254 }
255 }
256 return EResourceLocation::kNotFound;
257}

373EResourceLocation LocateResource(const char* name, const char* type, WDL_String& result, const char* bundleID, void*,
374 const char*)
375{
376 if (CStringHasContents(name))
377 {
378 #ifdef IGRAPHICS_METAL
379 auto itr = gTextureMap.find(name);
380
381 if (itr != gTextureMap.end())
382 {
383 result.Set(name);
384 return EResourceLocation::kPreloadedTexture;
385 }
386 #endif
387
388 if (GetResourcePathFromBundle(name, type, result, bundleID))
389 return EResourceLocation::kAbsolutePath;
390 }
391
392 return EResourceLocation::kNotFound;
393}
