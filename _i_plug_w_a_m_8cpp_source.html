75void IPlugWAM::OnEditorIdleTick()
76{
77 while (mParamChangeFromProcessor.ElementsAvailable())
78 {
79 ParamTuple p;
80 mParamChangeFromProcessor.Pop(p);
81 SendParameterValueFromDelegate(p.idx, p.value, false);
82 }
83
84 while (mMidiMsgsFromProcessor.ElementsAvailable())
85 {
86 IMidiMsg msg;
87 mMidiMsgsFromProcessor.Pop(msg);
88 SendMidiMsgFromDelegate(msg);
89 }
90
91 OnIdle();
92}

94// WAM onMessageN
95void IPlugWAM::onMessage(char* verb, char* res, double data)
96{
97 if (strcmp(verb, "TICK") == 0) // special case for DSP OnIdle()
98 {
99 OnEditorIdleTick();
100 }
101 else if (strcmp(verb, "SMMFUI") == 0)
102 {
103 uint8_t data[3];
104 char* pChar = strtok(res, ":");
105 int i = 0;
106 while (pChar != nullptr)
107 {
108 data[i++] = atoi(pChar);
109 pChar = strtok(nullptr, ":");
110 }
111
112 IMidiMsg msg = {0, data[0], data[1], data[2]};
113 ProcessMidiMsg(msg); // TODO: should queue to mMidiMsgsFromEditor?
114 }
115 else if (strcmp(verb, "SAMFUI") == 0) // SAMFUI
116 {
117 int data[2] = {-1, -1};
118 char* pChar = strtok(res, ":");
119 int i = 0;
120 while (pChar != nullptr)
121 {
122 data[i++] = atoi(pChar);
123 pChar = strtok(nullptr, ":");
124 }
125
126 OnMessage(data[0], data[1], sizeof(double), reinterpret_cast<void*>(&data));
127 }
128}
