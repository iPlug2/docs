56 void IPlugWAM::onProcess(WAM::AudioBus* pAudio, void* pData)
57 {
58  const int blockSize = GetBlockSize();
59 
60  SetChannelConnections(ERoute::kInput, 0, MaxNChannels(ERoute::kInput), !IsInstrument()); //TODO: go elsewhere
61  SetChannelConnections(ERoute::kOutput, 0, MaxNChannels(ERoute::kOutput), true); //TODO: go elsewhere
62  AttachBuffers(ERoute::kInput, 0, NChannelsConnected(ERoute::kInput), pAudio->inputs, blockSize);
63  AttachBuffers(ERoute::kOutput, 0, NChannelsConnected(ERoute::kOutput), pAudio->outputs, blockSize);
64  ProcessBuffers((float) 0.0f, blockSize);
65 
66  //emulate IPlugAPIBase::OnTimer - should be called on the main thread - how to do that in audio worklet processor?
67  if(mBlockCounter == 0)
68  {
69  while(mParamChangeFromProcessor.ElementsAvailable())
70  {
71  IParamChange p;
72  mParamChangeFromProcessor.Pop(p);
73  SendParameterValueFromDelegate(p.paramIdx, p.value, p.normalized);
74  }
75 
76  while (mMidiMsgsFromProcessor.ElementsAvailable())
77  {
78  IMidiMsg msg;
79  mMidiMsgsFromProcessor.Pop(msg);
80  SendMidiMsgFromDelegate(msg);
81  }
82 
83  OnIdle();
84 
85  mBlockCounter = 8; // 8 * 128 samples = 23ms @ 44100 sr
86  }
87 
88  mBlockCounter--;
89 }
