952 void DrawPressableEllipse(IGraphics& g, const IRECT& bounds, bool pressed, bool mouseOver, bool disabled)
953 {
954 IRECT handleBounds = bounds;
955 IRECT centreBounds = bounds.GetPadded(-mStyle.shadowOffset);
956 IRECT shadowBounds = bounds.GetTranslated(mStyle.shadowOffset, mStyle.shadowOffset);
957 const IBlend blend = mControl->GetBlend();
958 const float contrast = disabled ? -GRAYED_ALPHA : 0.f;
959
960 if (!pressed && !disabled && mStyle.drawShadows)
961 g.FillEllipse(GetColor(kSH), shadowBounds);
962
963 if (pressed)
964 {
965 if (mStyle.emboss)
966 {
967 shadowBounds.ReduceFromRight(mStyle.shadowOffset);
968 shadowBounds.ReduceFromBottom(mStyle.shadowOffset);
969 // Fill background with pressed color and shade it
970 g.FillEllipse(GetColor(kPR), bounds, &blend);
971 g.FillEllipse(GetColor(kSH), bounds, &blend);
972
973 // Inverse shading for recessed look - shadowBounds = inner shadow
974 g.FillEllipse(GetColor(kFG).WithContrast(contrast), shadowBounds /*, &blend*/);
975
976 // Fill in center with pressed color
977 g.FillEllipse(GetColor(kPR).WithContrast(contrast), centreBounds /*, &blend*/);
978 }
979 else
980 g.FillEllipse(GetColor(kPR).WithContrast(contrast), handleBounds /*, &blend*/);
981 }
982 else
983 {
984 // Embossed style unpressed
985 if (mStyle.emboss)
986 {
987 // Positive light TODO: use thes kPR color for now, maybe change the name?
988 g.FillEllipse(GetColor(kPR).WithContrast(contrast), bounds /*, &blend*/);
989
990 // Negative light TODO: clip this?
991 g.FillEllipse(GetColor(kSH).WithContrast(contrast), shadowBounds /*, &blend*/);
992
993 // Fill in foreground
994 g.FillEllipse(GetColor(kFG).WithContrast(contrast), centreBounds /*, &blend*/);
995
996 // Shade when hovered
997 if (mouseOver)
998 g.FillEllipse(GetColor(kHL), centreBounds, &blend);
999 }
1000 else
1001 {
1002 g.FillEllipse(GetColor(kFG).WithContrast(contrast), handleBounds /*, &blend*/);
1003
1004 // Shade when hovered
1005 if (mouseOver)
1006 g.FillEllipse(GetColor(kHL), handleBounds, &blend);
1007 }
1008 }
1009
1010 if (pressed && mControl->GetAnimationFunction())
1011 DrawSplash(g, handleBounds);
1012
1013 if (mStyle.drawFrame)
