646  {
647  g.FillRect(GetColor(kBG), bounds);
648 
649  IRECT handleBounds = GetAdjustedHandleBounds(bounds);
650  const float cornerRadius = mRoundness * (handleBounds.W() / 2.f);
651 
652  if (pressed)
653  {
654  g.FillRoundRect(GetColor(kPR), handleBounds, cornerRadius);
655 
656  //inner shadow
657  if (mDrawShadows && mEmboss)
658  {
659  g.PathRect(handleBounds.GetHSliced(mShadowOffset));
660  g.PathRect(handleBounds.GetVSliced(mShadowOffset));
661  g.PathFill(GetColor(kSH));
662  }
663  }
664  else
665  {
666  //outer shadow
667  if (mDrawShadows && !mEmboss)
668  g.FillRoundRect(GetColor(kSH), handleBounds.GetTranslated(mShadowOffset, mShadowOffset), cornerRadius);
669 
670  g.FillRoundRect(GetColor(kFG), handleBounds, cornerRadius);
671  }
672 
673  if(mouseOver)
674  g.FillRoundRect(GetColor(kHL), handleBounds, cornerRadius);
675 
676  if(mControl->GetAnimationFunction())
677  DrawFlashCircle(g);
678 
679  if(mDrawFrame)
680  g.DrawRoundRect(GetColor(kFR), handleBounds, cornerRadius, 0, mFrameThickness);
681 
682  return handleBounds;
683  }
