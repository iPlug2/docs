940 void DrawPressableEllipse(IGraphics& g, const IRECT& bounds, bool pressed, bool mouseOver, bool disabled)
941 {
942 IRECT handleBounds = bounds;
943 IRECT centreBounds = bounds.GetPadded(-mStyle.shadowOffset);
944 IRECT shadowBounds = bounds.GetTranslated(mStyle.shadowOffset, mStyle.shadowOffset);
945 const IBlend blend = mControl->GetBlend();
946 const float contrast = disabled ? -GRAYED_ALPHA : 0.f;
947
948 if(!pressed && !disabled && mStyle.drawShadows)
949 g.FillEllipse(GetColor(kSH), shadowBounds);
950
951 if (pressed)
952 {
953 if (mStyle.emboss)
954 {
955 shadowBounds.ReduceFromRight(mStyle.shadowOffset);
956 shadowBounds.ReduceFromBottom(mStyle.shadowOffset);
957 // Fill background with pressed color and shade it
958 g.FillEllipse(GetColor(kPR), bounds, &blend);
959 g.FillEllipse(GetColor(kSH), bounds, &blend);
960
961 // Inverse shading for recessed look - shadowBounds = inner shadow
962 g.FillEllipse(GetColor(kFG).WithContrast(contrast), shadowBounds/*, &blend*/);
963
964 // Fill in center with pressed color
965 g.FillEllipse(GetColor(kPR).WithContrast(contrast), centreBounds/*, &blend*/);
966 }
967 else
968 g.FillEllipse(GetColor(kPR).WithContrast(contrast), handleBounds/*, &blend*/);
969 }
970 else
971 {
972 // Embossed style unpressed
973 if (mStyle.emboss)
974 {
975 // Positive light TODO: use thes kPR color for now, maybe change the name?
976 g.FillEllipse(GetColor(kPR).WithContrast(contrast), bounds/*, &blend*/);
977
978 // Negative light TODO: clip this?
979 g.FillEllipse(GetColor(kSH).WithContrast(contrast), shadowBounds/*, &blend*/);
980
981 // Fill in foreground
982 g.FillEllipse(GetColor(kFG).WithContrast(contrast), centreBounds/*, &blend*/);
983
984 // Shade when hovered
985 if (mouseOver)
986 g.FillEllipse(GetColor(kHL), centreBounds, &blend);
987 }
988 else
989 {
990 g.FillEllipse(GetColor(kFG).WithContrast(contrast), handleBounds/*, &blend*/);
991
992 // Shade when hovered
993 if (mouseOver)
994 g.FillEllipse(GetColor(kHL), handleBounds, &blend);
995 }
996 }

999 DrawSplash(g, handleBounds);
1000
1001 if(mStyle.drawFrame)
