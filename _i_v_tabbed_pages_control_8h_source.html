53class IVTabbedPagesControl : public IContainerBase, public IVectorBase
54{
55public:
56 IVTabbedPagesControl(const IRECT& bounds, std::map<const char*, IVTabbedPageBase*> pages, const char* label = "",
57 const IVStyle& style = DEFAULT_STYLE.WithDrawFrame(false), float tabBarSize = 20.0f, float tabBarFrac = 0.5, EAlign tabsAlign = EAlign::Near)
58 : IContainerBase(bounds)
59 , IVectorBase(style.WithDrawShadows(false))
60 , mTabBarSize(tabBarSize)
61 , mTabBarFrac(tabBarFrac)
62 , mTabsAlign(tabsAlign)
63 {
64 AttachIControl(this, label);
65
66 for (auto page : pages)
67 {
68 AddPage(page.first, page.second);
69 }
70 }
71
72 ~IVTabbedPagesControl()
73 {
74 mPages.Empty(false);
75 }
76
77 void Draw(IGraphics& g) override
78 {
79 DrawLabel(g);
80
81 auto rcr = GetRoundedCornerRadius(GetTabBarArea());
82
83 auto rcrForTabCorners = mTabBarFrac == 1.0 ? 0.0 : rcr;
84
85 g.FillRoundRect(GetColor(kPR), GetPageArea(), mTabsAlign == EAlign::Near ? 0.0 : rcrForTabCorners,
86 mTabsAlign == EAlign::Far ? 0.0 : rcrForTabCorners,
87 rcr, rcr);
88
89 if (mStyle.drawFrame)
90 g.DrawRoundRect(GetColor(kFR), mRECT, rcr);
91 }
92
93 void OnAttached() override
94 {
95 // Set up tabs
96 AddChildControl(new IVTabSwitchControl(GetTabBarArea(),
97 [&](IControl* pCaller) {
98 ForAllPagesFunc([&](IVTabbedPageBase* pPage) {
99 bool tabForControl = strcmp(GetTabsControl()->GetSelectedLabelStr(),
100 pPage->GetLabelStr());
101 pPage->Hide(tabForControl);
102 });
103 GetUI()->GetBubbleControl()->Hide(true);
104 }, mPageNames, "", GetStyle()));
105
106 GetTabsControl()->SetShape(EVShape::EndsRounded);
107
108 // Add all pages, set their bounds and hide them
109 ForAllPagesFunc([&](IVTabbedPageBase* pPage) {
110 AddChildControl(pPage);
111 pPage->SetTargetAndDrawRECTs(GetPageArea());
112 pPage->Hide(true);
113 });
114
115 // Show the tabs and first page
116 GetTabsControl()->Hide(false);
117 GetPage(0)->Hide(false);
118 }
119
120 void OnResize() override
121 {
122 SetTargetRECT(MakeRects(mRECT));
123
124 if (NChildren())
125 {
126 GetTabsControl()->SetTargetAndDrawRECTs(GetTabBarArea());
127
128 ForAllPagesFunc([&](IVTabbedPageBase* pPage) {
129 pPage->SetTargetAndDrawRECTs(GetPageArea());
130 });
131 }
132 }
133
134 float GetTabBarSize() const { return mTabBarSize; }
135
136 IRECT GetPageArea() const { return mWidgetBounds.GetReducedFromTop(GetTabBarSize()); }
137
138 IRECT GetTabBarArea() const
139 {
140 return mWidgetBounds.GetFromTop(GetTabBarSize()).FracRectHorizontal(mTabBarFrac, mTabsAlign == EAlign::Far);
141 }
142
143 int NPages() const { return mPages.GetSize(); }
144
145private:
146 void AddPage(const char* pageName, IVTabbedPageBase* pPage)
147 {
148 pPage->SetLabelStr(pageName);
149 mPageNames.push_back(pageName);
150 mPages.Add(pPage);
151 }
152
153 void ForAllPagesFunc(std::function<void(IVTabbedPageBase* pControl)> func)
154 {
155 for (int i=0; i<mPages.GetSize(); i++)
156 {
157 func(mPages.Get(i));
158 }
159 }
160
161 IVTabSwitchControl* GetTabsControl() { return GetChild(0)->As<IVTabSwitchControl>(); }
162
163 IVTabbedPageBase* GetPage(int pageIdx) { return mPages.Get(pageIdx); }
164
165private:
166 WDL_PtrList<IVTabbedPageBase> mPages;
167 std::vector<const char*> mPageNames;
168 float mTabBarSize;
169 float mTabBarFrac;
170 EAlign mTabsAlign;
171};
