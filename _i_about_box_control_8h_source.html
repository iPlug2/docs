58 void HideAnimated(bool hide)
59 {
60 mWillHide = hide;
61
62 if (hide == false)
63 {
64 mHide = false;
65 }
66 else // hide subcontrols immediately
67 {
68 ForAllChildrenFunc([hide](int childIdx, IControl* pChild) {
69 pChild->Hide(hide);
70 });
71 }
72
73 SetAnimation([&](IControl* pCaller) {
74 auto progress = pCaller->GetAnimationProgress();
75
76 if (mWillHide)
77 SetBlend(IBlend(EBlend::Default, 1.0-progress));
78 else
79 SetBlend(IBlend(EBlend::Default, progress));
80
81 if (progress > 1.) {
82 pCaller->OnEndAnimation();
83 IPanelControl::Hide(mWillHide);
84 GetUI()->SetAllControlsDirty();
85 return;
86 }
87
88 }, mAnimationTime);
89
90 SetDirty(true);
91 }
