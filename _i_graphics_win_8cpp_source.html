65 void IGraphicsWin::CheckTabletInput(UINT msg)
66 {
67  if ((msg == WM_LBUTTONDOWN) || (msg == WM_RBUTTONDOWN) || (msg == WM_MBUTTONDOWN) || (msg == WM_MOUSEMOVE)
68  || (msg == WM_RBUTTONDBLCLK) || (msg == WM_LBUTTONDBLCLK) || (msg == WM_MBUTTONDBLCLK)
69  || (msg == WM_RBUTTONUP) || (msg == WM_LBUTTONUP) || (msg == WM_MBUTTONUP)
70  || (msg == WM_MOUSEHOVER) || (msg == WM_MOUSELEAVE))
71  {

73  const LONG_PTR c_MOUSEEVENTF_FROMTOUCH = 0xFF515700;
74 
75  LONG_PTR extraInfo = GetMessageExtraInfo();
76  SetTabletInput(((extraInfo & c_SIGNATURE_MASK) == c_MOUSEEVENTF_FROMTOUCH));
77  mCursorLock &= !mTabletInput;

441 LRESULT CALLBACK IGraphicsWin::ParamEditProc(HWND hWnd, UINT msg, WPARAM wParam, LPARAM lParam)
442 {
443  IGraphicsWin* pGraphics = (IGraphicsWin*) GetWindowLongPtr(GetParent(hWnd), GWLP_USERDATA);
444 
445  if (pGraphics && pGraphics->mParamEditWnd && pGraphics->mParamEditWnd == hWnd)
446  {
447  pGraphics->HideTooltip();
448 
449  switch (msg)
450  {
451  case WM_CHAR:
452  {
453  const IParam* pParam = pGraphics->mEdControl->GetParam();
454  // limit to numbers for text entry on appropriate parameters
455  if(pParam)
456  {
457  char c = wParam;
458 
459  if(c == 0x08) break; // backspace
460 
461  switch ( pParam->Type() )
462  {
463  case IParam::kTypeEnum:
464  case IParam::kTypeInt:
465  case IParam::kTypeBool:
466  if (c >= '0' && c <= '9') break;
467  else if (c == '-') break;
468  else if (c == '+') break;
469  else return 0;
470  case IParam::kTypeDouble:
471  if (c >= '0' && c <= '9') break;
472  else if (c == '-') break;
473  else if (c == '+') break;
474  else if (c == '.') break;
475  else return 0;
476  default:
477  break;
478  }
479  }
480  break;
481  }
482  case WM_KEYDOWN:
483  {
484  if (wParam == VK_RETURN)
485  {
486  pGraphics->mParamEditMsg = kCommit;
487  return 0;
488  }
489  else if (wParam == VK_ESCAPE)
490  {
491  pGraphics->mParamEditMsg = kCancel;
492  return 0;
493  }
494  break;
495  }
496  case WM_SETFOCUS:
497  {
498  pGraphics->mParamEditMsg = kEditing;
499  break;
500  }
501  case WM_KILLFOCUS:
502  {
503  pGraphics->mParamEditMsg = kCommit;
504  break;
505  }
506  // handle WM_GETDLGCODE so that we can say that we want the return key message
507  // (normally single line edit boxes don't get sent return key messages)
508  case WM_GETDLGCODE:
509  {
510  if (pGraphics->mEdControl->GetParam()) break;
511  LPARAM lres;
512  // find out if the original control wants it
513  lres = CallWindowProc(pGraphics->mDefEditProc, hWnd, WM_GETDLGCODE, wParam, lParam);
514  // add in that we want it if it is a return keydown
515  if (lParam && ((MSG*)lParam)->message == WM_KEYDOWN && wParam == VK_RETURN)
516  {
517  lres |= DLGC_WANTMESSAGE;
518  }
519  return lres;
520  }
521  case WM_COMMAND:
522  {
523  switch HIWORD(wParam)
524  {
525  case CBN_SELCHANGE:
526  {
527  if (pGraphics->mParamEditWnd)
528  {
529  pGraphics->mParamEditMsg = kCommit;
530  return 0;
531  }
532  }
533 
534  }
535  break; // Else let the default proc handle it.
536  }
537  }
538  return CallWindowProc(pGraphics->mDefEditProc, hWnd, msg, wParam, lParam);
539  }
540  return DefWindowProc(hWnd, msg, wParam, lParam);
541 }

661 void IGraphicsWin::MoveMouseCursor(float x, float y)
662 {
663  if (mTabletInput)
664  return;
665 
666  float scale = GetDrawScale() * GetScreenScale();
667 
668  POINT p;
669  p.x = std::round(x * scale);
670  p.y = std::round(y * scale);
671 
672  ::ClientToScreen((HWND)GetWindow(), &p);
673 
674  if (SetCursorPos(p.x, p.y))
675  {
676  GetCursorPos(&p);
677  ScreenToClient((HWND)GetWindow(), &p);
678 
679  mCursorX = p.x / scale;
680  mCursorY = p.y / scale;
681 
682  if (mCursorHidden && !mCursorLock)
683  {

685  mHiddenCursorY = p.y / scale;
686  }
687  }

1077  }
1078 
1079  if(titleWithPrefixNumbers)

1081  }

1083  }
