98bool IPluginBase::SerializeParams(IByteChunk& chunk) const
99{
100 TRACE
101 bool savedOK = true;
102 int i, n = mParams.GetSize();
103 for (i = 0; i < n && savedOK; ++i)
104 {
105 IParam* pParam = mParams.Get(i);
106 Trace(TRACELOC, "%d %s %f", i, pParam->GetName(), pParam->Value());
107 double v = pParam->Value();
108 savedOK &= (chunk.Put(&v) > 0);
109 }
110 return savedOK;

113int IPluginBase::UnserializeParams(const IByteChunk& chunk, int startPos)
114{
115 TRACE
116 int i, n = mParams.GetSize(), pos = startPos;
117 ENTER_PARAMS_MUTEX
118 for (i = 0; i < n && pos >= 0; ++i)
119 {
120 IParam* pParam = mParams.Get(i);
121 double v = 0.0;
122 pos = chunk.Get(&v, pos);
123 pParam->Set(v);
124 Trace(TRACELOC, "%d %s %f", i, pParam->GetName(), pParam->Value());
125 }
126
127 OnParamReset(kPresetRecall);
128 LEAVE_PARAMS_MUTEX
129
130 return pos;
131}

180{
181 WDL_PtrList<IParam> inParams, outParams;
182

184 {
185 IParam* pParam = GetParam(p);

195
196 assert(inParams.GetSize() == outParams.GetSize());

199 {
200 outParams.Get(p)->Set(inParams.Get(p)->Value());

284void IPluginBase::MakePreset(const char* name, ...)
285{
286 IPreset* pPreset = GetNextUninitializedPreset(&mPresets);
287 if (pPreset)
288 {
289 pPreset->mInitialized = true;
290 strcpy(pPreset->mName, name);
291
292 int i, n = NParams();
293
294 double v = 0.0;
295 va_list vp;
296 va_start(vp, name);
297 for (i = 0; i < n; ++i)
298 {
299 GET_PARAM_FROM_VARARG(GetParam(i)->Type(), vp, v);
300 pPreset->mChunk.Put(&v);
301 }
302 }

305void IPluginBase::MakePresetFromNamedParams(const char* name, int nParamsNamed, ...)
306{
307 TRACE
308 IPreset* pPreset = GetNextUninitializedPreset(&mPresets);
309 if (pPreset)
310 {
311 pPreset->mInitialized = true;
312 strcpy(pPreset->mName, name);
313
314 int i = 0, n = NParams();
315
316 WDL_TypedBuf<double> vals;
317 vals.Resize(n);
318 double* pV = vals.Get();
319 for (i = 0; i < n; ++i, ++pV)
320 {
321 *pV = PARAM_UNINIT;
322 }
323
324 va_list vp;
325 va_start(vp, nParamsNamed);
326 for (int i = 0; i < nParamsNamed; ++i)
327 {
328 int paramIdx = (int)va_arg(vp, int);
329 // This assert will fire if any of the passed-in param values do not match
330 // the type that the param was initialized with (int for bool, int, enum; double for double).
331 assert(paramIdx > kNoParameter && paramIdx < n);
332 GET_PARAM_FROM_VARARG(GetParam(paramIdx)->Type(), vp, *(vals.Get() + paramIdx));
333 }
334 va_end(vp);
335
336 pV = vals.Get();
337 for (int i = 0; i < n; ++i, ++pV)
338 {
339 if (*pV == PARAM_UNINIT) // Any that weren't explicitly set, use the defaults.
340 {
341 *pV = GetParam(i)->Value();
342 }
343 pPreset->mChunk.Put(pV);
344 }
345 }

408bool IPluginBase::RestorePreset(int idx)
409{
410 TRACE
411 bool restoredOK = false;
412 if (idx >= 0 && idx < mPresets.GetSize())
413 {
414 IPreset* pPreset = mPresets.Get(idx);
415
416 if (!(pPreset->mInitialized))
417 {
418 pPreset->mInitialized = true;
419 MakeDefaultUserPresetName(&mPresets, pPreset->mName);
420 restoredOK = SerializeState(pPreset->mChunk);
421 }
422 else
423 {
424 restoredOK = (UnserializeState(pPreset->mChunk, 0) > 0);
425 }
426
427 if (restoredOK)
428 {
429 mCurrentPresetIdx = idx;
430 OnPresetsModified();
431 OnRestoreState();
432 }
433 }
434 return restoredOK;
435}

482{
483 TRACE

485 int n = mPresets.GetSize();
486 for (int i = 0; i < n && savedOK; ++i)

492

496 savedOK &= (chunk.PutChunk(&(pPreset->mChunk)) > 0);

502int IPluginBase::UnserializePresets(const IByteChunk& chunk, int startPos)
503{
504 TRACE
505 WDL_String name;
506 int n = mPresets.GetSize(), pos = startPos;
507 for (int i = 0; i < n && pos >= 0; ++i)
508 {
509 IPreset* pPreset = mPresets.Get(i);
510 pos = chunk.GetStr(name, pos);
511 strcpy(pPreset->mName, name.Get());
512
513 Trace(TRACELOC, "%d %s", i, pPreset->mName);
514
515 pos = chunk.Get<bool>(&(pPreset->mInitialized), pos);
516 if (pPreset->mInitialized)
517 {
518 pos = UnserializeState(chunk, pos);
519 if (pos > 0)
520 {
521 pPreset->mChunk.Clear();
522 SerializeState(pPreset->mChunk);
523 }
524 }
525 }
526 RestorePreset(mCurrentPresetIdx);
527 return pos;
528}

621// confusing... IByteChunk will force storage as little endian on big endian platforms,
622// so when we use it here, since vst fxp/fxb files are big endian, we need to swap the endianess
623// regardless of the endianness of the host, and on big endian hosts it will get swapped back to
624// big endian
625bool IPluginBase::SavePresetAsFXP(const char* file) const
626{
627 if (CStringHasContents(file))
628 {
629 FILE* fp = fopen(file, "wb");
630
631 IByteChunk pgm;
632
633 int32_t chunkMagic = WDL_bswap32('CcnK');
634 int32_t byteSize = 0;
635 int32_t fxpMagic;
636 int32_t fxpVersion = WDL_bswap32(kFXPVersionNum);
637 int32_t pluginID = WDL_bswap32(GetUniqueID());
638 int32_t pluginVersion = WDL_bswap32(GetPluginVersion(true));
639 int32_t numParams = WDL_bswap32(NParams());
640 char prgName[28];
641 memset(prgName, 0, 28);
642 strcpy(prgName, GetPresetName(GetCurrentPresetIdx()));
643
644 pgm.Put(&chunkMagic);
645
646 if (DoesStateChunks())
647 {
648 IByteChunk state;
649 int32_t chunkSize;
650
651 fxpMagic = WDL_bswap32('FPCh');
652
653 IByteChunk::InitChunkWithIPlugVer(state);
654 SerializeState(state);
655
656 chunkSize = WDL_bswap32(state.Size());
657 byteSize = WDL_bswap32(state.Size() + 60);
658
659 pgm.Put(&byteSize);
660 pgm.Put(&fxpMagic);
661 pgm.Put(&fxpVersion);
662 pgm.Put(&pluginID);
663 pgm.Put(&pluginVersion);
664 pgm.Put(&numParams);
665 pgm.PutBytes(prgName, 28); // not PutStr (we want all 28 bytes)
666 pgm.Put(&chunkSize);
667 pgm.PutBytes(state.GetData(), state.Size());
668 }
669 else
670 {
671 fxpMagic = WDL_bswap32('FxCk');
672 // byteSize = WDL_bswap32(20 + 28 + (NParams() * 4) );
673 pgm.Put(&byteSize);
674 pgm.Put(&fxpMagic);
675 pgm.Put(&fxpVersion);
676 pgm.Put(&pluginID);
677 pgm.Put(&pluginVersion);
678 pgm.Put(&numParams);
679 pgm.PutBytes(prgName, 28); // not PutStr (we want all 28 bytes)
680
681 for (int i = 0; i < NParams(); i++)
682 {
683 WDL_EndianFloat v32;
684 v32.f = (float)GetParam(i)->GetNormalized();
685 unsigned int swapped = WDL_bswap32(v32.int32);
686 pgm.Put(&swapped);
687 }
688 }
689
690 fwrite(pgm.GetData(), pgm.Size(), 1, fp);
691 fclose(fp);
692
693 return true;
694 }
695 return false;
696}

698bool IPluginBase::SaveBankAsFXB(const char* file) const
699{
700 if (CStringHasContents(file))
701 {
702 FILE* fp = fopen(file, "wb");
703
704 IByteChunk bnk;
705
706 int32_t chunkMagic = WDL_bswap32('CcnK');
707 int32_t byteSize = 0;
708 int32_t fxbMagic;
709 int32_t fxbVersion = WDL_bswap32(kFXBVersionNum);
710 int32_t pluginID = WDL_bswap32(GetUniqueID());
711 int32_t pluginVersion = WDL_bswap32(GetPluginVersion(true));
712 int32_t numPgms = WDL_bswap32(NPresets());
713 int32_t currentPgm = WDL_bswap32(GetCurrentPresetIdx());
714 char future[124];
715 memset(future, 0, 124);
716
717 bnk.Put(&chunkMagic);
718
719 if (DoesStateChunks())
720 {
721 IByteChunk state;
722 int32_t chunkSize;
723
724 fxbMagic = WDL_bswap32('FBCh');
725
726 IByteChunk::InitChunkWithIPlugVer(state);
727 SerializePresets(state);
728
729 chunkSize = WDL_bswap32(state.Size());
730 byteSize = WDL_bswap32(160 + state.Size());
731
732 bnk.Put(&byteSize);
733 bnk.Put(&fxbMagic);
734 bnk.Put(&fxbVersion);
735 bnk.Put(&pluginID);
736 bnk.Put(&pluginVersion);
737 bnk.Put(&numPgms);
738 bnk.Put(&currentPgm);
739 bnk.PutBytes(&future, 124);
740
741 bnk.Put(&chunkSize);
742 bnk.PutBytes(state.GetData(), state.Size());
743 }
744 else
745 {
746 fxbMagic = WDL_bswap32('FxBk');
747
748 bnk.Put(&byteSize);
749 bnk.Put(&fxbMagic);
750 bnk.Put(&fxbVersion);
751 bnk.Put(&pluginID);
752 bnk.Put(&pluginVersion);
753 bnk.Put(&numPgms);
754 bnk.Put(&currentPgm);
755 bnk.PutBytes(&future, 124);
756
757 int32_t fxpMagic = WDL_bswap32('FxCk');
758 int32_t fxpVersion = WDL_bswap32(kFXPVersionNum);
759 int32_t numParams = WDL_bswap32(NParams());
760
761 for (int p = 0; p < NPresets(); p++)
