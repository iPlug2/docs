325 void IPluginBase::MakePresetFromNamedParams(const char* name, int nParamsNamed, ...)
326 {
327  TRACE;
328  IPreset* pPreset = GetNextUninitializedPreset(&mPresets);
329  if (pPreset)
330  {
331  pPreset->mInitialized = true;
332  strcpy(pPreset->mName, name);
333 
334  int i = 0, n = NParams();
335 
336  WDL_TypedBuf<double> vals;
337  vals.Resize(n);
338  double* pV = vals.Get();
339  for (i = 0; i < n; ++i, ++pV)
340  {
341  *pV = PARAM_UNINIT;
342  }
343 
344  va_list vp;
345  va_start(vp, nParamsNamed);
346  for (int i = 0; i < nParamsNamed; ++i)
347  {
348  int paramIdx = (int) va_arg(vp, int);
349  // This assert will fire if any of the passed-in param values do not match
350  // the type that the param was initialized with (int for bool, int, enum; double for double).
351  assert(paramIdx >= 0 && paramIdx < n);
352  GET_PARAM_FROM_VARARG(GetParam(paramIdx)->Type(), vp, *(vals.Get() + paramIdx));
353  }
354  va_end(vp);
355 
356  pV = vals.Get();
357  for (int i = 0; i < n; ++i, ++pV)
358  {
359  if (*pV == PARAM_UNINIT) // Any that weren't explicitly set, use the defaults.
360  {
361  *pV = GetParam(i)->Value();
362  }
363  pPreset->mChunk.Put(pV);
364  }
365  }

652 // confusing... IByteChunk will force storage as little endian on big endian platforms,
653 // so when we use it here, since vst fxp/fxb files are big endian, we need to swap the endianess
654 // regardless of the endianness of the host, and on big endian hosts it will get swapped back to
655 // big endian
656 bool IPluginBase::SaveProgramAsFXP(const char* file) const
657 {
658  if (CStringHasContents(file))
659  {
660  FILE* fp = fopen(file, "wb");
661 
662  IByteChunk pgm;
663 
664  int32_t chunkMagic = WDL_bswap32('CcnK');
665  int32_t byteSize = 0;
666  int32_t fxpMagic;
667  int32_t fxpVersion = WDL_bswap32(kFXPVersionNum);
668  int32_t pluginID = WDL_bswap32(GetUniqueID());
669  int32_t pluginVersion = WDL_bswap32(GetPluginVersion(true));
670  int32_t numParams = WDL_bswap32(NParams());
671  char prgName[28];
672  memset(prgName, 0, 28);
673  strcpy(prgName, GetPresetName(GetCurrentPresetIdx()));
674 
675  pgm.Put(&chunkMagic);
676 
677  if (DoesStateChunks())
678  {
679  IByteChunk state;
680  int32_t chunkSize;
681 
682  fxpMagic = WDL_bswap32('FPCh');
683 
684  IByteChunk::InitChunkWithIPlugVer(state);
685  SerializeState(state);
686 
687  chunkSize = WDL_bswap32(state.Size());
688  byteSize = WDL_bswap32(state.Size() + 60);
689 
690  pgm.Put(&byteSize);
691  pgm.Put(&fxpMagic);
692  pgm.Put(&fxpVersion);
693  pgm.Put(&pluginID);
694  pgm.Put(&pluginVersion);
695  pgm.Put(&numParams);
696  pgm.PutBytes(prgName, 28); // not PutStr (we want all 28 bytes)
697  pgm.Put(&chunkSize);
698  pgm.PutBytes(state.GetData(), state.Size());
699  }
700  else
701  {
702  fxpMagic = WDL_bswap32('FxCk');
703  //byteSize = WDL_bswap32(20 + 28 + (NParams() * 4) );
704  pgm.Put(&byteSize);
705  pgm.Put(&fxpMagic);
706  pgm.Put(&fxpVersion);
707  pgm.Put(&pluginID);
708  pgm.Put(&pluginVersion);
709  pgm.Put(&numParams);
710  pgm.PutBytes(prgName, 28); // not PutStr (we want all 28 bytes)
711 
712  for (int i = 0; i< NParams(); i++)
713  {
